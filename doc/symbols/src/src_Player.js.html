 13  *
 14  * @class Kaikki pelaajat, niin ihmispelaajat kuin botitkin, ovat tämän luokan jäseniä.
 15  *
 16  * @param {Object} [attr]  Objekti joka sisältää joitain valmiiksi asettuja arvoja uudelle pelaajalle.

 32  * @property {Integer} rightLegObj   Oikea jalka
 33  * @property {Integer} fireObj       Suuliekkianimaatio

 35  * @property {Integer} showFire      Näytetäänkö suuliekki
 36  * @property {Float}   prevPosX      Pelaajan edellinen sijainti
 37  * @property {Float}   prevPosY      Pelaajan edellinen sijainti
 38  * @property {Float}   legPos        Jalkojen asento kävelyanimaatiossa

 41  * @property {Integer} weapon        Käytössä oleva ase
 42  * @property {Integer} lastShoot     Koska on viimeksi ammuttu
 43  * @property {Integer} lastSound     Koska on viimeksi soitettu ampumisääni (käytetään moottorisahassa)
 44  * @property {Integer} health        Terveys
 45  * @property {Integer} prevHealth    Terveys edellisellä kierroksella
 46  * @property {Integer} kills         Tappojen lukumäärä
 47  * @property {Integer} deaths        Kuolemien lukumäärä

 49  * @property {Integer} death         Onko pelaaja kuollut

 51  * @property {Integer} zombie        Onko tämä pelaaja botti

 54  * @property {Float}   rotation      Tällä käännetään objektia (TurnObject)
 55  * @property {Float}   sideStep      Botin sivuaskeleet
 56  * @property {integer} nextAction    Ajankohta jolloin botille arvotaan uusi suunta
 57  * @property {Integer} tooClose      Botti on liian lähellä seinää tai toista pelaajaa
 58  * @property {Float}   fightRotate   Kääntymisnopeus kun ollaan havaittu vastustaja
 59  * @property {Float}   shootingAngle Ammutaan kun uhri on tämän kulman sisällä botin edessä
 60  * @property {Integer} shootedBy     Kuka on viimeksi ampunut pelaajaa
 61  * @property {Float}   fov           Field of View
 62  * @property {Integer} idleSound     Esim. moottorisahan tyhjäkäynti
 63  * @property {Integer} idleTimer     Koska soitettu viimeksi
 64  * @property {Integer} hasAmmos      Onko pelaajalla ammuksia nykyisessä aseessa
 65  * @property {Byte}    team          Joukkue
 66  * @property {String}  mapName       Pelaajalla ladattuna oleva kartta
 67  * @property {Integer} outOfMap      Kuinka monta päivityskierrosta pelaaja on ollut poissa kartalta
 68  * @property {Integer} spawnTime     Syntymäaika
 69  * @property {Integer} visible       Ukon vilkuttaminen syntymän jälkeen
 70  * @property {Byte}    admin         Onko pelaaja admin vai ei
 71  * @property {Byte}    kicked        Onko pelaaja potkittu pois
 72  * @property {String}  kickReason    Mikä syy on annettu potkuille
 73  * @property {Byte}    kickerId      Kuka potkaisi
 74  * @property {Integer} hackTestX     Pelaajan viimeisin sijainti nopeuden tarkistuksessa
 75  * @property {Integer} hackTestY     Pelaajan viimeisin sijainti nopeuden tarkistuksessa
 76  * @property {Integer} lag           Pelaajan lagi millisekunneissa
 77  * @property {Integer} spHackTimer   Viimeisimmässä nopeuden tarkistuksessa otettu Timer()
 78  * @property {Byte}    handShooted   Kummalla kädellä on viimeksi ammuttu (pistooli) 0=vasen 1=oikea
 79  */
 80 function Player(attr) {
 81   // Jeejee, luodaan client.
 82   for (var prop in attr) {
 83     if (attr.hasOwnProperty(prop)) {
 84       this[prop] = attr.prop;
 85     }
 86   }
 87 }
