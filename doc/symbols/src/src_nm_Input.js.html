  4 
  5 /**#nocode+*/
  6 var log = require('./Utils').log
  7   , NET = require('./Constants').NET
  8   , colors = require('colors');
  9 /**#nocode-*/
 10 
 11 function Input(server) {
 12   this.server = server;
 13 
 14   process.stdin.resume();
 15   process.stdin.setEncoding('utf8');
 16 
 17   process.on('SIGINT', function () {
 18     // Suljetaan palvelin kun tulee SIGINT
 19     console.log();
 20     log.warn('Received SIGINT');
 21     server.close();
 22   });
 23 
 24   server.on('closed', function () {
 25     // Nyt ollaan suljettu, voidaan sulkea koko prosessi. Mutta odotetaan toki puoli sekuntia,
 26     // että muut funktiot jotka kuuntelevat closed-eventtiä voisivat toimia.
 27     process.stdin.destroySoon();
 28   });
 29 
 30 
 31   process.stdin.on('data', function (chunk) {
 32     // Lähetetään data pelaajille
 33     chunk = chunk.trim();
 34 
 35     log.write('<Server>'.blue + ' %0', chunk);
 36 
 37     var msgData = {
 38       msgType: NET.SERVERMSG,
 39       msgText: chunk
 40     };
 41 
 42     var playerIds = Object.keys(server.players);
 43     for (var i = playerIds.length; i--;) {
 44       var plr = server.players[playerIds[i]];
 45       if (plr.active && !plr.zombie) {
 46         server.messages.add(plr.playerId, msgData);
 47       }
 48     }
 49   });
 50 }
 51 
 52 exports = module.exports = Input;
