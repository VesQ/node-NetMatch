  5 "use strict";
  6 
  7 /**#nocode+*/
  8 var log = require('./Utils').log
  9   , NET = require('./Constants').NET
 10   , colors = require('colors')
 11   , tty = require('tty')
 12   , readline = require('readline')
 13   , splitString = require('./Utils').splitString;
 14 /**#nocode-*/
 15 
 16 /**
 17  * @class Hoitaa konsoliin liittyvät komennot ja toiminnallisuudet.
 18  *
 19  * @param {Server} server  NetMatch-palvelin, johon tämä instassi kuuluu
 20  */
 21 function Input(server) {
 22   // Konstruktorin sisältöä ei dokumentoida
 23   /**#nocode+*/
 24 
 25   var self = this
 26     , rli = readline.createInterface(process.stdin, process.stdout, function (partial) {
 27       return [server.commands.suggest(partial), partial];
 28     });
 29 
 30   process.stdin.resume();
 31   process.stdin.setEncoding('utf8');
 32 
 33   if (process.platform !== "win32") {
 34     process.on('SIGINT', function processSIGINT() {
 35       log.notice('Received ' + 'SIGINT'.red);
 36       // Lakataan kuuntelemasta stdin-syötettä.
 37       // Node sulkeutuu jahka kaikki eventit on kutsuttu, eikä uusia ole lisätty event-luuppiin.
 38       process.stdin.destroy();
 39       server.close();
 40     });
 41   }
 42 
 43   rli.on('close', function rliClose() {
 44     process.stdin.destroy();
 45     server.close();
 46   });
 47 
 48   rli.on('line', function rliLine(input){
 49     var msg = input.trim();
 50     if (!msg) { return; }
 51     // Käsitellään serverikomennot
 52     server.commands.call(msg.split(' ')[0], splitString(msg).splice(1));
 53   });
 54 
 55   // Tästä eteenpäin dokumentoidaan taas jos on jotain dokumentoitavaa
 56   /**#nocode-*/
 57 }
 58 
 59 module.exports = Input;
