169 /**
170  * Lataa asetukset annetusta tiedostosta. Tämä metodi hakee tiedostoa node-NetMatchin juuresta.
171  *
172  * @param {String} [config="config"]  Asetustiedoston nimi, ilman .json päätettä.
173  *
174  * @returns {Boolean}  Pystytäänkö peliä jatkamaan nykyisillä asetuksilla
175  */
176 Config.prototype.load = function (config) {
177   if (config.substr(-5) !== '.json') {
178     config += '.json';
179   }
180   var filePath = path.resolve(__dirname, '..', config)
181     , loadedConfig
182     , i, mapPath;
183 
184   if (!path.existsSync(filePath)) {
185     log.error('Config %0 doesn\'t exist', filePath.green);
186     log.error('Use `%0` to load your own config', 'netmatch -c /path/to/your/config.json'.yellow);
187     return true;
188   }
189 
190   log.info('Loading config from %0', filePath.green);
191   loadedConfig = cjson.load(filePath);
192   // Laajennetaan tämän Configin ominaisuuksia ladatulla json-tiedostolla
193   cjson.extend(this, loadedConfig);
194 
195   // Tarkistetaan onko maksimimäärä liian suuri ja kaadetaan palvelin mikäli on
196   if (this.maxPlayers > 64) {
197     log.fatal('maxPlayers set to %0 while the hardcoded maximum is %1!', String(this.maxPlayers).magenta, 64);
198     return false;
199   }
200 
201   // Tarkistetaan onko kartta merkkijono. Jos se on merkkijono, niin muutetaan se listaksi jossa
202   // on vain yksi arvo.
203   if ('string' === typeof this.map) {
204     this.map = [this.map];
205   }
206 
207   // Tarkistetaan että kartta todellakin on lista tässä vaiheessa
208   if (!Array.isArray(this.map)) {
209     log.fatal('Invalid value for "map" in config %0', filePath.green);
210     return false;
211   }
212 
213   // Tarkistetaan onko määritellyt kartat olemassa
214   for (i = 0; i < this.map.length; i++) {
215     mapPath = path.resolve(__dirname, '..', 'maps', this.map[i] + '.json');
216 
217     if (!path.existsSync(mapPath)) {
218       log.fatal('Map %0 doesn\'t exist in %1', this.map[i].green, mapPath.green);
219       return false;
220     }
221   }
222 
223   if (this.mapDownloadUrl[this.mapDownloadUrl.length - 1] !== '/') {
224     this.mapDownloadUrl = this.mapDownloadUrl + '/';
225   }
226 
227   // Tarkistetaan että portti on numero
228   if (this.port && 'number' !== typeof this.port) {
229     log.fatal('You must set %0 to be a number in %1!', 'port'.yellow, filePath.green);
230     return false;
231   }
232 
233   return true;
234 };
