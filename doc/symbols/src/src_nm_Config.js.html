111 /**
112  * Lataa asetukset annetusta tiedostosta. Tämä metodi hakee tiedostoa node-NetMatchin juuresta.
113  *
114  * @param {String} [config="config"]  Asetustiedoston nimi, ilman .json päätettä.
115  */
116 Config.prototype.load = function (config) {
117   var filePath = path.resolve(__dirname, '..', config + '.json')
118     , loadedConfig;
119 
120   if (!path.existsSync(filePath)) {
121     log.error('Config %0 doesn\'t exist', filePath.green);
122     return;
123   }
124 
125   log.info('Loading config from %0', filePath.green);
126   loadedConfig = cjson.load(filePath);
127   // Laajennetaan tämän Configin ominaisuuksia ladatulla json-tiedostolla
128   cjson.extend(this, loadedConfig);
129 };
