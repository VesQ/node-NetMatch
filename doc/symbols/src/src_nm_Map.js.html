 22 /**
 23  * Lataa uuden kartan.
 24  * @class Karttaluokka
 25  *
 26  * @param {Server} server  Tämän kartan NetMatch-palvelin
 27  * @param {String} name    Kartan nimi, joka ladataan (esim. Luna)
 28  *
 29  * @property {Boolean} loaded                 Ladattiinko kartta onnistuneesti
 30  * @property {String}  name                   Kartan nimi
 31  * @property {String}  author                 Kartan tekijä
 32  * @property {String}  crc32                  Kartan CRC32-tunniste, jota tarvitaan CoolBasicin puolella
 33  * @property {Object}  config                 Kartan asetukset sisältävä objekti
 34  * @property {Integer} config.maxPlayers      Maksimipelaajat
 35  * @property {Integer} config.botCount        Bottien määrä
 36  * @property {Integer} config.botDepartLimit  Kun pelaajamäärä ylittää tämän arvon, poistuu botti pelistä
 37  * @property {Array}   config.botNames        Bottien nimet
 38  * @property {Array}   config.botWeapons      Bottien aseet (1 = pistooli, 2 = konepistooli, 3 = sinko,
 39  *                                            4 = haulikko, 5 = kranaatinlaukaisin, 6 = moottorisaha)
 40  * @property {Integer} config.healthItems     Lääkintäpakkausten määrä
 41  * @property {Integer} config.mgunItems       Konekiväärin ammuslaatikoiden määrä
 42  * @property {Integer} config.bazookaItems    Singon rakettilaatikoiden määrä
 43  * @property {Integer} config.shotgunItems    Haulikon haulilaatikoiden määrä
 44  * @property {Integer} config.launcherItems   Kranaatinlaukaisimen kranaattilaatikoiden määrä
 45  * @property {Integer} config.chainsawItems   Moottirisahan bensakanistereiden määrä
 46  * @property {Integer} tileSize               Yhden tilen leveys/korkeus pikseleissä
 47  * @property {Integer} width                  Kartan leveys
 48  * @property {Integer} height                 Kartan korkeus
 49  * @property {Array}   data                   Törmäyskerroksen data yksiulotteisessa taulukossa
 50 */
 51 function Map(server, name) {
 52   var filePath = path.resolve(__dirname, '..', 'maps', name + '.json')
 53     , data, buffer, uint8View;
 54 
 55   this.server = server;
 56   this.loaded = false;
 57 
 58   if (!path.existsSync(filePath)) {
 59     log.error('Map %0 doesn\'t exist in %1', name.green, filePath.green);
 60     return;
 61   }
 62 
 63   log.info('Loading map %0 from %1', name.green, filePath.green);
 64   data = cjson.load(filePath);
 65   // Laajennetaan tämän kartan ominaisuuksia ladatulla json-tiedostolla
 66   cjson.extend(this, data);
 67 
 68   // Muunnetaan törmäyskerroksen data 8-bittiseksi, yksiulotteiseksi taulukoksi
 69   buffer = new ArrayBuffer(this.width * this.height);
 70   uint8View = new Uint8Array(buffer);
 71   for (var y = 0; y < this.height; ++y) {
 72     for (var x = 0; x < this.width; ++x) {
 73       var index = y * this.width + x;
 74       uint8View[index] = this.data[y][x];
 75     }
 76   }
 77   this.data = uint8View;
 78 
 79   this.loaded = true;
 80 }

 88 Map.prototype.isColliding = function (x, y) {
 89   var tileX = Math.ceil((x + this.width * this.tileSize / 2) / this.tileSize) - 1
 90     , tileY = Math.ceil((-y + this.height * this.tileSize / 2) / this.tileSize) - 1
 91     , index;
 92 
 93   if (tileX < 0 || tileX >= this.width || tileY < 0 || tileY >= this.height) {
 94     // Ollaan kartan ulkopuolella, eli törmätään.
 95     return true;
 96   }
 97 
 98   // Lasketaan tilen paikka yksiulotteisessa taulukossa
 99   index = tileY * this.width + tileX;
100 
101   if (this.server.debug) {
102     if ('undefined' === typeof this.data || 'undefined' === typeof this.data[index]) {
103       log.error('Map.data[%0] (%1, %2) is undefined', index, tileY, tileX);
104     }
105   }
106 
107   // Nyt tarkistetaan että ollaanko seinän sisällä
108   if (this.data[index]) {
109     return true;
110   }
111 
112   // Jos päästiin tänne asti niin ei olla törmätty
113   return false;
114 };

116 /**
117  * Etsii satunnaisen paikan kartalta, joka ei ole seinän sisällä, ja palauttaa kyseisen paikan
118  * koordinaatit objektissa, jolla on kentät x ja y.
119  *
120  * @returns {Object}  Objekti, jolla on kentät x ja y, jotka ovat löydetyn paikan koordinaatit
121  */
122 Map.prototype.findSpot = function () {
123   var randTileX, randTileY, index, returnObj = {};
124   // Etsitään vapaata paikkaa kartalta "vain" 10 000 kertaa
125   for (var i = 9999; --i;) {
126     randTileX = rand(0, this.width - 1);
127     randTileY = rand(0, this.height - 1);
128     // Lasketaan tilen paikka yksiulotteisessa taulukossa
129     index = randTileY * this.width + randTileX;
130 
131     if (!this.data[index]) {
132       // Ei ollut seinän sisällä
133       returnObj.x = (randTileX * this.tileSize)
134                   - (this.width * this.tileSize) / 2
135                   + this.tileSize / 2;
136       returnObj.y = -((randTileY * this.tileSize)
137                   - (this.height * this.tileSize) / 2
138                   + this.tileSize / 2);
139       break;
140     }
141   }
142   return returnObj;
143 };

193 /**
194  * Etsii annettujen koordinaattien ja kulman muodostavalta suoralta annetulta maksimietäisyydeltä
195  * seinän koordinaatit ja palauttaa ne objektin kenttinä x ja y. Jos seinää ei löytynyt ennen
196  * maksimietäisyyden saavuttamista, ei funktio palauta mitään arvoa.
197  *
198  * @param {Number} x       Aloituspisteen x-koordinaatti maailmankoordinaateissa
199  * @param {Number} y       Aloituspisteen y-koordinaatti maailmankoordinaateissa
200  * @param {Number} angle   Mihin suuntaan suora muodostuu (asteina)
201  * @param {Number} [dist]  Kuinka pitkältä katsotaan. Jos tätä ei anneta, niin etäisyyttä ei
202  *                         rajoiteta.
203  *
204  * @returns {Object}  Palauttaa objektin jolla on kentät x ja y jotka sisältävät maksimietäisyyden
205  *                    sisältä löytyneen seinän törmäyskoordinaatit. Tätä ei palauteta, jos ei
206  *                    löytynyt seinää.
207  */
208 Map.prototype.findWall = function (x, y, angle, dist) {
209   var startP = {}, endP = {}, returnP;
210 
211   // Tarkistetaan etäisyys
212   if ('number' !== typeof dist) {
213     dist = (this.width + this.height) * this.tileSize;
214   }
215 
216   // Muunnetaan x ja y maailmankoordinaateista "näyttökoordinaateiksi"
217   startP.x = x + this.width * this.tileSize / 2;
218   startP.y = -y + this.height * this.tileSize / 2;
219 
220   // Lasketaan loppupiste
221   endP.x = startP.x + Math.cos((-angle / 180) * Math.PI) * dist;
222   endP.y = startP.y + Math.sin((-angle / 180) * Math.PI) * dist;
223 
224   // Suoritetaan raycast
225   returnP = this.rayCast(startP, endP);
226 
227   if (!returnP) {
228     // Välissä ei ollut seinää, ei palauteta mitään.
229     //console.log("No hits with raycast");
230     return;
231   }
232 
233   // Muunnetaan "näyttökoordinaatit" maailmankoordinaateiksi
234   returnP.x = returnP.x - (this.width * this.tileSize / 2);
235   returnP.y = (this.height * this.tileSize / 2) - returnP.y;
236 
237   // Palautetaan piste
238   return returnP;
239 };

279 /**
280  * Tekee raycastin kahden pisteen välillä ja palauttaa kohdan, jossa pisteiden välillä kulkeva
281  * suora osuu ensimmäisen kerran seinään.
282  * @private
283  *
284  * @param {Object} origP1  Alkupiste, jolla on kentissä x ja y koordinaatit
285  * @param {Object} origP2  Loppupiste, jolla on kentissä x ja y koordinaatit
286  *
287  * @returns {Object}  Ensimmäinen vastaan tullut seinä ja tarkka osumakohta. Jos pisteiden välillä
288  *                    ei ollut seinää, ei funktio palauta mitään.
289  */
290 Map.prototype.rayCast = function (origP1, origP2) {
291   // Pisteiden normalisaatio
292   var p1 = {x: origP1.x / this.tileSize, y: origP1.y / this.tileSize}
293     , p2 = {x: origP2.x / this.tileSize, y: origP2.y / this.tileSize}
294     // Seuraavien muuttujien kommentointi löytyy myöhemmin funktiossa
295     , stepX, stepY
296     , rayDirX, rayDirY
297     , ratioX, ratioY
298     , deltaX, deltaY
299     , testTile
300     , maxX, maxY
301     , endTile
302     , hit
303     , colP
304     , index;
305 
306   // Ylitetäänkö minkään laatan rajoja
307   if (truncateNumber(p1.x) === truncateNumber(p2.x) && truncateNumber(p1.y) === truncateNumber(p2.y)) {
308     // Ei ylitä minkään laatan rajoja, joten ei voi olla törmäystä.
309     return;
310   }
311 
312   // Kumpaan suuntaan mennään x- ja y-suunnassa
313   stepX = (p2.x > p1.x) ? 1 : -1;
314   stepY = (p2.y > p1.y) ? 1 : -1;
315 
316   // Säteen suunta
317   rayDirX = p2.x - p1.x;
318   rayDirY = p2.y - p1.y;
319 
320   // Kuinka pitkälle liikutaan kummallakin akselilla kun toisella akselilla hypätään seuraavaan
321   // kokonaiseen tileen
322   ratioX = rayDirX / rayDirY;
323   ratioY = rayDirY / rayDirX;
324 
325   // Muutos x- ja y-suunnassa
326   deltaY = Math.abs(p2.x - p1.x);
327   deltaX = Math.abs(p2.y - p1.y);
328 
329   // Alustetaan testiä varten käytettävät kokonaislukumuuttujat alkutilekoordinaatteihin
330   // Huom: Käytetään normalisoituja versioita pisteestä origP1
331   testTile = {x: truncateNumber(p1.x), y: truncateNumber(p1.y)};
332 
333   // Alustetaan ei-kokonaislukuhyppäys liikkumalla seuraavan tilen reunalle ja jakamalla saatu
334   // arvo vastakkaisen akselin kokonaisluvulla.
335   // Jos liikutaan positiiviseen suuntaan, siirrytään nykyisen tilen päähän, muulloin alkuun.
336   if (stepX > 0) {
337     maxX = deltaX * (1.0 - (p1.x % 1));
338   } else {
339     maxX = deltaX * (p1.x % 1);
340   }
341   if (stepY > 0) {
342     maxY = deltaY * (1.0 - (p1.y % 1));
343   } else {
344     maxY = deltaY * (p1.y % 1);
345   }
346 
347   // Lopputile
348   endTile = {x: truncateNumber(p2.x), y: truncateNumber(p2.y)};
349 
350   // Nyt liikutaan!
351   hit = 0;
352   colP = {x: 0.0, y: 0.0};
353 
354   while (testTile.x !== endTile.x || testTile.y !== endTile.y) {
355     if (maxX < maxY) {
356       maxX += deltaX;
357       testTile.x += stepX;
358 
359       if (testTile.x < 0 || testTile.x >= this.width || testTile.y < 0 || testTile.y >= this.height) {
360         // Ollaan kartan ulkopuolella, eli törmätään.
361         hit = 1;
362       } else {
363         // Lasketaan tilen paikka yksiulotteisessa taulukossa
364         index = testTile.y * this.width + testTile.x;
365         // Tarkistetaan onko tilekerroksessa hit-dataa
366         hit = this.data[index];
367       }
368       if (hit) {
369         // Raycast löysi törmäyksen
370         colP.x = testTile.x;
371         if (stepX < 0) { colP.x += 1.0; } // Jos mennään vasemmalle päin, lisätään yksi.
372         colP.y = p1.y + ratioY * (colP.x - p1.x);
373         colP.x *= this.tileSize; // Skaalataan törmäyspiste ylöspäin
374         colP.y *= this.tileSize;
375         // Palautetaan osumapiste
376         return colP;
377       }
378     } else {
379       maxY += deltaY;
380       testTile.y += stepY;
381 
382       if (testTile.x < 0 || testTile.x >= this.width || testTile.y < 0 || testTile.y >= this.height) {
383         // Ollaan kartan ulkopuolella, eli törmätään.
384         hit = 0;
385       } else {
386         // Lasketaan tilen paikka yksiulotteisessa taulukossa
387         index = testTile.y * this.width + testTile.x;
388         // Tarkistetaan onko tilekerroksessa hit-dataa
389         hit = this.data[index];
390       }
391       if (hit) {
392         // Raycast löysi törmäyksen
393         colP.y = testTile.y;
394         if (stepY < 0) { colP.y += 1.0; } // Jos mennään ylöspäin, lisätään yksi.
395         colP.x = p1.x + ratioX * (colP.y - p1.y);
396         colP.x *= this.tileSize; // Skaalataan törmäyspiste ylöspäin
397         colP.y *= this.tileSize;
398         // Palautetaan osumapiste
399         return colP;
400       }
401     }
402   }
403 
404   // Jos tänne asti ollaan päästy, niin törmäystä ei ole löydetty. Ei siis palauteta mitään.
405 };

408 /**
409  * Ympyrä-tilekartta törmäys.
410  *
411  * @see http://stackoverflow.com/a/402010/1152564
412  *
413  * @param {Number} x  Ympyrän keskipisteen x-koordinaatti maailmankoordinaateissa
414  * @param {Number} y  Ympyrän keskipisteen y-koordinaatti maailmankoordinaateissa
415  * @param {Number} r  Ympyrän säteen pituus
416  *
417  * @return {Object}  Objektilla on kentät "left", "right", "up" ja/tai "down". Jos kentän arvo on
418  *                   tosi, niin siitä suunnasta löytyy törmäys.
419  */
420 Map.prototype.circleCollision = function (x, y, r) {
421   var tileX, tileY, leftTileX, rightTileX, upTileY, downTileY
422     , circleDistance
423     , ret = {left: false, right: false, up: false, down: false};
424 
425 
426   // Muunnetaan x ja y maailmankoordinaateista "näyttökoordinaateiksi"
427   x = x + this.width * this.tileSize / 2;
428   y = -y + this.height * this.tileSize / 2;
429 
430   // Otetaan ylös tilekoordinaatit nykyisestä tilestä sekä viereisistä
431   tileX = Math.floor(x / this.tileSize);
432   tileY = Math.floor(y / this.tileSize);
433   leftTileX = tileX - 1;
434   rightTileX = tileX + 1;
435   upTileY = tileY - 1;
436   downTileY = tileY + 1;
437 
438   // Tarkistetaan törmäykset kaikkiin nykyistä tileä ympäröiviin kahdeksaan tileen.
439   // Aloitetaan vasemmalta ylhäältä ja mennään vasemmalta oikealla, ylhäältä alas.
440   if (this.checkCircleCollision(x, y, r, leftTileX, upTileY)) {
441     ret.left = true;
442     ret.up = true;
443   }
444   if (this.checkCircleCollision(x, y, r, tileX, upTileY)) {
445     ret.up = true;
446   }
447   if (this.checkCircleCollision(x, y, r, rightTileX, upTileY)) {
448     ret.right = true;
449     ret.up = true;
450   }
451   if (this.checkCircleCollision(x, y, r, leftTileX, tileY)) {
452     ret.left = true;
453   }
454   if (this.checkCircleCollision(x, y, r, rightTileX, tileY)) {
455     ret.right = true;
456   }
457   if (this.checkCircleCollision(x, y, r, leftTileX, downTileY)) {
458     ret.left = true;
459     ret.down = true;
460   }
461   if (this.checkCircleCollision(x, y, r, tileX, downTileY)) {
462     ret.down = true;
463   }
464   if (this.checkCircleCollision(x, y, r, rightTileX, downTileY)) {
465     ret.right = true;
466     ret.down = true;
467   }
468 
469   // Suoritetaan tarkastus vielä nykyiseen tileen
470   if (this.checkCircleCollision(x, y, r, tileX, tileY)) {
471     ret.right = true;
472     ret.left = true;
473     ret.up = true;
474     ret.down = true;
475   }
476   return ret;
477 };

479 /**
480  * @private
481  */
482 Map.prototype.checkCircleCollision = function (x, y, r, tileX, tileY) {
483   var hit, index, halfTile, circleDist, cornedDist_sq;
484 
485   if (tileX < 0 || tileY < 0 || tileX >= this.width || tileY >= this.height) {
486     // Jos tilekoordinaatti on kartan rajojen ulkopuolella, lasketaan törmäys siihen.
487     hit = 1;
488   } else {
489     // Muulloin tarkastetaan törmäysdata tiledatasta.
490     index = tileY * this.width + tileX;
491     hit = this.data[index];
492   }
493 
494   // Jos tilessä ei ole törmäystä asetettuna, voidaan tarkastus jättää jo tässä vaiheessa sikseen.
495   if (!hit) {
496     return false;
497   }
498 
499   // Alustetaan joitakin muuttujia
500   circleDist = {};
501   halfTile = this.tileSize / 2;
502 
503   // Tästä lähtee http://stackoverflow.com/a/402010/1152564
504   circleDist.x = Math.abs(x - (tileX * this.tileSize) - halfTile);
505   circleDist.y = Math.abs(y - (tileY * this.tileSize) - halfTile);
506 
507   if (circleDist.x > (halfTile + r)) { return false; }
508   if (circleDist.y > (halfTile + r)) { return false; }
509 
510   if (circleDist.x <= halfTile) { this.debugBox(tileX, tileY); return true; }
511   if (circleDist.y <= halfTile) { this.debugBox(tileX, tileY); return true; }
512 
513   /*
514   cornedDist_sq = Math.pow(circleDist.x - halfTile, 2) + Math.pow(circleDist.y - halfTile, 2);
515 
516   if (cornedDist_sq <= Math.pow(r, 2)) {
517     this.debugBox(tileX, tileY);
518     return true;
519   }
520   */
521 
522   return false;
523 };

525 /**
526  * Piirtää boksin tilen paikalle.
527  * @param {Integer} tileX
528  * @param {Integer} tileY
529  * @param {Boolean} [fill=false]  Piirretäänkö täytetty boksi
530  */
531 Map.prototype.debugBox = function (tileX, tileY, fill) {
532   var boxX, boxY, self = this;
533   if (!this.server.debug) { return; }
534   if ('boolean' !== typeof fill) { fill = false; }
535 
536   boxX = tileX * this.tileSize - (this.width * this.tileSize / 2);
537   boxY = (this.height * this.tileSize / 2) - tileY * this.tileSize;
538 
539   this.server.loopPlayers(function (player) {
540     if (player.debugState && player.debugState < 20) {
541       player.debugState += 1;
542       self.server.messages.add(player.id, {
543         msgType: NET.DEBUGDRAWING,
544         drawType: DRAW.BOX,
545         drawVars: [
546           boxX,
547           boxY,
548           self.tileSize,
549           self.tileSize,
550           fill
551         ]
552       });
553     }
554   });
555 };
