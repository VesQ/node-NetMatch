 17 /**
 18  * Lataa uuden kartan.
 19  * @class Karttaluokka
 20  *
 21  * @param {Server} server  Tämän kartan NetMatch-palvelin
 22  * @param {String} name    Kartan nimi, joka ladataan (esim. Luna)
 23  *
 24  * @property {Boolean} loaded                 Ladattiinko kartta onnistuneesti
 25  * @property {String}  name                   Kartan nimi
 26  * @property {String}  author                 Kartan tekijä
 27  * @property {String}  crc32                  Kartan CRC32-tunniste, jota tarvitaan CoolBasicin puolella
 28  * @property {Object}  config                 Kartan asetukset sisältävä objekti
 29  * @property {Integer} config.maxPlayers      Maksimipelaajat
 30  * @property {Integer} config.botCount        Bottien määrä
 31  * @property {Integer} config.botDepartLimit  Kun pelaajamäärä ylittää tämän arvon, poistuu botti pelistä
 32  * @property {Array}   config.botNames        Bottien nimet
 33  * @property {Array}   config.botWeapons      Bottien aseet (1 = pistooli, 2 = konepistooli, 3 = sinko,
 34  *                                            4 = haulikko, 5 = kranaatinlaukaisin, 6 = moottorisaha)
 35  * @property {Integer} config.healthItems     Lääkintäpakkausten määrä
 36  * @property {Integer} config.mgunItems       Konekiväärin ammuslaatikoiden määrä
 37  * @property {Integer} config.bazookaItems    Singon rakettilaatikoiden määrä
 38  * @property {Integer} config.shotgunItems    Haulikon haulilaatikoiden määrä
 39  * @property {Integer} config.launcherItems   Kranaatinlaukaisimen kranaattilaatikoiden määrä
 40  * @property {Integer} config.chainsawItems   Moottirisahan bensakanistereiden määrä
 41  * @property {Integer} tileSize               Yhden tilen leveys/korkeus pikseleissä
 42  * @property {Integer} width                  Kartan leveys
 43  * @property {Integer} height                 Kartan korkeus
 44  * @property {Array}   data                   Törmäyskerroksen data kaksiulotteisessa xy-taulukossa
 45 */
 46 function Map(server, name) {
 47   var filePath = path.resolve(__dirname, '..', 'maps', name + '.json')
 48     , data;
 49 
 50   this.server = server;
 51   this.loaded = false;
 52 
 53   if (!path.existsSync(filePath)) {
 54     log.error('Map %0 doesn\'t exist in %1', name.green, filePath.green);
 55     return;
 56   }
 57 
 58   log.info('Loading map %0 from %1', name.green, filePath.green);
 59   data = cjson.load(filePath);
 60   // Laajennetaan tämän kartan ominaisuuksia ladatulla json-tiedostolla
 61   cjson.extend(this, data);
 62 
 63   // Alustetaan kartan tavarat
 64   this.initItems();
 65 
 66   this.loaded = true;
 67 }

 69 /**
 70  * Törmätäänkö kartan hit-kerrokseen annetuissa pelikoordinaateissa.
 71  *
 72  * @param {Number} x  Tarkistettava x-koordinaatti pelikoordinaateissa
 73  * @param {Number} y  Tarkistettava y-koordinaatti pelikoordinaateissa
 74  */
 75 Map.prototype.isColliding = function (x, y) {
 76   var tileX = Math.ceil((x + this.width * this.tileSize / 2) / this.tileSize) - 1
 77     , tileY = Math.ceil((-y + this.height * this.tileSize / 2) / this.tileSize) - 1;
 78 
 79   if (tileX < 0 || tileX >= this.width || tileY < 0 || tileY >= this.height) {
 80     // Ollaan kartan ulkopuolella, eli törmätään.
 81     return true;
 82   }
 83 
 84   if (this.server.debug) {
 85     if ('undefined' === typeof this.data ||
 86         'undefined' === typeof this.data[tileY] ||
 87         'undefined' === typeof this.data[tileY][tileX] ) {
 88       log.error('Map.data[%0][%1] is undefined', tileY, tileX);
 89     }
 90   }
 91 
 92   // Nyt tarkistetaan että ollaanko seinän sisällä
 93   if (this.data[tileY][tileX]) {
 94     return true;
 95   }
 96 
 97   // Jos päästiin tänne asti niin ei olla törmätty
 98   return false;
 99 };

101 /**
102  * Etsii satunnaisen paikan kartalta, joka ei ole seinän sisällä, ja palauttaa kyseisen paikan
103  * koordinaatit objektissa, jolla on kentät x ja y.
104  *
105  * @returns {Object}  Objekti, jolla on kentät x ja y, jotka ovat löydetyn paikan koordinaatit
106  */
107 Map.prototype.findSpot = function () {
108   var randTileX, randTileY, returnObj = {};
109   // Etsitään vapaata paikkaa kartalta "vain" 10 000 kertaa
110   for (var i = 9999; --i;) {
111     randTileX = rand(0, this.width - 1);
112     randTileY = rand(0, this.height - 1);
113     if (!this.data[randTileY][randTileX]) {
114       // Ei ollut seinän sisällä
115       returnObj.x = (randTileX * this.tileSize)
116                   - (this.width * this.tileSize) / 2
117                   + this.tileSize / 2;
118       returnObj.y = -((randTileY * this.tileSize)
119                   - (this.height * this.tileSize) / 2
120                   + this.tileSize / 2);
121       break;
122     }
123   }
124   return returnObj;
125 };

130 Map.prototype.initItems = function () {
131   var itemId = 0;
132 
133   for (var i = this.config.healthItems - 1; i--;) {
134     new Item(this.server, this, ++itemId, ITM.HEALTH);
135   }
136   for (i = this.config.mgunItems - 1; i--;) {
137     new Item(this.server, this, ++itemId, ITM.AMMO);
138   }
139   for (i = this.config.bazookaItems - 1; i--;) {
140     new Item(this.server, this, ++itemId, ITM.ROCKET);
141   }
142   for (i = this.config.shotgunItems - 1; i--;) {
143     new Item(this.server, this, ++itemId, ITM.SHOTGUN);
144   }
145   for (i = this.config.launcherItems - 1; i--;) {
146     new Item(this.server, this, ++itemId, ITM.LAUNCHER);
147   }
148   for (i = this.config.chainsawItems - 1; i--;) {
149     new Item(this.server, this, ++itemId, ITM.FUEL);
150   }
151 };

153 /**
154  * Etsii annettujen koordinaattien ja kulman muodostavalta suoralta lähimmän seinän koordinaatit
155  * ja palauttaa ne objektin kenttinä x ja y.
156  *
157  * @param {Number} x      Aloituspisteen x-koordinaatti
158  * @param {Number} y      Aloituspisteen y-koordinaatti
159  * @param {Number} angle  Mihin suuntaan suora muodostuu (asteina)
160  *
161  * @returns {Object}  Palauttaa objektin jolla on kentät x ja y jotka sisältävät lähimmän seinän
162  *                    koordinaatit.
163  */
164 Map.prototype.findWall = function (x, y, angle, dist) {
165   var startP = {}, endP = {}, returnP;
166 
167   // Muunnetaan x ja y maailmankoordinaateista "näyttökoordinaateiksi"
168   startP.x = x + this.width * this.tileSize / 2;
169   startP.y = -y + this.height * this.tileSize / 2;
170 
171   // Lasketaan loppupiste
172   endP.x = startP.x + Math.cos((-angle / 180) * Math.PI) * dist;
173   endP.y = startP.y + Math.sin((-angle / 180) * Math.PI) * dist;
174 
175   // Suoritetaan raycast
176   returnP = this.rayCast(startP, endP);
177 
178   if (!returnP) {
179     // Välissä ei ollut seinää, ei palauteta mitään.
180     //console.log("No hits with raycast");
181     return;
182   }
183 
184   // Muunnetaan "näyttökoordinaatit" maailmankoordinaateiksi
185   returnP.x = returnP.x - (this.width * this.tileSize / 2);
186   returnP.y = (this.height * this.tileSize / 2) - returnP.y;
187 
188   // Palautetaan piste
189   return returnP;
190 }

192 /** Raycast */
193 Map.prototype.rayCast = function (origP1, origP2) {
194   // Pisteiden normalisaatio
195   var p1 = {x: origP1.x / this.tileSize, y: origP1.y / this.tileSize}
196     , p2 = {x: origP2.x / this.tileSize, y: origP2.y / this.tileSize}
197     // Seuraavien muuttujien kommentointi löytyy myöhemmin funktiossa
198     , stepX, stepY
199     , rayDirX, rayDirY
200     , ratioX, ratioY
201     , deltaX, deltaY
202     , testTile
203     , maxX, maxY
204     , endTile
205     , hit
206     , colP;
207 
208   // Ylitetäänkö minkään laatan rajoja
209   if (truncateNumber(p1.x) === truncateNumber(p2.x) && truncateNumber(p1.y) === truncateNumber(p2.y)) {
210     // Ei ylitä minkään laatan rajoja, joten ei voi olla törmäystä.
211     return;
212   }
213 
214   // Kumpaan suuntaan mennään x- ja y-suunnassa
215   stepX = (p2.x > p1.x) ? 1 : -1;
216   stepY = (p2.y > p1.y) ? 1 : -1;
217 
218   // Säteen suunta
219   rayDirX = p2.x - p1.x;
220   rayDirY = p2.y - p1.y;
221 
222   // Kuinka pitkälle liikutaan kummallakin akselilla kun toisella akselilla hypätään seuraavaan
223   // kokonaiseen tileen
224   ratioX = rayDirX / rayDirY;
225   ratioY = rayDirY / rayDirX;
226 
227   // Muutos x- ja y-suunnassa
228   deltaY = Math.abs(p2.x - p1.x);
229   deltaX = Math.abs(p2.y - p1.y);
230 
231   // Alustetaan testiä varten käytettävät kokonaislukumuuttujat alkutilekoordinaatteihin
232   // Huom: Käytetään normalisoituja versioita pisteestä origP1
233   testTile = {x: truncateNumber(p1.x), y: truncateNumber(p1.y)};
234 
235   // Alustetaan ei-kokonaislukuhyppäys liikkumalla seuraavan tilen reunalle ja jakamalla saatu
236   // arvo vastakkaisen akselin kokonaisluvulla.
237   // Jos liikutaan positiiviseen suuntaan, siirrytään nykyisen tilen päähän, muulloin alkuun.
238   if (stepX > 0) {
239     maxX = deltaX * (1.0 - (p1.x % 1));
240   } else {
241     maxX = deltaX * (p1.x % 1);
242   }
243   if (stepY > 0) {
244     maxY = deltaY * (1.0 - (p1.y % 1));
245   } else {
246     maxY = deltaY * (p1.y % 1);
247   }
248 
249   // Lopputile
250   endTile = {x: truncateNumber(p2.x), y: truncateNumber(p2.y)};
251 
252   // Nyt liikutaan!
253   hit = 0;
254   colP = {x: 0.0, y: 0.0};
255 
256   while (testTile.x !== endTile.x || testTile.y !== endTile.y) {
257     if (maxX < maxY) {
258       maxX += deltaX;
259       testTile.x += stepX;
260 
261       if (testTile.x < 0 || testTile.x >= this.width || testTile.y < 0 || testTile.y >= this.height) {
262         // Ollaan kartan ulkopuolella, eli törmätään.
263         hit = 1;
264       } else {
265         // Tarkistetaan onko tilekerroksessa hit-dataa
266         hit = this.data[testTile.y][testTile.x];
267       }
268       if (hit) {
269         // Raycast löysi törmäyksen
270         colP.x = testTile.x;
271         if (stepX < 0) { colP.x += 1.0; } // Jos mennään vasemmalle päin, lisätään yksi.
272         colP.y = p1.y + ratioY * (colP.x - p1.x);
273         colP.x *= this.tileSize; // Skaalataan törmäyspiste ylöspäin
274         colP.y *= this.tileSize;
275         // Palautetaan osumapiste
276         return colP;
277       }
278     } else {
279       maxY += deltaY;
280       testTile.y += stepY;
281 
282       if (testTile.x < 0 || testTile.x >= this.width || testTile.y < 0 || testTile.y >= this.height) {
283         // Ollaan kartan ulkopuolella, eli törmätään.
284         hit = 0;
285       } else {
286         // Tarkistetaan onko tilekerroksessa hit-dataa
287         hit = this.data[testTile.y][testTile.x];
288       }
289       if (hit) {
290         // Raycast löysi törmäyksen
291         colP.y = testTile.y;
292         if (stepY < 0) { colP.y += 1.0; } // Jos mennään ylöspäin, lisätään yksi.
293         colP.x = p1.x + ratioX * (colP.y - p1.y);
294         colP.x *= this.tileSize; // Skaalataan törmäyspiste ylöspäin
295         colP.y *= this.tileSize;
296         // Palautetaan osumapiste
297         return colP;
298       }
299     }
300   }
301 
302   // Jos tänne asti ollaan päästy, niin törmäystä ei ole löydetty. Ei siis palauteta mitään.
303 }
