 16 /**
 17  * Lataa uuden kartan.
 18  * @class Karttaluokka
 19  *
 20  * @param {String} name  Kartan nimi, joka ladataan (esim. Luna)
 21  *
 22  * @property {Boolean} loaded                 Ladattiinko kartta onnistuneesti
 23  * @property {String}  name                   Kartan nimi
 24  * @property {String}  author                 Kartan tekijä
 25  * @property {String}  crc32                  Kartan CRC32-tunniste, jota tarvitaan CoolBasicin puolella
 26  * @property {Object}  config                 Kartan asetukset sisältävä objekti
 27  * @property {Integer} config.maxPlayers      Maksimipelaajat
 28  * @property {Integer} config.botCount        Bottien määrä
 29  * @property {Integer} config.botDepartLimit  Kun pelaajamäärä ylittää tämän arvon, poistuu botti pelistä
 30  * @property {Array}   config.botNames        Bottien nimet
 31  * @property {Array}   config.botWeapons      Bottien aseet (1 = pistooli, 2 = konepistooli, 3 = sinko,
 32  *                                            4 = haulikko, 5 = kranaatinlaukaisin, 6 = moottorisaha)
 33  * @property {Integer} config.healthItems     Lääkintäpakkausten määrä
 34  * @property {Integer} config.mgunItems       Konekiväärin ammuslaatikoiden määrä
 35  * @property {Integer} config.bazookaItems    Singon rakettilaatikoiden määrä
 36  * @property {Integer} config.shotgunItems    Haulikon haulilaatikoiden määrä
 37  * @property {Integer} config.launcherItems   Kranaatinlaukaisimen kranaattilaatikoiden määrä
 38  * @property {Integer} config.chainsawItems   Moottirisahan bensakanistereiden määrä
 39  * @property {Integer} tileWidth              Yhden tilen leveys pikseleissä
 40  * @property {Integer} tileHeight             Yhden tilen korkeus pikseleissä
 41  * @property {Integer} width                  Kartan leveys
 42  * @property {Integer} height                 Kartan korkeus
 43  * @property {Array}   data                   Törmäyskerroksen data kaksiulotteisessa xy-taulukossa
 44 */
 45 function Map(name) {
 46   var filePath = path.resolve(__dirname, '..', 'maps', name + '.json')
 47     , data;
 48 
 49   this.loaded = false;
 50 
 51   if (!path.existsSync(filePath)) {
 52     log.error('Map "%0" doesn\'t exist in "%1"', name, filePath);
 53     return;
 54   }
 55 
 56   log.info('Loading map "%0" from "%1"', name, filePath);
 57   data = cjson.load(filePath);
 58   // Laajennetaan tämän kartan ominaisuuksia ladatulla json-tiedostolla
 59   cjson.extend(this, data);
 60 
 61   this.loaded = true;
 62 }

 64 /**
 65  * Törmätäänkö kartan hit-kerrokseen annetuissa pelikoordinaateissa.
 66  *
 67  * @param {Number} x  Tarkistettava x-koordinaatti pelikoordinaateissa
 68  * @param {Number} y  Tarkistettava y-koordinaatti pelikoordinaateissa
 69  */
 70 Map.prototype.isColliding = function (x, y) {
 71   var tileX = Math.ceil((x + this.width * this.tileWidth / 2) / this.tileWidth) - 1
 72     , tileY = Math.ceil((-y + this.height * this.tileHeight / 2) / this.tileHeight) - 1;
 73 
 74   if (tileX < 0 || tileX >= this.width || tileY < 0 || tileY >= this.height) {
 75     // Ollaan kartan ulkopuolella, eli törmätään.
 76     return true;
 77   }
 78 
 79   if (argv.d) {
 80     if ('undefined' === typeof this.data ||
 81         'undefined' === typeof this.data[tileY] ||
 82         'undefined' === typeof this.data[tileY][tileX] ) {
 83       log.error('Map.data[%0][%1] is undefined', tileY, tileX);
 84     }
 85   }
 86 
 87   // Nyt tarkistetaan että ollaanko seinän sisällä
 88   if (this.data[tileY][tileX]) {
 89     return true;
 90   }
 91 
 92   // Jos päästiin tänne asti niin ei olla törmätty
 93   return false;
 94 }

 96 /**
 97  * Etsii satunnaisen paikan kartalta, joka ei ole seinän sisällä, ja palauttaa kyseisen paikan
 98  * koordinaatit objektissa, jolla on kentät x ja y.
 99  *
100  * @returns {Object}  Objekti, jolla on kentät x ja y, jotka ovat löydetyn paikan koordinaatit
101  */
102 Map.prototype.findSpot = function () {
103   var randTileX, randTileY, returnObj = {};
104   // Etsitään vapaata paikkaa kartalta "vain" 10 000 kertaa
105   for (var i = 9999; --i;) {
106     randTileX = rand(0, this.width - 1);
107     randTileY = rand(0, this.height - 1);
108     if (!this.data[randTileY][randTileX]) {
109       // Ei ollut seinän sisällä
110       returnObj.x = (randTileX * this.tileWidth)
111                   - (this.width * this.tileWidth) / 2
112                   + this.tileWidth / 2;
113       returnObj.y = -((randTileY * this.tileHeight)
114                   - (this.height * this.tileHeight) / 2
115                   + this.tileHeight / 2);
116       break;
117     }
118   }
119   return returnObj;
120 }
