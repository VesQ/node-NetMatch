1 /**
  2  * @fileOverview Sisältää hyödyllisiä funktioita, eli {@link Utils}-nimiavaruuden toteutuksen.
  3  */
  4 var argv = require('optimist')
  5   .default({d: false}).alias({'d' : 'debug'}).argv
  6   , colors = require('colors')
  7   , Logger = require('cbNetwork').Logger;
  8 
  9 /**
 10  * @namespace Sisältää hyödyllisiä funktioita.
 11  */
 12 var Utils = {
 13   /**
 14    * Yleinen funktio lokiin/konsoliin (stdout) kirjoittamista varteen. Tämä on instanssi cbNetwork-noden
 15    * <a href="http://vesq.github.com/cbNetwork-node/doc/symbols/Logger.html">Logger</a>-luokasta.
 16    * Voit käyttää tätä näin:
 17    * @example
 18    * var log = Utils.log;
 19    *
 20    * log.write('Perusviesti, tämä on ilman värejä.');
 21    * log.info('Jotain tiedotettavaa tapahtui. Tulostuu vihreänä.');
 22    * log.warn('Varoitus! Tulostuu keltaisena.');
 23    * log.notice('Ilmoitus! Tulostuu keltaisena.');
 24    * log.error('VIRHE! Tulostuu punaisena ja lihavoituna.');
 25    * log.fatal('KRIITTINEN VIRHE! Tulostuu punaisena ja lihavoituna.');
 26    */
 27   log: new Logger('[NetMatch %t] '.grey, (argv.d > 1) ? true : false),
 28 
 29   /**
 30    * Palauttaa nykyisen palvelimen ajan millisekunteina, toimii kuten CoolBasicin Timer().
 31    */
 32   timer: function () {
 33     return Date.now();
 34   },
 35 
 36   /**
 37    * Palauttaa satunnaisen luvun väliltä minVal...maxVal tarkkuudella floatVal
 38    * @param {Number} minVal      Pienin mahdollinen luku
 39    * @param {Number} maxVal      Suurin mahdollinen luku
 40    * @param {Number} [floatVal]  Palautettavan satunnaisen luvun tarkkuus. Mikäli tätä ei anneta,
 41    *                             palautetaan kokonaisluku.
 42    */
 43   rand: function (minVal, maxVal, floatVal) {
 44     var randVal = minVal + (Math.random() * (maxVal - minVal));
 45     return typeof floatVal === 'undefined' ? Math.round(randVal) : randVal.toFixed(floatVal);
 46   },
 47 
 48   /**
 49    * Pitää kulman välillä 0-360
 50    * @param {Number} angle  Kulma
 51    * @returns {Number}
 52    */
 53   wrapAngle: function (a) {
 54     a = a / 360;
 55     return (a - Math.floor(a)) * 360;
 56   },
 57 
 58   /**
 59    * Palauttaa kahden pisteen välisen etäisyyden
 60    * @param {Number} x1  Ensimmäisen pisteen x-koordinaatti
 61    * @param {Number} y1  Ensimmäisen pisteen y-koordinaatti
 62    * @param {Number} x2  Toisen pisteen x-koordinaatti
 63    * @param {Number} y2  Toisen pisteen y-koordinaatti
 64    * @returns {Number}   Pisteiden välinen etäisyys
 65    */
 66   distance: function (x1, y1, x2, y2) {
 67     var dx = x1 - x2; // Vaakasuuntainen etäisyys
 68     var dy = y1 - y2; // Pystysuuntainen etäisyys
 69     return Math.sqrt( dx*dx + dy*dy );
 70   }
 71 };
 72 
 73 exports = module.exports = Utils;
