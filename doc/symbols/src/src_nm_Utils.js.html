  6 "use strict";
  7 
  8 /**#nocode+*/
  9 var argv = require('optimist')
 10   .default({d: false}).alias({'d' : 'debug'}).argv
 11   , colors = require('colors')
 12   , Logger = require('cbNetwork').Logger;
 13 /**#nocode-*/
 14 
 15 /**
 16  * @namespace Sisältää hyödyllisiä funktioita.
 17  */
 18 var Utils = {
 19   /**
 20    * Yleinen funktio lokiin/konsoliin (stdout) kirjoittamista varteen. Tämä on instanssi cbNetwork-noden
 21    * <a href="http://vesq.github.com/cbNetwork-node/doc/symbols/Logger.html">Logger</a>-luokasta.
 22    * Voit käyttää tätä näin:
 23    * @example
 24    * var log = Utils.log;
 25    *
 26    * log.write('Perusviesti, tämä on ilman värejä.');
 27    * log.info('Jotain tiedotettavaa tapahtui. Tulostuu vihreän INFO-tagin kanssa.');
 28    * log.warn('Varoitus! Tulostuu keltaisen WARN-tagin kanssa.');
 29    * log.notice('Ilmoitus! Tulostuu keltaisen NOTICE-tagin kanssa.');
 30    * log.error('VIRHE! Tulostuu punaisen ja lihavoidun ERROR-tagin kanssa.');
 31    * log.fatal('KRIITTINEN VIRHE! Tulostuu punaisen ja lihavoidun FATAL-tagin kanssa.');
 32    */
 33   log: new Logger('[NetMatch %t] '.grey, argv.d && (argv.d > 1 ? argv.d - 1 : 1)),
 34 
 35   /**
 36    * Palauttaa satunnaisen luvun väliltä minVal...maxVal, mahdollisesti liukulukuna
 37    * @param {Number} minVal       Pienin mahdollinen luku
 38    * @param {Number} maxVal       Suurin mahdollinen luku
 39    * @param {Boolean} [floatVal]  Palautetaanko liukulukuna. Jos tätä ei anneta, palautetaan
 40    *                              kokonaislukuna.
 41    */
 42   rand: function (minVal, maxVal, floatVal) {
 43     var randVal = minVal + (Math.random() * (maxVal - minVal));
 44     return typeof floatVal === 'undefined' ? Math.round(randVal) : randVal;
 45   },
 46 
 47   /**
 48    * Pitää kulman välillä 0-360
 49    * @param {Number} angle  Kulma
 50    * @returns {Number}
 51    */
 52   wrapAngle: function (a) {
 53     a = a / 360;
 54     return (a - Math.floor(a)) * 360;
 55   },
 56 
 57   /**
 58    * Palauttaa kahden pisteen välisen etäisyyden
 59    * @param {Number} x1  Ensimmäisen pisteen x-koordinaatti
 60    * @param {Number} y1  Ensimmäisen pisteen y-koordinaatti
 61    * @param {Number} x2  Toisen pisteen x-koordinaatti
 62    * @param {Number} y2  Toisen pisteen y-koordinaatti
 63    * @returns {Number}   Pisteiden välinen etäisyys
 64    */
 65   distance: function (x1, y1, x2, y2) {
 66     var dx = x1 - x2; // Vaakasuuntainen etäisyys
 67     var dy = y1 - y2; // Pystysuuntainen etäisyys
 68     return Math.sqrt( dx*dx + dy*dy );
 69   },
 70 
 71   /**
 72    * Palauttaa kahden pisteen välisen kulman asteina väliltä -180...180º
 73    * @param {Number} x1  Ensimmäisen pisteen x-koordinaatti
 74    * @param {Number} y1  Ensimmäisen pisteen y-koordinaatti
 75    * @param {Number} x2  Toisen pisteen x-koordinaatti
 76    * @param {Number} y2  Toisen pisteen y-koordinaatti
 77    * @returns {Number}   Pisteiden välinen kulma asteina välillä 0...360º
 78    */
 79   getAngle: function (x1, y1, x2, y2) {
 80     var radAngle = Math.atan2(y1 - y2, x1 - x2); // Kulma radiaaneina välillä -pi...pi
 81     return (radAngle / Math.PI) * 180;
 82   },
 83 
 84   /**
 85    * Poistaa numeron perästä desimaalit riippumatta numeron etumerkistä.
 86    * @param {Number} num  Numero, jonka perästä desimaalit poistetaan
 87    */
 88   truncateNumber: function (num) {
 89     if (num<0) { return Math.ceil(num); }
 90     return Math.floor(num);
 91   },
 92 
 93   /**
 94    * Palauttaa taulukon, jossa on merkkijono paloiteltuna sanoiksi, ottaa huomioon "merkki jonot"
 95    * @param {String} str  Merkkijono, joka paloitellaan
 96    * @returns {Array}  Paloiteltu jono
 97    */
 98   splitString: function (str) {
 99     var reg = /\ (?!\w+")/;
100     return str.split(reg);
101   },
102 
103   /**
104    * Kuten Array.prototype.join, mutta tukee viimeisen erottimen vaihtoa.
105    * @param {Array}  arr     Lista, joka yhdistetään
106    * @param {String} delim   Ensimmäinen erotin, jota käytetään muihin kuin viimeiseen väliin
107    * @param {String} delim2  Erotin, joka laitetaan viimeiseen väliin
108    * @returns {String}  Yhdistetty jono
109    */
110   join: function (arr, delim, delim2) {
111     return arr.length > 1 ? arr.slice(0, arr.length - 1).join(delim) +
112       ' ' + delim2 + arr[arr.length - 1] : arr[0];
113   }
114 
115 };
116 
117 module.exports = Utils;
