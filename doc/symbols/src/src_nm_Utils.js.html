1 /**
  2  * @fileOverview Sisältää hyödyllisiä funktioita, eli {@link Utils}-nimiavaruuden toteutuksen.
  3  */
  4 var argv = require('optimist')
  5   .default({d: false}).alias({'d' : 'debug'}).argv
  6   , colors = require('colors')
  7   , Logger = require('cbNetwork').Logger;
  8 
  9 /**
 10  * @namespace Sisältää hyödyllisiä funktioita.
 11  */
 12 var Utils = {
 13   /**
 14    * Yleinen funktio lokiin/konsoliin (stdout) kirjoittamista varteen. Tämä on instanssi cbNetwork-noden
 15    * <a href="http://vesq.github.com/cbNetwork-node/doc/symbols/Logger.html">Logger</a>-luokasta.
 16    * Voit käyttää tätä näin:
 17    * @example
 18    * var log = Utils.log;
 19    *
 20    * log.write('Perusviesti, tämä on ilman värejä.');
 21    * log.info('Jotain tiedotettavaa tapahtui. Tulostuu vihreän INFO-tagin kanssa.');
 22    * log.warn('Varoitus! Tulostuu keltaisen WARN-tagin kanssa.');
 23    * log.notice('Ilmoitus! Tulostuu keltaisen NOTICE-tagin kanssa.');
 24    * log.error('VIRHE! Tulostuu punaisen ja lihavoidun ERROR-tagin kanssa.');
 25    * log.fatal('KRIITTINEN VIRHE! Tulostuu punaisen ja lihavoidun FATAL-tagin kanssa.');
 26    */
 27   log: new Logger('[NetMatch %t] '.grey, argv.d),
 28 
 29   /**
 30    * Palauttaa satunnaisen luvun väliltä minVal...maxVal, mahdollisesti liukulukuna
 31    * @param {Number} minVal       Pienin mahdollinen luku
 32    * @param {Number} maxVal       Suurin mahdollinen luku
 33    * @param {Boolean} [floatVal]  Palautetaanko liukulukuna. Jos tätä ei anneta, palautetaan
 34    *                              kokonaislukuna.
 35    */
 36   rand: function (minVal, maxVal, floatVal) {
 37     var randVal = minVal + (Math.random() * (maxVal - minVal));
 38     return typeof floatVal === 'undefined' ? Math.round(randVal) : randVal;
 39   },
 40 
 41   /**
 42    * Pitää kulman välillä 0-360
 43    * @param {Number} angle  Kulma
 44    * @returns {Number}
 45    */
 46   wrapAngle: function (a) {
 47     a = a / 360;
 48     return (a - Math.floor(a)) * 360;
 49   },
 50 
 51   /**
 52    * Palauttaa kahden pisteen välisen etäisyyden
 53    * @param {Number} x1  Ensimmäisen pisteen x-koordinaatti
 54    * @param {Number} y1  Ensimmäisen pisteen y-koordinaatti
 55    * @param {Number} x2  Toisen pisteen x-koordinaatti
 56    * @param {Number} y2  Toisen pisteen y-koordinaatti
 57    * @returns {Number}   Pisteiden välinen etäisyys
 58    */
 59   distance: function (x1, y1, x2, y2) {
 60     var dx = x1 - x2; // Vaakasuuntainen etäisyys
 61     var dy = y1 - y2; // Pystysuuntainen etäisyys
 62     return Math.sqrt( dx*dx + dy*dy );
 63   },
 64 
 65   /**
 66    * Palauttaa kahden pisteen välisen kulman asteina
 67    * @param {Number} x1  Ensimmäisen pisteen x-koordinaatti
 68    * @param {Number} y1  Ensimmäisen pisteen y-koordinaatti
 69    * @param {Number} x2  Toisen pisteen x-koordinaatti
 70    * @param {Number} y2  Toisen pisteen y-koordinaatti
 71    * @returns {Number}   Pisteiden välinen kulma asteina välillä 0...360º
 72    */
 73   getAngle: function (x1, y1, x2, y2) {
 74     var radAngle = Math.atan2(y1 - y2, x1 - x2) + Math.PI; // Kulma radiaaneina välillä 0...2pi
 75     return (radAngle / Math.PI) * 180;
 76   },
 77 
 78   /**
 79    * Poistaa numeron perästä desimaalit riippumatta numeron etumerkistä.
 80    * @param {Number} num  Numero, jonka perästä desimaalit poistetaan
 81    */
 82   truncateNumber: function (num) {
 83     if (num<0) return Math.ceil(num);
 84     else return Math.floor(num);
 85   },
 86 
 87   /**
 88    * Palauttaa taulukon, jossa on merkkijono paloiteltuna sanoiksi, ottaa huomioon "merkki jonot"
 89    * @param {String} str  Merkkijono, joka paloitellaan
 90    * @returns {Array}  Paloiteltu jono
 91    */
 92   splitString: function (str) {
 93     var reg = /\ (?!\w+")/;
 94     return str.split(reg);
 95   },
 96 
 97   /**
 98    * Kuten Array.prototype.join, mutta tukee viimeisen erottimen vaihtoa.
 99    * @param {Array}  arr     Lista, joka yhdistetään
100    * @param {String} delim   Ensimmäinen erotin, jota käytetään muihin kuin viimeiseen väliin
101    * @param {String} delim2  Erotin, joka laitetaan viimeiseen väliin
102    * @returns {String}  Yhdistetty jono
103    */
104   join: function (arr, delim, delim2) {
105     return arr.length > 1 ? arr.slice(0, arr.length - 1).join(delim) +
106       ' ' + delim2 + arr[arr.length - 1] : arr[0];
107   }
108 
109 };
110 
111 module.exports = Utils;
