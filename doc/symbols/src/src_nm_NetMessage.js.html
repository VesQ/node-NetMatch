 13 /**
 14  * Alustaa uuden viestisäilön.
 15  * @class Viestien säilytys
 16  *
 17  * @param {Server} server  Tämän viestisäilön {@link Server}-instanssi
 18  */
 19 function NetMessages(server) {
 20   this.server = server;
 21   /**
 22    * Sisältää clienteille lähetettävät viestit
 23    * @private
 24    */
 25   this.data = {};
 26 }
 27 
 28 /**
 29  * Luo uuden clientille lähetettävän viestin.
 30  *
 31  * @param {Byte}      toPlayer          Pelaajan ID kelle viesti lähetetään.
 32  * @param {Object}    data              Pelaajalle lähetettävä data
 33  * @param {Byte}      data.msgType      Viestityyppi, kts. {@link NET}
 34  * @param {String}    data.msgText      Viestin teksti
 35  * @param {Player}    data.player       Kuka viestin lähetti
 36  * @param {Short}     data.bullet       Ammus
 37  * @param {Byte}      data.itemId       Tavaran tunnus
 38  * @param {Byte}      data.itemType     Tavaran tyyppi
 39  * @param {Byte}      data.weapon       Ase
 40  * @param {Short}     data.x            Sijaintitietoa
 41  * @param {Short}     data.y            Sijaintitietoa
 42  * @param {Player}    data.player2      Kehen tapahtuma kohdistui
 43  * @param {Boolean}   data.sndPlay      Soitetaanko ääni
 44  * @param {Boolean}   data.handShooted  Kumpi käsi ampui (pistooli) 0 = vasen, 1 = oikea
 45  * @param {Byte}      data.drawType     Minkä kuvion piirto on kyseessä
 46  * @param {Integer[]} data.drawVars     Suoraan listassa piirtoa varten tarvittavat arvot
 47  */
 48 NetMessages.prototype.add = function (toPlayer, data) {
 49   if ('undefined' === typeof this.data[toPlayer]) {
 50     this.data[toPlayer] = [];
 51   }
 52   this.data[toPlayer].push(data);
 53 };
 54 
 55 /**
 56  * Lähettää kaikille clienteille viestin.
 57  * @see NetMessages#add
 58  *
 59  * @param {Object} data      Pelaajalle lähetettävä data
 60  * @param {Byte} [butNotTo]  Pelaajan ID, kelle EI lähetetä tätä pakettia.
 61  */
 62 NetMessages.prototype.addToAll = function (data, butNotTo) {
 63   var self = this;
 64 
 65   if ('number' !== typeof butNotTo) {
 66     butNotTo = 0;
 67   }
 68 
 69   this.server.loopPlayers(function (plr) {
 70     if (plr.active && !plr.zombie && (plr.id !== butNotTo)) {
 71       self.add(plr.id, data);
 72     }
 73   });
 74 };
 75 
 76 /**
 77  * Lähettää kaikille annetun joukkueen jäsenille.
 78  * @see NetMessages#add
 79  *
 80  * @param {Byte} team    Joukkue jonka jäsenille lähetetään viesti
 81  * @param {Object} data  Pelaajalle lähetettävä data
 82  */
 83 NetMessages.prototype.addToTeam = function (team, data) {
 84   var self = this;
 85 
 86   this.server.loopPlayers(function (plr) {
 87     if (plr.active && !plr.zombie && (plr.team === team)) {
 88       self.add(plr.id, data);
 89     }
 90   });
 91 };

 93 /**
 94  * Lisää data-pakettiin yksittäiselle pelaajalle kuuluvat viestit oikein jäsenneltynä.
 95  * Kts. cbNetwork-node toteutus luokasta <a href="http://vesq.github.com/cbNetwork-node/doc/symbols/Packet.html">Packet</a>.
 96  *
 97  * @param {Player} toPlayer  Kenen viestit haetaan
 98  * @param {Packet} data      Mihin pakettiin tiedot lisätään
 99  */
100 NetMessages.prototype.fetch = function (toPlayer, data) {
101   var d, b;
102 
103   if ('undefined' === typeof this.data[toPlayer.id] || this.data[toPlayer.id].length === 0) {
104     return false;
105   }
106   // Tämän viestin data laitetaan d-muuttujaan, jotta tarvitsisi kirjoittaa vähemmän.
107   d = this.data[toPlayer.id][0];
108   while (d) {
109     if (!d.hasOwnProperty('msgType')) {
110       log.error('Virheellistä dataa NetMessages-objektissa!');
111       console.dir(d);
112       continue;
113     }
114 
115     // Lisätään dataa riippuen siitä minkälaista dataa pitää lähettää.
116     switch (d.msgType) {
117       case NET.LOGIN:
118         // Joku on liittynyt peliin
119         data.putByte(d.msgType);
120         data.putByte(d.player.id);      // Kuka liittyi
121         data.putString(d.msgText);      // Liittymisteksti
122         data.putByte(d.player.zombie);  // Oliko liittynyt botti?
123         break;
124 
125       case NET.LOGOUT:
126         // Joku on poistunut pelistä
127         data.putByte(d.msgType);
128         data.putByte(d.player.id);
129         break;
130 
131       case NET.NEWBULLET:
132         // Uusi ammus on ammuttu
133         if (d.weapon === WPN.CHAINSAW) {
134           // Moottorisahalla "ammutaan"
135           data.putByte(d.msgType);
136           data.putShort(d.bullet.id); // Ammuksen tunnus
137           data.putByte(d.player.id);  // Kuka ampui
138 
139           // Tungetaan samaan tavuun useampi muuttuja:
140           b = ((d.weapon % 16) << 0)
141             + (d.sndPlay << 4);
142           data.putByte(b);
143 
144           // Ammuksen sijainti
145           data.putShort(d.x);
146           data.putShort(d.y);
147           data.putShort(0);  // Ammuksen kulma, mutta koska moottirisahalla ei ole kulmaa, on tämä 0
148         } else {
149           // Jokin muu kuin moottorisaha
150           if ('undefined' !== typeof d.bullet) {
151             data.putByte(d.msgType);
152             data.putShort(d.bullet.id);
153             data.putByte(d.player.id);
154 
155             // Tungetaan samaan tavuun useampi muuttuja:
156             b = ((d.weapon % 16) << 0)  // Millä aseella (mod 16 ettei vie yli 4 bittiä)
157               + (d.sndPlay << 4)        // Soitetaanko ääni
158               + (d.handShooted << 5);   // Kummalla kädellä ammuttiin
159             data.putByte(b);
160 
161             // Ammuksen sijainti
162             data.putShort(d.bullet.x);
163             data.putShort(d.bullet.y);
164             data.putShort(d.bullet.angle);
165           }
166         }
167         break;
168 
169       case NET.TEXTMESSAGE:
170         // Tsättiviesti
171         data.putByte(d.msgType);
172         data.putByte(d.player.id);
173         data.putString(d.msgText);
174         break;
175 
176       case NET.SERVERMSG:
177         // Palvelimen generoima viesti
178         data.putByte(d.msgType);
179         data.putString(d.msgText);
180         break;
181 
182       case NET.BULLETHIT:
183         // Osumaviesti
184         data.putByte(d.msgType);
185         data.putShort(d.bullet.id);   // Ammuksen tunnus
186         if (d.player) {
187           data.putByte(d.player.id);  // Keneen osui
188         } else {
189           data.putByte(0);
190         }
191         data.putShort(d.x);           // Missä osui
192         data.putShort(d.y);           // Missä osui
193         data.putByte(d.weapon);       // Mistä aseesta ammus on
194         break;
195 
196       case NET.ITEM:
197         // Tavaraviesti
198         data.putByte(d.msgType);
199         data.putByte(d.itemId);     // Tavaran tunnus
200         data.putByte(d.itemType);   // Tavaran tyyppi
201         data.putShort(d.x);         // Missä tavara on
202         data.putShort(d.y);         // Missä tavara on
203         break;
204 
205       case NET.KILLMESSAGE:
206         // Tappoviesti! Buahahahaaaa
207         data.putByte(d.msgType);
208         data.putByte(d.player.id);       // Tappaja
209         data.putByte(d.player2.id);      // Tapettu
210         data.putByte(d.weapon);          // Ase
211         data.putShort(d.player.kills);   // Tappajan tapot
212         data.putShort(d.player.deaths);  // Tappajan kuolemat
213         data.putShort(d.player2.kills);  // Uhrin tapot
214         data.putShort(d.player2.deaths); // Uhrin kuolemat
215         break;
216 
217       case NET.KICKED:
218         // Pelaaja potkittiin
219         data.putByte(d.msgType);
220         if (d.player) {
221           data.putByte(d.player.id); // Kuka potkaisi
222         } else {
223           data.putByte(0);           // Palvelin potkaisi
224         }
225         data.putByte(d.player2.id);  // Kenet potkittiin
226         data.putString(d.msgText);   // Potkujen syy
227         break;
228 
229       case NET.TEAMINFO:
230         // Lähetetään pelaajan joukkue
231         data.putByte(d.msgType);
232         data.putByte(d.player.id);   // Pelaaja
233         data.putByte(d.player.team); // Pelaajan joukkue
234         break;
235 
236       case NET.SPEEDHACK:
237         // Tämä client on haxor!
238         data.putByte(d.msgType);
239         // UNIMPLEMENTED
240         // Login( False, gCurrentPlayerId )
241         break;
242 
243       case NET.DEBUGDRAWING:
244         // Debug-piirtelytavaraa tulossa
245         data.putByte(d.msgType);
246         data.putByte(d.drawType);
247         if (d.drawVars) {
248           for (var i=0; i < d.drawVars.length; i++) {
249             data.putShort(d.drawVars[i]);
250           }
251         }
252         break;
253 
254       default:
255         log.error('VIRHE: Pelaajalle %0 (%1) oli osoitettu tuntematon paketti:',
256           toPlayer.name.green, String(toPlayer).id.magenta);
257         console.dir(d);
258     }
259 
260     // Poistetaan viesti muistista
261     this.data[toPlayer.id].splice(0, 1);
262 
263     // Siirrytään seuraavaan viestiin
264     d = this.data[toPlayer.id][0];
265   }
266 };
