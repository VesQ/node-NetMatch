 14 /**
 15  * @class Viestien säilytys
 16  */
 17 var NetMessages = function (server) {
 18   this.server = server;
 19   /**
 20    * Sisältää clienteille lähetettävät viestit
 21    * @private
 22    */
 23   this.data = {};
 24 }
 25 
 26 /**
 27  * Luo uuden clientille lähetettävän viestin.
 28  *
 29  * @param {Byte}    toPlayer          Pelaajan ID kelle viesti lähetetään.
 30  * @param {Object}  data              Pelaajalle lähetettävä data
 31  * @param {Byte}    data.msgType      Viestityyppi, kts. {@link NET}
 32  * @param {String}  data.msgText      Viestin teksti
 33  * @param {Byte}    data.playerId     Kuka viestin lähetti
 34  * @param {Short}   data.bulletId     Ammuksen tunnus
 35  * @param {Byte}    data.itemId       Tavaran tunnus
 36  * @param {Byte}    data.itemType     Tavaran tyyppi
 37  * @param {Byte}    data.weapon       Ase
 38  * @param {Short}   data.x            Sijaintitietoa
 39  * @param {Short}   data.y            Sijaintitietoa
 40  * @param {Byte}    data.playerId2    Pelaajatunnus (kehen tapahtuma kohdistui)
 41  * @param {Boolean} data.sndPlay      Soitetaanko ääni
 42  * @param {Boolean} data.handShooted  Kumpi käsi ampui (pistooli) 0 = vasen, 1 = oikea
 43  */
 44 NetMessages.prototype.add = function (toPlayer, data) {
 45   if ('undefined' === typeof this.data[toPlayer]) {
 46     this.data[toPlayer] = [];
 47   }
 48   this.data[toPlayer].push(data);
 49 }
 50 
 51 /**
 52  * Lähettää kaikille clienteille viestin.
 53  * @see NetMessages#add
 54  *
 55  * @param {Object} data      Pelaajalle lähetettävä data
 56  * @param {Byte} [butNotTo]  Pelaajan ID, kelle EI lähetetä tätä pakettia.
 57  */
 58 NetMessages.prototype.addToAll = function (data, butNotTo) {
 59   var playerIds = Object.keys(this.server.players)
 60     , plr
 61     , iterator = playerIds.length;
 62 
 63   if ('number' !== typeof butNotTo) {
 64     butNotTo = 0;
 65   }
 66 
 67   while (iterator--) {
 68     plr = this.server.players[playerIds[iterator]];
 69     if (plr.active && !plr.zombie && (plr.playerId !== butNotTo)) {
 70       this.add(plr.playerId, data);
 71     }
 72   }
 73 }

 75 /**
 76  * Lisää data-pakettiin yksittäiselle pelaajalle kuuluvat viestit oikein jäsenneltynä.
 77  * Kts. cbNetwork-node toteutus luokasta <a href="http://vesq.github.com/cbNetwork-node/doc/symbols/Packet.html">Packet</a>.
 78  *
 79  * @param {Byte} toPlayer  Kenen viestit haetaan
 80  * @param {Packet} data    Mihin pakettiin tiedot lisätään
 81  */
 82 NetMessages.prototype.fetch = function (toPlayer, data) {
 83   var d, b;
 84 
 85   if ('undefined' === typeof this.data[toPlayer] || this.data[toPlayer].length === 0) {
 86     return false;
 87   }
 88   // Tämän viestin data laitetaan d-muuttujaan, jotta tarvitsisi kirjoittaa vähemmän.
 89   while (d = this.data[toPlayer][0]) {
 90     if (!d.hasOwnProperty('msgType')) {
 91       log.error('Virheellistä dataa NetMessages-objektissa!');
 92       console.dir(d);
 93       continue;
 94     }
 95 
 96     // Lisätään dataa riippuen siitä minkälaista dataa pitää lähettää.
 97     switch (d.msgType) {
 98       case NET.LOGIN:
 99         // Joku on liittynyt peliin
100         data.putByte(d.msgType);
101         data.putByte(d.playerId);   // Kuka liittyi
102         data.putString(d.msgText);  // Liittymisteksti
103         data.putByte(d.playerId2);  // Kenen tilalle pelaaja tuli
104         break;
105 
106       case NET.LOGOUT:
107         // Joku on poistunut pelistä
108         data.putByte(d.msgType);
109         data.putByte(d.playerId);
110         break;
111 
112       case NET.NEWBULLET:
113         // Uusi ammus on ammuttu
114         if (d.weapon === WPN.CHAINSAW) {
115           // Moottorisahalla "ammutaan"
116           data.putByte(d.msgType);
117           data.putShort(d.bulletId);  // Ammuksen tunnus
118           data.putByte(d.playerId);   // Kuka ampui
119 
120           // Tungetaan samaan tavuun useampi muuttuja:
121           b = ((d.weapon % 16) << 0)
122             + (d.sndPlay << 4);
123           data.putByte(b);
124 
125           // Ammuksen sijainti
126           data.putShort(d.x);
127           data.putShort(d.y);
128           data.putShort(0);  // Ammuksen kulma, mutta koska moottirisahalla ei ole kulmaa, on tämä 0
129         } else {
130           // Jokin muu kuin moottorisaha
131           var bullet;
132           for (var i = this.server.bullets.length; i--;) {
133             if (this.server.bullets[i].bulletId === d.bulletId) {
134               bullet = this.server.bullets[i];
135               break;
136             }
137           }
138           if ('undefined' !== typeof bullet) {
139             data.putByte(d.msgType);
140             data.putShort(d.bulletId);
141             data.putByte(d.playerId);
142 
143             // Tungetaan samaan tavuun useampi muuttuja:
144             b = ((d.weapon % 16) << 0)  // Millä aseella (mod 16 ettei vie yli 4 bittiä)
145               + (d.sndPlay << 4)        // Soitetaanko ääni
146               + (d.handShooted << 5)    // Kummalla kädellä ammuttiin
147             data.putByte(b);
148 
149             // Ammuksen sijainti
150             data.putShort(bullet.x);
151             data.putShort(bullet.y);
152             data.putShort(bullet.angle);
153           }
154         }
155         break;
156 
157       case NET.TEXTMESSAGE:
158         // Tsättiviesti
159         data.putByte(d.msgType);
160         data.putByte(d.playerId);
161         data.putString(d.msgText);
162         break;
163 
164       case NET.SERVERMSG:
165         // Palvelimen generoima viesti
166         data.putByte(d.msgType);
167         data.putString(d.msgText);
168         break;
169 
170       case NET.BULLETHIT:
171         // Osumaviesti
172         data.putByte(d.msgType);
173         data.putShort(d.bulletId);  // Ammuksen tunnus
174         data.putByte(d.playerId);   // Keneen osui
175         data.putShort(d.x);          // Missä osui
176         data.putShort(d.y);          // Missä osui
177         data.putByte(d.weapon);     // Mistä aseesta ammus on
178         break;
179 
180       case NET.ITEM:
181         // Tavaraviesti
182         data.putByte(d.msgType);
183         data.putByte(d.itemId);     // Tavaran tunnus
184         data.putByte(d.itemType);   // Tavaran tyyppi
185         data.putShort(d.x);         // Missä tavara on
186         data.putShort(d.y);         // Missä tavara on
187         break;
188 
189       case NET.KILLMESSAGE:
190         // Tappoviesti! Buahahahaaaa
191         data.putByte(d.msgType);
192         data.putByte(d.playerId);   // Tappaja
193         data.putByte(d.playerId2);  // Tapettu
194         data.putByte(d.weapon);     // Ase
195         // UNIMPLEMENTED
196         data.putShort(0);           // Tappajan tapot
197         data.putShort(0);           // Tappajan kuolemat
198         data.putShort(0);           // Uhrin tapot
199         data.putShort(0);           // Uhrin kuolemat
200         break;
201 
202       case NET.KICKED:
203         // Pelaaja potkittiin
204         data.putByte(d.msgType);
205         data.putByte(d.playerId);   // Kuka potkaisi
206         data.putByte(d.playerId2);  // Kenet potkittiin
207         data.putString(d.msgText);  // Potkujen syy
208         break;
209 
210       case NET.TEAMINFO:
211         // Lähetetään pelaajan joukkue
212         data.putByte(d.msgType);
213         data.putByte(d.playerId);   // Pelaaja
214         // UNIMPLEMENTED
215         data.putByte(1);            // Pelaajan joukkue
216         break;
217 
218       case NET.SPEEDHACK:
219         // Tämä client on haxor!
220         data.putByte(d.msgType);
221         // UNIMPLEMENTED
222         // Login( False, gCurrentPlayerId )
223         break;
224 
225       default:
226         log.error('VIRHE: Pelaajalle <'+toPlayer+'> oli osoitettu tuntematon paketti:');
227         console.dir(d);
228     }
229 
230     // Poistetaan viesti muistista
231     this.data[toPlayer].splice(0, 1);
232   }
233 }
