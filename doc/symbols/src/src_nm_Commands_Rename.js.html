1 /**
  2  * @fileOverview Toteutus komennolle rename: {@link Commands.rename}
  3  */
  4 
  5 "use strict";
  6 
  7 var log = require('../Utils').log;
  8 
  9 /**
 10  * Uudelleennimeää pelaajan.
 11  * @methodOf Commands
 12  *
 13  * @param {Player} who   Pelaaja, joka uudelleennimetään. Nimi tai ID kelpaa
 14  * @param {String} name  Uusi nimi
 15  */
 16 var rename = {
 17   /**#nocode+*/
 18   params: [
 19     {name: 'who',  type: 'player', optional: false, help: 'Player who needs to be renamed'},
 20     {name: 'name', type: 'string', optional: false, help: 'New name'}
 21   ],
 22   help: 'Renames a player.',
 23   remote: true,
 24   action: function commandsRename() {
 25     var plr, playerIds, player
 26       , plrName = arguments[1]
 27       , newName = arguments[2];
 28 
 29     // Jos annettiinkin ID, niin vaihdetaan se nimeksi
 30     if (this.players[plrName]) {
 31       plrName = this.players[plrName].name;
 32     }
 33 
 34     // Luupataan kaikki pelaajat ja etsitään haluamamme pelaaja.
 35     playerIds = Object.keys(this.players);
 36     for (var i = playerIds.length; i--;) {
 37       plr = this.players[playerIds[i]];
 38       plr.sendNames = true; // Kaikkien pitää päivittää nimitiedot
 39       // Ei anneta vaihtaa nimeä, jos uusi nimi on jo käytössä
 40       if (plr.name === newName) {
 41         log.notice('Name "%0" is already in use!', newName.green);
 42         return;
 43       }
 44       if (plr.name === plrName && !plr.zombie && plr.active) {
 45         player = plr;
 46       }
 47     }
 48 
 49     // Jos pelaaja löytyi, vaihdetaan nimi.
 50     if (!player) {
 51       log.notice('Sorry, player couldn\'t be found or you tried to rename a bot.');
 52     } else {
 53       player.name = newName;
 54       log.info('Renamed "%0" -> "%1" >:)', plrName.green, newName.green);
 55     }
 56   }
 57   /**#nocode-*/
 58 };
 59 
 60 module.exports = rename;
