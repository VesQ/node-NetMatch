  5 "use strict";

  7 /**
  8  * Kertoo kauanko palvelin on ollut päällä.
  9  * @methodOf Commands
 10  */
 11 var uptime = {
 12   /**#nocode+*/
 13   params: [],
 14   help: 'Displays how long has the server been running.',
 15   remote: true,
 16   action: function commandsUptime() {
 17     var t = secondsToTime(Math.floor(process.uptime()));
 18     this.serverMessage('This server has been running for ' + t.d +
 19       ' days ' + t.h + ' hours ' + t.m + ' minutes and ' + t.s + ' seconds.');
 20   }
 21   /**#nocode-*/
 22 };
 23 
 24 
 25 // http://codeaid.net/javascript/convert-seconds-to-hours-minutes-and-seconds-(javascript) Vähän editoituna
 26 /** @ignore */
 27 function secondsToTime(secs) {
 28   var days = Math.floor(secs / (60 * 60 * 24))
 29     , divisor_for_hours = secs % (60 * 60 * 24)
 30     , hours = Math.floor(divisor_for_hours / (60 * 60))
 31     , divisor_for_minutes = secs % (60 * 60)
 32     , minutes = Math.floor(divisor_for_minutes / 60)
 33     , divisor_for_seconds = divisor_for_minutes % 60
 34     , seconds = Math.ceil(divisor_for_seconds);
 35   return {"d": days, "h": hours, "m": minutes, "s": seconds};
 36 }
 37 
 38 module.exports = uptime;
