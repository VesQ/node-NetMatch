1 /**
  2  * @fileOverview Sisältää {@link Registration}-luokan toteutuksen, eli siis palvelimen lisäämisen
  3  * ja päivittämisen netissä olevaan palvelinlistaukseen.
  4  */
  5 
  6 /**#nocode+*/
  7 var log = require('./Utils').log
  8   , http = require('cbNetwork').HTTP;
  9 /**#nocode-*/
 10 
 11 /**
 12  * Alustaa moduulin. Kun haluat lisätä palvelimen palvelinlistaukseen, käytä metodia {@link #add}.
 13  * @class Palvelinlistauksen päivittelyyn liittyvät toiminnot.
 14  *
 15  * @param {Server} server  Tähän rekisteröintiin liittyvä NetMatch-palvelin.
 16  *
 17  * @property {Boolean} registered  Onko palvelin rekisteröity vai ei
 18  */
 19 function Registration(server) {
 20   this.server = server;
 21   this.registered = false;
 22   this.updateWait = undefined;
 23 }
 24 
 25 /**
 26  * Pyytää päästä palvelinlistaukseen. Kutsuu automaattisesti metodia {@link #update}.
 27  *
 28  * @param {Function} [callback]  Tätä funktiota kutsutaan, kun rekisteröinti on saatu päätökseen.
 29  *                               Jos rekisteröinti epäonnistui, annetaan funktiolle parametrina
 30  *                               virheilmoitus merkkijonona.
 31  */
 32 Registration.prototype.apply = function (callback) {
 33   if (this.registered) {
 34     callback('Server already registered.');
 35     return;
 36   }
 37 
 38   log.info('Registering server to %0...', this.server.config.regHost.green);
 39   // Luodaan pyyntö
 40   var reg = this
 41     , config = this.server.config
 42     , url = config.regHost
 43     + config.regPath
 44     + '?profile=' + encodeURI('NetMatch')
 45     + '&ver='     + this.server.VERSION
 46     + '&mode=reg'
 47     + '&desc=' + encodeURI(config.description)
 48     + '&addr=' + encodeURI(config.address)
 49     + '&port=' + config.port
 50     + (config.devBuild ? '&devbuild=1' : '');
 51 
 52   // Lähetetään pyyntö
 53   http.get(url, function registerGet(status, data) {
 54     log.debug(data);
 55     if (status !== 200 || data !== 'ok') {
 56       callback('[REG]'.red + ' Server returned: ' + data.red);
 57       return;
 58     }
 59   });
 60 
 61   // Odotellaan vastausta
 62   this.server.on('register', function registerReply() {
 63     // Rekisteröinti onnistui!
 64     this.registration.registered = true;
 65     // Lähetetään palvelimen tiedot listausta varten
 66     this.registration.update(function (e) {
 67       if (!e) {
 68         // Kaikki meni hyvin käynnistetään päivitys-luuppi.
 69         reg.updateWait = setInterval(function updateRegInterval() {
 70           reg.server.registration.update();
 71         }, 60000);
 72         callback();
 73       } else { callback(e); }
 74     });
 75   });
 76 };
 77 
 78 
 79 /**
 80  * Pyytää, että poistetaan palvelin listauksesta.
 81  *
 82  * @param {Function} [callback]  Tätä funktiota kutsutaan, kun epä-rekisteröinti on saatu päätökseen.
 83  *                               Jos epä-rekisteröinti epäonnistui, annetaan funktiolle parametrina
 84  *                               virheilmoitus merkkijonona.
 85  */
 86 Registration.prototype.remove = function (callback) {
 87   if (!this.registered) {
 88     callback('Server is not registered!');
 89     return;
 90   }
 91 
 92   // Lopetetaan päivittäminen
 93   clearTimeout(this.updateWait);
 94 
 95   var server = this.server
 96     , config = server.config
 97     // Luodaan pyyntö
 98     , url = config.regHost
 99     + config.regPath
100     + '?profile=' + encodeURI('NetMatch')
101     + '&addr=' + config.address
102     + '&port=' + config.port
103     + '&mode=unreg';
104 
105   // Lähetetään pyyntö
106   http.get(url, function (status, data) {
107     // Katsotaan, että kaikki meni putkeen
108     if (status !== 200 || data !== 'ok') {
109       callback('[UNREG]'.red + ' Server returned: ' + data.red);
110     } else { callback(); }
111   });
112 };
113 
114 
115 /**
116  * Päivittää palvelimen listauksen.
117  *
118  * @param {Function} [callback]  Tätä funktiota kutsutaan, kun päivitys on saatu päätökseen.
119  *                               Jos päivitys epäonnistui, annetaan funktiolle parametrina
120  *                               virheilmoitus merkkijonona.
121  */
122 Registration.prototype.update = function (callback) {
123   if ('function' !== typeof callback) {
124     /** @ignore */
125     callback = function (e) {
126       if (e) {
127         log.debug(e);
128       }
129     };
130   }
131   if (!this.registered) {
132     callback('Server is not registered, can not update!');
133     return;
134   }
135   log.debug('Called update');
136 
137   // Temppimuuttujia
138   var reg = this, srv = reg.server, config = srv.config, state = srv.gameState
139     // Listataan pelaajien nimet
140     , plrNames = Object.keys(srv.players)
141       .filter(function (p) { // Filtteröidään inaktiiviset
142          p = srv.players[p]; return p.active; })
143       .map(function (p) { return srv.players[p].name; })
144     , plrs = plrNames.length
145     // Luodaan merkkijono, jossa on palvelimen tiedot
146     , data =
147       [ plrs
148       , state.botCount
149       , state.map.name
150       , state.maxPlayers
151       , encodeURI(plrNames.join('|'))
152       ].join(',')
153     // Luodaan pyyntö
154     , url = config.regHost + config.regPath
155     + '?profile=NetMatch&mode=update'
156     + '&addr=' + config.address + '&port=' + config.port
157     + '&data=' + data;
158 
159   // Lähetetään pyyntö
160   http.get(url, function (status, data) {
161     // Katsotaan, että kaikki meni putkeen
162     if (status !== 200 || data !== 'ok') {
163       callback('[UPD]'.red + ' Server returned: ' + data.red);
164     } else {
165       callback();
166     }
167   });
168 };
169 
170 module.exports = Registration;
