 20 /**
 21  * Luo uuden palvelimen annettuun porttiin ja osoitteeseen. Kun palvelimeen tulee dataa, emittoi
 22  * se <i>message</i> eventin, kts. {@link NetMatch#event:message}
 23  *
 24  * @class Yleiset NetMatchin funktiot ja ominaisuudet
 25  *
 26  * @param {Object} c            Asetukset, kts. {@link NetMatch.config}
 27  * @param {Number} c.port       Portti, jota palvelin kuuntelee.
 28  * @param {String} [c.address]  IP-osoite, jota palvelin kuuntelee. Jos tätä ei anneta, järjestelmä
 29  *                              yrittää kuunnella kaikkia osoitteita.
 30  */
 31 function NetMatch(c) {
 32   if('object' !== typeof c || !c.hasOwnProperty('port')) {
 33     console.log('Starting NetMatch-server failed!');
 34     return false;
 35   }
 36   /**
 37    * cbNetwork-node UDP-palvelin
 38    * @type cbNetwork.Server
 39    * @see <a href="http://vesq.github.com/cbNetwork-node/doc/symbols/Server.html">cbNetwork.Server</a>
 40    */
 41   this.server = new cbNetwork.Server(c.port, c.address);
 42   
 43   var self = this;
 44   this.server.on('message', function (client) {
 45     self.emit('message', client);
 46   });
 47   
 48   /**
 49    * Asetukset tälle palvelimelle
 50    * @type Object
 51    * @see NetMatch.config
 52    */
 53   this.config = NetMatch.config;
 54   
 55   // Asetetaan nykyisen pelin tila
 56   this.gameState.playerCount = 0;
 57   this.gameState.gameMode = this.config.gameMode;
 58   this.gameState.map = this.config.map;
 59   this.gameState.mapCRC = -1170754068; // TODO: häx
 60   this.gameState.maxPlayers = this.config.maxPlayers;
 61   
 62   // Nyt alustetaan pelaajat
 63   this.players = {};
 64   for(var i = 1; i <= this.config.maxPlayers; ++i) {
 65     var pl = new Player();
 66     pl.playerId = i;
 67     pl.active = false;
 68     pl.team = 1;
 69     pl.botName = "Bot_" + i;
 70     pl.clientId = "";
 71     pl.active = false;
 72     pl.loggedIn = false;
 73     pl.name = "";
 74     var skill = 21 - pl.playerId; // Botteja hieman eritasoisiksi vissiinkin?
 75     pl.fightRotate = 1.5 + (skill / 1.5);
 76     pl.shootingAngle = 4.0 + (pl.playerId * 1.5);
 77     pl.fov = 100 + (skill * 3.5);
 78     pl.hasAmmos = false;
 79     pl.admin = false;
 80     pl.kicked = false;
 81     pl.kickReason = "";
 82     
 83     this.players[i] = pl;
 84   }
 85   
 86   //console.log(this.players);
 87 }

176  *   // Lähetetään clientille sen lähettämä data takaisin

178  * });
179  *

186 NetMatch.prototype.login = function (client) {
187   var data = client.data
188     , replyData
189     , version = data.getString()
190     , nickname
191     , playerIds;
192   
193   // Täsmääkö clientin ja serverin versiot
194   if (version !== NetMatch.VERSION) {
195     // Eivät täsmää, lähetetään virheilmoitus
196     replyData = new Packet(3);
197     replyData.putByte(NET.LOGIN);
198     replyData.putByte(NET.LOGINFAILED);
199     replyData.putByte(NET.WRONGVERSION);
200     replyData.putString(NetMatch.VERSION);
201     client.reply(replyData);
202     return false;
203   }
204   
205   // Versio on OK, luetaan pelaajan nimi
206   nickname = data.getString().trim();
207   console.log('Player "' + nickname + '" is trying to connect...');
208   
209   // Käydään kaikki nimet läpi ettei samaa nimeä vain ole jo suinkin olemassa
210   playerIds = Object.keys(this.players);
211   for (var i = playerIds.length; i--;) {
212     var player = this.players[playerIds[i]];
213     if (player.name.toLowerCase() == nickname) {
214       if (player.kicked || !player.active) {
215         player.name = "";
216       } else {
217         // Nimimerkki oli jo käytössä.
218         replyData = new Packet(3);
219         replyData.putByte(NET.LOGIN);
220         replyData.putByte(NET.LOGINFAILED);
221         replyData.putByte(NET.NICKNAMEINUSE);
222         client.reply(replyData);
223         return false;
224       }
225     }
226   }
227   
228   // Etsitään vapaa "pelipaikka"
229   for (var i = playerIds.length; i--;) {
230     var player = this.players[playerIds[i]];
231     if (this.gameState.playerCount < this.config.maxPlayers && !player.active) {
232       // Tyhjä paikka löytyi
233       player.clientId = client.address + ':' + client.id;
234       player.active = true;
235       player.loggedIn = false;
236       player.name = nickname;
237       // UNIMPLEMENTED
238       player.x = Math.round(-100 + Math.random() * 200);
239       player.y = Math.round(-100 + Math.random() * 200);
240       player.hackTestX = player.x;
241       player.hackTestY = player.y;
242       player.angle = Math.floor(Math.random() * 360 + 1);
243       player.zombie = false;
244       player.health = 100;
245       player.kills = 0;
246       player.deaths = 0;
247       player.weapon = WPN.PISTOL;
248       player.lastActivity = new Date().getTime();
249       player.spawnTime = player.lastActivity;
250       player.admin = false;
251       player.kicked = false;
252       player.kickReason = "";
253       if (this.gameState.gameMode == "TDM") {
254         // UNIMPLEMENTED
255         // Tasainen jako joukkueihin
256         player.team = Math.floor(Math.random()*2 + 1) + 1; // Rand(1,2)
257       }
258       
259       // Lähetetään vastaus clientille
260       replyData = new Packet(16);
261       replyData.putByte(NET.LOGIN);
262       replyData.putByte(NET.LOGINOK);
263       replyData.putByte(player.playerId);
264       replyData.putByte(this.gameState.gameMode);
265       replyData.putString(this.gameState.map);
266       replyData.putInt(this.gameState.mapCRC);
267       // UNIMPLEMENTED
268       replyData.putString(""); // Kartan URL josta sen voi ladata, mikäli se puuttuu
269       client.reply(replyData);
270       console.log(player.name + " logged in (" + client.address + ")");
271       
272       // Lisätään viestijonoon ilmoitus uudesta pelaajasta
273       var msgData = {
274         msgType: NET.LOGIN,
275         msgText: nickname,
276         playerId: player.playerId,
277         playerId2: player.zombie
278       };
279       for (var j = playerIds.length; j--;) {
280         var plr = this.players[playerIds[j]];
281         // Lähetetään viesti kaikille muille paitsi boteille ja itselle
282         if (plr.active && !plr.zombie && (plr.playerId != player.playerId)) {
283           NetMessages.add(plr.playerId, msgData);
284         }
285       }
286       return true;
287     }
288   }
289   
290   // Vapaita paikkoja ei ollut
291   console.log('No free slots.');
292   replyData = new Packet(3);
293   replyData.putByte(NET.LOGIN);
294   replyData.putByte(NET.LOGINFAILED);
295   replyData.putByte(NET.TOOMANYPLAYERS);
296   client.reply(replyData);
297   return false;
298 }
