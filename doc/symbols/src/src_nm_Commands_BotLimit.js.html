1 /**
  2  * @fileOverview Toteutus komennolle botlimit: {@link Commands.botlimit}
  3  */
  4 
  5 "use strict";
  6 
  7 /**
  8  * Kertoo kauanko palvelin on ollut päällä.
  9  * @methodOf Commands
 10  */
 11 var botlimit = {
 12   /**#nocode+*/
 13   params: [
 14     {name: 'count',  type: 'integer', optional: true, help: 'When this amount of players is reached, bots start to leave'}
 15   ],
 16   help: 'Sets the bot depart limit',
 17   remote: true,
 18   action: function commandsBotLimit() {
 19     var msg, value;
 20     // Jos ei annettu parametreja, niin tulostetaan nykyinen bottilimitti. arguments[0] on kutsuja
 21     if (arguments.length < 2) {
 22       msg = 'Current bot depart limit is ' + this.gameState.botDepartLimit + ' and current bot count is ' + this.gameState.botCount;
 23       if (arguments[0]) {
 24         this.serverMessage(msg, arguments[0]);
 25       } else {
 26         console.log(msg);
 27       }
 28       return;
 29     }
 30     value = arguments[1];
 31     if (value >= 0) {
 32       this.gameState.botDepartLimit = value;
 33     } else {
 34       // Jos annettu arvo oli pienempi kuin nolla niin käytetään kartan asetuksissa määriteltyä
 35       // arvoa, jos se on olemassa. Muulloin 0
 36       if (this.gameState.map.config && this.gameState.map.config.botDepartLimit > 0) {
 37         this.gameState.botDepartLimit = this.gameState.map.config.botDepartLimit;
 38       } else {
 39         this.gameState.botDepartLimit = 0;
 40       }
 41     }
 42   }
 43   /**#nocode-*/
 44 };
 45 
 46 module.exports = botlimit;
