  4 
  5 var log = require('../Utils').log;
  6 
  7 /**
  8  * Suorittaa komennon toisena pelaajana. >:)
  9  * @methodOf Commands
 10  *
 11  * @param {player}  as   Kenenä komentoa kutsutaan
 12  * @param {command} cmd  Mitä komentoa kutsutaan
 13  * @param {string} args  komennon parametrit
 14  */
 15 var AsPlayer = {
 16   /**#nocode+*/
 17   params: [
 18     {name: 'as',   type: 'player',  optional: false, help: 'Player who to call as'},
 19     {name: 'cmd',  type: 'command', optional: false, help: 'Command to call'},
 20     {name: 'args', type: 'string',  optional: true,  help: 'Command\'s arguments.'}
 21   ],
 22   help: 'Calls a command as specified player.',
 23   remote: true,
 24   action: function commandsAsPlayer() {
 25     var plr = this.getPlayer(arguments[1])
 26       , cmd = arguments[2]
 27       , args =  Array.prototype.slice.call(arguments).splice(3);
 28     if (!plr) { log.warn('Couldn\'t find player "%0"', arguments[1]); return; }
 29     log.info('Calling `%0` as "%1" with arguments `%2`', cmd, plr.name, args.join(' ') || 'none');
 30     this.commands.call(cmd, args, plr);
 31   }
 32   /**#nocode-*/
 33 };
 34 
 35 module.exports = AsPlayer;
