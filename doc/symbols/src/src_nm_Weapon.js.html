144 /**
145  * Luo uuden ammuksen.
146  *
147  * @class Ammus
148  *
149  * @param {Server} server     Palvelimen sisältävä muuttuja, instanssi luokasta {@link Server}.
150  * @param {Integer} playerId  Pelaajan, joka ampui ammuksen, ID
151  * @param {Integer} [extraBullet=0]
152  *
153  * @property {Short}   bulletId     Ammuksen tunnus
154  * @property {Integer} playerId     Kuka ampui
155  * @property {Byte}    weapon       Millä aseella tämä on ammuttu
156  * @property {Integer} x            Paikkatietoa
157  * @property {Integer} y            Paikkatietoa
158  * @property {Integer} angle        Ammuksen kulma
159  * @property {Float}   prevPosX     Ammuksen edellinen paikka
160  * @property {Float}   prevPosY     Ammuksen edellinen paikka
161  * @property {Byte}    moved        Onko ammusta vielä liikutettu yhtään
162  * @property {Integer} timeShooted  Milloin ammus on ammuttu
163  */
164 function Bullet(server, playerId, extraBullet) {
165   if ('undefined' === typeof extraBullet) {
166     extraBullet = 0;
167   }
168   var player
169     , weaponConfig
170     , bPos
171     , spread
172     , randomSpread
173     , msgData
174     , playerIds
175     , plr;
176 
177   // UNIMPLEMENTED
178   // Ei tehdä ammusta jos erä on päättynyt
179 
180   // Haetaan pelaaja joka ampui
181   player = server.players[playerId];
182   // Luodaan ammus
183   this.bulletId = ++server.lastBulletId;  // Ammuksen tunnus
184   this.weapon = player.weapon;            // Millä aseella ammuttu
185   this.playerId = playerId;               // Kuka ampui
186   this.timeShooted = timer();             // Koska ammuttu
187   this.moved = 0;
188   this.x = player.x;
189   this.y = player.y;
190   this.angle = player.angle;
191 
192   weaponConfig = Weapons[player.weapon];
193 
194   // Mistä ammus lähtee pelaajaan nähden
195   bPos = weaponConfig.bulletYaw;
196   if (player.weapon === WPN.PISTOL) {
197     if (!player.handShooted) {
198       bPos = -bPos;
199       player.handShooted = 1;
200     } else {
201       player.handShooted = 0;
202     }
203   }
204   this.x += Math.cos((this.angle) / 180 * Math.PI) * weaponConfig.bulletForth;
205   this.y += Math.sin((this.angle) / 180 * Math.PI) * weaponConfig.bulletForth;
206   this.x += Math.cos((this.angle - 90) / 180 * Math.PI) * bPos;
207   this.y += Math.sin((this.angle - 90) / 180 * Math.PI) * bPos;
208   this.prevPosX = this.x;
209   this.prevPosY = this.y;
210 
211   // Hajonta
212   spread = weaponConfig.spread;
213   if (player.weapon === WPN.LAUNCHER) {
214     if (extraBullet) {
215       spread = -spread;
216     }
217   } else {
218     spread = rand(-spread, spread);
219   }
220 
221   this.angle += spread;
222   if (player.weapon === WPN.SHOTGUN) {
223     randomSpread = rand(0, 20);
224     this.x += Math.cos((this.angle / 180) * Math.PI) * randomSpread;
225     this.y += Math.sin((this.angle / 180) * Math.PI) * randomSpread;
226   }
227 
228   // Tarkista että onko ammus kartan sisällä ja liikuttele taaksepäin kunnes ei ole
229   for (var i = 50; i--;) {
230     if (!server.gameState.map.isColliding(this.x, this.y)) {
231       break;
232     }
233     this.x -= Math.cos((this.angle / 180) * Math.PI);
234     this.y -= Math.sin((this.angle / 180) * Math.PI);
235   }
236   // Jos mentiin edellinen for-looppi loppuun asti, niin failataan ammuksen luonti.
237   if (i < 0) {
238     log.warn("Player %0 tried to shoot a bullet while inside of a wall!", player.name);
239     server.lastBulletId--;
240     return;
241   }
242 
243   // Lisätään palvelimen bullets-listaan tämä ammus
244   server.bullets.push(this);
245 
246   // Lisätään ammusviesti lähetettäväksi jokaiselle pelaajalle
247   msgData = {
248     msgType: NET.NEWBULLET,
249     bulletId: this.bulletId,
250     sndPlay: !extraBullet,
251     weapon: player.weapon,
252     playerId: playerId,
253     x: this.x,
254     y: this.y,
255     handShooted: player.handShooted
256   };
257   playerIds = Object.keys(server.players);
258   for (var i = playerIds.length; i--;) {
259     plr = server.players[playerIds[i]];
260     if (plr.active && !plr.zombie) {
261       server.messages.add(plr.playerId, msgData);
262     }
263   }
264 
265   if (player.weapon === WPN.LAUNCHER && !extraBullet) {
266     // Jos tämä on ammuttu kranaatinheittimellä niin tehdään vielä toinen ammus koska 2 kranaattia
267     new Bullet(server, playerId, 1);
268   }
269 
270   // Haulikosta lähtee monta kutia
271   if (player.weapon === WPN.SHOTGUN && extraBullet < 6) {
272     new Bullet(server, playerId, extraBullet + 1);
273   }
274 }
