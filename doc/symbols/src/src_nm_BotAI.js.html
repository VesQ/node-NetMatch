 23 /**
 24  * Alustaa uuden tekoälyn.
 25  * @class Bottien tekoäly
 26  *
 27  * @param {Server} server  NetMatch-palvelin
 28  * @param {Player} player  {@link Player}-luokan instanssi johon tämä tekoäly kuuluu.
 29  *
 30  * @property {Player}   player        Player-luokan instanssi johon tämä tekoäly kuuluu.
 31  * @property {Number}   nextAngle     Kulma johon botin pitäisi liikkua
 32  * @property {Number}   lastAngle     Kuinka paljon botin ja kohdekulman ero oli edellisellä kierroksella
 33  * @property {Number}   rotation      Tällä käännetään objektia (TurnObject)
 34  * @property {Number}   sideStep      Botin sivuaskeleet
 35  * @property {Number}   nextAction    Ajankohta jolloin botille arvotaan uusi suunta
 36  * @property {Boolean}  tooClose      Botti on liian lähellä seinää tai toista pelaajaa
 37  */
 38 function BotAI(server, player) {
 39   this.server = server;
 40   this.player = player;
 41   this.nextAngle = 0;
 42   this.lastAngle = 0;
 43   this.rotation = 0;
 44   this.sideStep = 0;
 45   this.nextAction = 0;
 46   this.tooClose = false;
 47   this.fightRotate = 0;
 48   this.shootingAngle = 0;
 49 
 50   // Alustetaan botin tekoälyä ohjaavat asetukset
 51   this.initBotAI();
 52 }
 53 
 54 /**
 55  * Alustaa botin tekoälyä ohjaavat asetukset.
 56  */
 57 BotAI.prototype.initBotAI = function () {
 58   /** Koska aikaisintaan arvotaan botille uusi suunta */
 59   this.minNextAction = 500;
 60 
 61   /** Koska viimeistään arvotaan botille uusi suunta */
 62   this.maxNextAction = 1000;
 63 
 64   /** Kun botille arvotaan uusi kääntyminen niin tämä on maksimi. */
 65   this.randRotation = 90;
 66 
 67   /** Botin nopeus kun se on havainnut esteen */
 68   this.minSpeed = 80;
 69 
 70   /** Botin nopeus kun tie on vapaa */
 71   this.maxSpeed = 200;
 72 
 73   /** Jos matkaa esteeseen on vähemmän kuin tämä niin aletaan etsiä uutta suuntaa */
 74   this.wakeupDist = 100;
 75 
 76   /**
 77    * Kun pitää päätellä uusi suunta niin se tehdään katselemalla näin monta astetta molempiin
 78    * suuntiin.
 79    */
 80   this.exploreAngle = 50;
 81 
 82   /**
 83    * Kun botti on lähellä estettä niin tällä määrätään kuinka jyrkällä käännöksellä yritetään
 84    * väistää. Pienempi arvo on jyrkempi käännös.
 85    */
 86   this.dodgeRotation = 0.2;
 87 
 88   /** Kuinka kaukaa voidaan reagoida viholliseen (pikseleissä) */
 89   this.fightDist = 900;
 90 
 91   /** Botin field of vision, kuinka leveästä kulmasta voidaan poimia pelaaja (0-180 astetta) */
 92   this.fov = 70;
 93 
 94   /** Kuinka nopeasti botti kääntyy, kun se näkee pelaajan */
 95   this.fightRotate = 12.5;
 96 
 97   /** Kun botti on pelaajaan nähden tässä kulmassa, aloittaa se ampumisen */
 98   this.shootingAngle = 32;
 99 };
100 
101 /**
102  * Muuttaa botin asetuksia riippuen annetusta parametrista.
103  *
104  * @param {Number} param  Arvo väliltä 1...64, jonka perusteella botin taitoja rukataan.
105  */
106 BotAI.prototype.setSkill = function (param) {
107   if (param < 1) { param = 1; }
108   else if (param > 64) { param = 64; }
109   var skill = 65 - param;
110   this.fightRotate = 1.5 + (skill / 4.8);
111   this.shootingAngle = 4.0 + (param * 4.8);
112   this.fov = 58 + (skill * 2);
113 };

115 /**
116  * Yksittäisen botin tekoälyn päivitys.
117  */
118 BotAI.prototype.update = function () {
119   var self = this
120     , map = this.server.gameState.map
121     , minDist
122     , picker
123     , pickWall
124     , dist1, dist2, dodge
125     , objectiveAngle
126     , moveDirection = 1
127     , pickedPlayer, pickedDist, pickedDirection
128     , weaponConfig = Weapons[this.player.weapon]
129     , sAngle, isProtected, reloadFactor
130     , speed, forwardSpeed, sidestepSpeed;
131   // Mikäli botti ei ole liian lähellä seinää ja on aika arpoa sille uusi suunta
132   // niin tehdään se nyt.
133   if (!this.tooClose && this.nextAction < Date.now()) {
134     // Arvotaan seuraavan käännöksen ajankohta eli koska tähän iffiin tullaan seuraavaksi
135     this.nextAction = rand(this.minNextAction, this.maxNextAction) + Date.now();
136     // Arvotaan botille uusi kääntyminen
137     this.rotation = rand(-90, 90, true);
138 
139     if (this.sideStep !== 0) {
140       this.sideStep = rand(-1, 1, true);
141     }
142   }
143 
144   // Käännetään bottia
145   if (this.player.zombie) {
146     this.player.turn(this.server.game.movePerSec(this.rotation));
147   }
148 
149   // Hoidetaan bottien varsinainen tekoäly, kun estettä ei ole liian lähellä.
150   if (!this.tooClose) {
151     // Lasketaan etäisyys edessä olevaan esteeseen.
152     // Etäisyys lasketaan objektin keskeltä sekä reunoista eli objektin koko leveydeltä.
153 
154     // Luodaan väliaikainen poimintaobjekti
155     picker = new Obj(this.player.x, this.player.y, this.player.angle);
156 
157     // Poimitaan botin nenän edestä mahdollinen seinä
158     pickWall = map.findWall(picker.x, picker.y, picker.angle, this.wakeupDist);
159     this.debugRayCast(pickWall, picker, true);
160     if (pickWall) {
161       minDist = distance(picker.x, picker.y, pickWall.x, pickWall.y);
162     }
163 
164     // Poimitaan botin vasemman reunan puolelta seinä
165     picker.move(0, -15);
166     pickWall = map.findWall(picker.x, picker.y, picker.angle, this.wakeupDist);
167     this.debugRayCast(pickWall, picker, true);
168     if (pickWall) {
169       minDist = Math.min(minDist || 99999, distance(picker.x, picker.y, pickWall.x, pickWall.y));
170     }
171 
172     // Poimitaan botin oikean reunan puolelta seinä
173     picker.move(0, 30);
174     pickWall = map.findWall(picker.x, picker.y, picker.angle, this.wakeupDist);
175     this.debugRayCast(pickWall, picker, true);
176     if (pickWall) {
177       minDist = Math.min(minDist || 99999, distance(picker.x, picker.y, pickWall.x, pickWall.y));
178     }
179 
180     // Jos tarpeeksi lähellä on seinä niin reagoidaan siihen nyt
181     if (minDist) {
182       // Käännetään poimintaobjekti samaan kulmaan kuin botti
183       picker.angle = this.player.angle;
184 
185       // Käännetään katsetta toiselle sivulle ja lasketaan etäisyys lähimpään esteeseen
186       picker.turn(-this.exploreAngle);
187       pickWall = map.findWall(picker.x, picker.y, picker.angle);
188       this.debugRayCast(pickWall, picker, true);
189       if (pickWall) {
190         dist1 = distance(picker.x, picker.y, pickWall.x, pickWall.y);
191       } else {
192         // Ei luultavasti ylitetty minkään laatan rajoja kun törmäystä tarkistettiin.
193         dist1 = 0;
194       }
195 
196       // Ja sitten vielä toiseen suuntaan.
197       picker.turn(this.exploreAngle * 2);
198       pickWall = map.findWall(picker.x, picker.y, picker.angle);
199       this.debugRayCast(pickWall, picker, true);
200       if (pickWall) {
201         dist2 = distance(picker.x, picker.y, pickWall.x, pickWall.y);
202       } else {
203         // Ei luultavasti ylitetty minkään laatan rajoja kun törmäystä tarkistettiin.
204         dist2 = 0;
205       }
206 
207       // Tutkitaan kumpaan suuntaan on pidempi matka seuraavaan esteeseen ja suunnataan sinne.
208       // Kääntymisen jyrkkyyteen vaikuttaa vielä etäisyys esteeseen eli mitä lähempänä ollaan niin
209       // sitä jyrkemmin käännytään
210       if (dist1 > dist2) {
211         dodge = -(this.wakeupDist - minDist);
212       } else {
213         dodge = this.wakeupDist - minDist;
214       }
215 
216       // Asetetaan kääntymisnopeus
217       this.rotation = dodge / this.dodgeRotation;
218       // Asetetaan tavoitekulma
219       this.nextAngle = wrapAngle(this.player.angle + dodge);
220       // Asetetaan vielä tooClose-muuttuja päälle eli tekoälyä ei päivitetä ennen kuin objekti on
221       // kääntynyt tavoitekulmaan. Samalla myös objektin nopeutta vähennetään.
222       this.tooClose = true;
223 
224       this.lastAngle = this.player.angle - this.nextAngle;
225       if (this.lastAngle > 180) { this.lastAngle -= 360; }
226       if (this.lastAngle < -180) { this.lastAngle += 360; }
227     }
228   } else {
229     // Botti on liian lähellä jotain estettä.
230     // tooClose-muuttuja nollataan vain jos tekoälyn asettama tavoitekulma on saavutettu.
231     objectiveAngle = this.player.angle - this.nextAngle;
232     if (objectiveAngle > 180) { objectiveAngle -= 360; }
233     if (objectiveAngle < -180) { objectiveAngle += 360; }
234     if ((objectiveAngle < 0 && this.lastAngle >= 0) || (objectiveAngle > 0 && this.lastAngle <= 0)) {
235       // Objektin kulma on nyt riittävän lähellä tavoitekulmaa joten kääntäminen voidaan lopettaa.
236       this.rotation = 0;
237       this.tooClose = false;
238     }
239     this.lastAngle = objectiveAngle;
240   }
241 
242   // Taisteluäly
243   /** @ignore */
244   this.server.loopPlayers( function botAIFight(player) {
245     // Käydään läpi kaikki muut pelaajat
246     var dist, direction;
247     if (self.server.gameState.gameMode > 1 && player.team === self.player.team) {
248       // Jos ollaan jossain muussa pelimoodissa kuin DM ja pelaaja on samalla puolella kuin botti,
249       // ei tarkistettava pelaaja ole vihollinen. Siirrytään seuraavaan.
250       return;
251     }
252     if (player.id === self.player.id || player.health <= 0 || !player.loggedIn) {
253       // Ei tarkisteta bottia itseään, kuollutta pelaajaa tai pelaajaa joka ei ole kirjautuneena.
254       return;
255     }
256     dist = distance(self.player.x, self.player.y, player.x, player.y);
257     if (dist > self.fightDist) {
258       // Pelaaja on liian kaukana, ei reagoida.
259       return;
260     }
261     if (self.server.gameState.map.findWall2(self.player.x, self.player.y, player.x, player.y)) {
262       // Oli seinä välissä, joten ei reagoida pelaajaan.
263       return;
264     }
265     if (pickedDist && pickedDist < dist) {
266       // Bottia lähempänä on joku toinen pelaaja.
267       return;
268     }
269     direction = getAngle(player.x, player.y, self.player.x, self.player.y) - self.player.angle;
270     if (direction > 180) { direction -= 360; }
271     if (direction < -180) { direction += 360; }
272     if (Math.abs(direction) > self.fov) {
273       // Pelaaja oli näkökentän ulkopuolella, ei reagoida.
274       return;
275     }
276     // Nyt kaikki tarkistukset on tehty ja voidaan poimia pelaaja.
277     pickedPlayer = player;
278     pickedDist = dist;
279     pickedDirection = direction;
280   });
281   // Onko joku uhri näkösällä
282   if (pickedPlayer) {
283     // Piirretään uhriin viiva
284     this.debugLine(this.player.x, this.player.y, pickedPlayer.x, pickedPlayer.y);
285     if (this.sideStep === 0) {
286       this.sideStep = rand(-1, 1, true);
287     }
288     // Nollataan liikkumistekoäly
289     this.tooClose = false;
290     // Asetetaan kääntyminen kohti uhria
291     this.rotation = pickedDirection * this.fightRotate;
292     // Aseesta riippuen etäisyys kohteeseen ei saa olla liian pieni
293     weaponConfig = Weapons[this.player.weapon];
294     if (pickedDist < weaponConfig.safeRange) {
295       moveDirection = -1;
296     }
297     // Ammutaan vain jos kulma on riittävän pieni eikä pelaajalle ole nakkina
298     sAngle = this.shootingAngle;
299     if (this.player.weapon === WPN.CHAINSAW) { sAngle *= 2; }
300     if ((pickedPlayer.spawnTime + this.server.config.spawnProtection <= Date.now()) &&
301         (this.player.spawnTime + this.server.config.spawnProtection <= Date.now()) &&
302         Math.abs(pickedDirection) < sAngle &&
303         pickedDist > weaponConfig.safeRange / 2 && pickedDist <= weaponConfig.shootRange)
304     {
305       // Kaikki ampumiseen vaadittavat asiat ovat kunnossa!
306 
307       // Jos botilla on pistooli niin luodaan sille satunnaisuutta liipasunopeuteen
308       if (this.player.weapon === WPN.PISTOL) {
309         reloadFactor = rand(1.2, 2, true);
310       } else {
311         reloadFactor = 1;
312       }
313 
314       // Sitten vielä tarkistetaan että onko viime ampumisesta kulunut tarpeeksi aikaa
315       if (!this.player.lastShoot || this.player.lastShoot + weaponConfig.reloadTime * reloadFactor < Date.now()) {
316         // Jea, ammutaan!
317         this.server.createBullet(this.player);
318         this.player.lastShoot = Date.now();
319       }
320     }
321   }
322 
323   // Siirretään bottia
324   speed = this.maxSpeed;
325   // Jos ollaan liian lähellä jotain estettä niin pienemmällä vauhdilla
326   if (this.tooClose) { speed = this.minSpeed; }
327 
328   forwardSpeed = moveDirection * 100 / weaponConfig.weight;
329   forwardSpeed *= this.server.game.movePerSec(speed);
330   sidestepSpeed = this.sideStep * 100.0 / weaponConfig.weight;
331   sidestepSpeed *= this.server.game.movePerSec(PLR.SIDESTEP_SPEED * 0.8);
332   this.player.move(forwardSpeed, sidestepSpeed);
333 
334   // Törmäystarkistus
335   this.checkCollisions();
336 };

338 /**
339  * Törmäystarkistus, estää botin menemisen seinien sisään.
340  */
341 BotAI.prototype.checkCollisions = function () {
342   var collision = this.server.gameState.map.circleCollision(this.player.x, this.player.y, PLR.COL_RADIUS)
343     , collides;
344 
345   if (collision.left) {
346     //log.debug('Bot %0 is colliding from left', this.player.name);
347     if (this.player.x < this.player.lastValidX) {
348       this.player.x = this.player.lastValidX;
349     }
350     collides = true;
351   } else if (collision.right) {
352     //log.debug('Bot %0 is colliding from right', this.player.name);
353     if (this.player.x > this.player.lastValidX) {
354       this.player.x = this.player.lastValidX;
355     }
356     collides = true;
357   }
358   if (collision.up) {
359     //log.debug('Bot %0 is colliding from up', this.player.name);
360     if (this.player.y > this.player.lastValidY) {
361       this.player.y = this.player.lastValidY;
362     }
363     collides = true;
364   } else if (collision.down) {
365     //log.debug('Bot %0 is colliding from down', this.player.name);
366     if (this.player.y < this.player.lastValidY) {
367       this.player.y = this.player.lastValidY;
368     }
369     collides = true;
370   }
371 
372   this.player.lastValidX = this.player.x;
373   this.player.lastValidY = this.player.y;
374 
375   if (collides) {
376     return true;
377   }
378   return false;
379 };

381 /**
382  * Debuggaa raycastia
383  */
384 BotAI.prototype.debugRayCast = function (collision, picker, drawLineToCollision) {
385   if (true || !this.server.debug) { return; }
386   if (!collision && drawLineToCollision) { return; }
387   var self = this
388     , startX, startY, endX, endY, colX, colY;
389 
390   startX = Math.round(picker.x);
391   startY = Math.round(picker.y);
392   if (collision) {
393     colX = Math.round(collision.x);
394     colY = Math.round(collision.y);
395     if (drawLineToCollision) {
396       endX = colX;
397       endY = colY;
398     }
399   }
400   if (!collision || !drawLineToCollision) {
401     endX = Math.round(picker.x + Math.cos(picker.angle / 180 * Math.PI) * this.wakeupDist);
402     endY = Math.round(picker.y + Math.sin(picker.angle / 180 * Math.PI) * this.wakeupDist);
403   }
404 
405   this.server.loopPlayers(function (player) {
406     if (player.debugState) {
407       player.debugState += 1;
408       self.server.messages.add(player.id, {
409         msgType: NET.DEBUGDRAWING,
410         drawType: DRAW.LINE,
411         drawVars: [
412           startX,
413           startY,
414           endX,
415           endY
416         ]
417       });
418       if (drawLineToCollision || !collision) { return; }
419       self.server.messages.add(player.id, {
420         msgType: NET.DEBUGDRAWING,
421         drawType: DRAW.CIRCLE,
422         drawVars: [
423           colX,
424           colY,
425           5,
426           0
427         ]
428       });
429     }
430   });
431 };

433 /**
434  * Piirtää debug-viivan yhdestä pisteestä toiseen
435  */
436 BotAI.prototype.debugLine = function (x1, y1, x2, y2) {
437   if (true || !this.server.debug) { return; }
438   var self = this;
439   this.server.loopPlayers(function (player) {
440     if (player.debugState) {
441       player.debugState += 1;
442       self.server.messages.add(player.id, {
443         msgType: NET.DEBUGDRAWING,
444         drawType: DRAW.LINE,
445         drawVars: [
446           x1,
447           y1,
448           x2,
449           y2
450         ]
451       });
452     }
453   });
454 };
