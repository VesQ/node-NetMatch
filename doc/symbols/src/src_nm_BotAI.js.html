 19 /**
 20  * Alustaa uuden tekoälyn.
 21  * @class Bottien tekoäly
 22  *
 23  * @param {Server} server  NetMatch-palvelin
 24  * @param {Player} player  {@link Player}-luokan instanssi johon tämä tekoäly kuuluu.
 25  *
 26  * @property {Player}   player        Player-luokan instanssi johon tämä tekoäly kuuluu.
 27  * @property {Number}   nextAngle     Kulma johon botin pitäisi liikkua
 28  * @property {Number}   lastAngle     Kuinka paljon botin ja kohdekulman ero oli edellisellä kierroksella
 29  * @property {Number}   rotation      Tällä käännetään objektia (TurnObject)
 30  * @property {Number}   sideStep      Botin sivuaskeleet
 31  * @property {Number}   nextAction    Ajankohta jolloin botille arvotaan uusi suunta
 32  * @property {Boolean}  tooClose      Botti on liian lähellä seinää tai toista pelaajaa
 33  * @property {Number}   fightRotate   Kääntymisnopeus kun ollaan havaittu vastustaja
 34  * @property {Number}   shootingAngle Ammutaan kun uhri on tämän kulman sisällä botin edessä
 35  * @property {Object}   config        Botin tekoälyä ohjaavat asetukset: {@link BotAI#initConfig}
 36 */
 37 function BotAI(server, player) {
 38   this.server = server;
 39   this.player = player;
 40   this.nextAngle = 0;
 41   this.lastAngle = 0;
 42   this.rotation = 0;
 43   this.sideStep = 0;
 44   this.nextAction = 0;
 45   this.tooClose = false;
 46   this.fightRotate = 0;
 47   this.shootingAngle = 0;
 48   this.config = {};
 49 
 50   // Alustetaan botin tekoälyä ohjaavat asetukset
 51   this.initConfig();
 52 }
 53 
 54 /**
 55  * Alustaa botin tekoälyä ohjaavat asetukset.
 56  */
 57 BotAI.prototype.initConfig = function () {
 58   /** Koska aikaisintaan arvotaan botille uusi suunta */
 59   this.config.minNextAction = 500;
 60 
 61   /** Koska viimeistään arvotaan botille uusi suunta */
 62   this.config.maxNextAction = 1000;
 63 
 64   /** Kun botille arvotaan uusi kääntyminen niin tämä on maksimi. */
 65   this.config.randRotation = 90;
 66 
 67   /** Botin nopeus kun se on havainnut esteen */
 68   this.config.minSpeed = 80;
 69 
 70   /** Botin nopeus kun tie on vapaa */
 71   this.config.maxSpeed = 200;
 72 
 73   /** Jos matkaa esteeseen on vähemmän kuin tämä niin aletaan etsiä uutta suuntaa */
 74   this.config.wakeupDist = 100;
 75 
 76   /**
 77    * Kun pitää päätellä uusi suunta niin se tehdään katselemalla näin monta astetta molempiin
 78    * suuntiin.
 79    */
 80   this.config.exploreAngle = 50;
 81 
 82   /**
 83    * Kun botti on lähellä estettä niin tällä määrätään kuinka jyrkällä käännöksellä yritetään
 84    * väistää. Pienempi arvo on jyrkempi käännös.
 85    */
 86   this.config.dodgeRotation = 0.2;
 87 }

 89 /**
 90  * Yksittäisen botin tekoälyn päivitys.
 91  */
 92 BotAI.prototype.update = function () {
 93   var self = this
 94     , map = this.server.gameState.map
 95     , wakeupDist = this.config.wakeupDist
 96     , superLargeDist = (map.width + map.height) * map.tileSize
 97     , minDist
 98     , picker
 99     , pickWall
100     , dist1, dist2, dodge
101     , objectiveAngle
102     , moveDirection = 1
103     , speed, forwardSpeed, sidestepSpeed
104     , weaponWeight = Weapons[this.player.weapon].weight;
105   // Mikäli botti ei ole liian lähellä seinää ja on aika arpoa sille uusi suunta
106   // niin tehdään se nyt.
107   if (!this.tooClose && this.nextAction < Date.now()) {
108     // Arvotaan seuraavan käännöksen ajankohta eli koska tähän iffiin tullaan seuraavaksi
109     this.nextAction = rand(this.config.minNextAction, this.config.maxNextAction) + Date.now();
110     // Arvotaan botille uusi kääntyminen
111     this.rotation = rand(-90, 90, true);
112 
113     if (this.sideStep !== 0) {
114       this.sideStep = rand(-1, 1, true);
115     }
116   }
117 
118   // Käännetään bottia
119   if (this.player.zombie) {
120     this.player.turn(this.server.game.movePerSec(this.rotation));
121   }
122 
123   // Seuraavaksi alkaa varsinainen tekoäly jossa tutkitaan ympäristöä.
124   // Tämä tehdään kuitenkin vain mikäli botti ei ole liian lähellä jotakin estettä.
125   if (!this.tooClose) {
126     // Lasketaan etäisyys edessä olevaan esteeseen.
127     // Etäisyys lasketaan objektin keskeltä sekä reunoista eli objektin koko leveydeltä.
128 
129     // Luodaan väliaikainen poimintaobjekti
130     picker = new Obj(this.player.x, this.player.y, this.player.angle);
131 
132     // Poimitaan botin nenän edestä mahdollinen seinä
133     pickWall = map.findWall(picker.x, picker.y, picker.angle, wakeupDist);
134     this.debugRayCast(pickWall, picker);
135     if (pickWall) {
136       minDist = distance(picker.x, picker.y, pickWall.x, pickWall.y);
137     }
138 
139     // Poimitaan botin vasemman reunan puolelta seinä
140     picker.move(0, -15);
141     pickWall = map.findWall(picker.x, picker.y, picker.angle, wakeupDist);
142     this.debugRayCast(pickWall, picker);
143     if (pickWall) {
144       minDist = Math.min(minDist, distance(picker.x, picker.y, pickWall.x, pickWall.y));
145     }
146 
147     // Poimitaan botin oikean reunan puolelta seinä
148     picker.move(0, 30);
149     pickWall = map.findWall(picker.x, picker.y, picker.angle, wakeupDist);
150     this.debugRayCast(pickWall, picker);
151     if (pickWall) {
152       minDist = Math.min(minDist, distance(picker.x, picker.y, pickWall.x, pickWall.y));
153     }
154 
155     // Jos tarpeeksi lähellä on seinä niin reagoidaan siihen nyt
156     if (minDist) {
157       // Käännetään poimintaobjekti samaan kulmaan kuin botti
158       picker.angle = this.player.angle;
159 
160       // Käännetään katsetta toiselle sivulle ja lasketaan etäisyys lähimpään esteeseen
161       picker.turn(-this.config.exploreAngle);
162       pickWall = map.findWall(picker.x, picker.y, picker.angle, superLargeDist);
163       this.debugRayCast(pickWall, picker);
164       dist1 = distance(picker.x, picker.y, pickWall.x, pickWall.y);
165 
166       // Ja sitten vielä toiseen suuntaan.
167       picker.turn(this.config.exploreAngle * 2);
168       pickWall = map.findWall(picker.x, picker.y, picker.angle, superLargeDist);
169       this.debugRayCast(pickWall, picker);
170       dist2 = distance(picker.x, picker.y, pickWall.x, pickWall.y);
171 
172       // Tutkitaan kumpaan suuntaan on pidempi matka seuraavaan esteeseen ja suunnataan sinne.
173       // Kääntymisen jyrkkyyteen vaikuttaa vielä etäisyys esteeseen eli mitä lähempänä ollaan niin
174       // sitä jyrkemmin käännytään
175       if (dist1 > dist2) {
176         dodge = -(wakeupDist - minDist);
177       } else {
178         dodge = wakeupDist - minDist;
179       }
180 
181       // Asetetaan kääntymisnopeus
182       this.rotation = dodge / this.config.dodgeRotation;
183       // Asetetaan tavoitekulma
184       this.nextAngle = wrapAngle(this.player.angle + dodge);
185       // Asetetaan vielä tooClose-muuttuja päälle eli tekoälyä ei päivitetä ennen kuin objekti on
186       // kääntynyt tavoitekulmaan. Samalla myös objektin nopeutta vähennetään.
187       this.tooClose = true;
188 
189       this.lastAngle = this.player.angle - this.nextAngle;
190       if (this.lastAngle > 180) { this.lastAngle -= 360; }
191       if (this.lastAngle < -180) { this.lastAngle += 360; }
192     }
193   } else {
194     // Botti on liian lähellä jotain estettä.
195     // tooClose-muuttuja nollataan vain jos tekoälyn asettama tavoitekulma on saavutettu.
196     objectiveAngle = this.player.angle - this.nextAngle;
197     if (objectiveAngle > 180) { objectiveAngle -= 360; }
198     if (objectiveAngle < -180) { objectiveAngle += 360; }
199     if ((objectiveAngle < 0 && this.lastAngle >= 0) || (objectiveAngle > 0 && this.lastAngle <= 0)) {
200       // Objektin kulma on nyt riittävän lähellä tavoitekulmaa joten kääntäminen voidaan lopettaa.
201       this.rotation = 0;
202       this.tooClose = false;
203     }
204     this.lastAngle = objectiveAngle;
205   }
206 
207   // Siirretään bottia
208   speed = this.config.maxSpeed;
209   // Jos ollaan liian lähellä jotain estettä niin pienemmällä vauhdilla
210   if (this.tooClose) { speed = this.config.minSpeed; }
211 
212   forwardSpeed = moveDirection * 100 / weaponWeight;
213   forwardSpeed *= this.server.game.movePerSec(speed);
214   sidestepSpeed = this.sideStep * 100.0 / weaponWeight;
215   sidestepSpeed *= this.server.game.movePerSec(PLR.SIDESTEP_SPEED * 0.8);
216   this.player.move(forwardSpeed, sidestepSpeed);
217 };

219 /**
220  * Debuggaa raycastia
221  */
222 BotAI.prototype.debugRayCast = function (collision, picker) {
223   if (!this.server.debug) {
224     return;
225   }
226   var self = this
227     , startX, startY, endX, endY, colX, colY;
228 
229   startX = Math.round(picker.x);
230   startY = Math.round(picker.y);
231   endX = Math.round(picker.x + Math.cos(picker.angle / 180 * Math.PI) * this.config.wakeupDist);
232   endY = Math.round(picker.y + Math.sin(picker.angle / 180 * Math.PI) * this.config.wakeupDist);
233   if (collision) {
234     colX = Math.round(collision.x);
235     colY = Math.round(collision.y);
236   }
237 
238   this.server.loopPlayers(function (player) {
239     if (player.debugState) {
240       player.debugState += 1;
241       self.server.messages.add(player.id, {
242         msgType: NET.DEBUGDRAWING,
243         drawType: DRAW.LINE,
244         drawVars: [
245           startX,
246           startY,
247           endX,
248           endY
249         ]
250       });
251       if (!collision) { return; }
252       self.server.messages.add(player.id, {
253         msgType: NET.DEBUGDRAWING,
254         drawType: DRAW.CIRCLE,
255         drawVars: [
256           colX,
257           colY,
258           5,
259           0
260         ]
261       });
262     }
263   });
264 }
