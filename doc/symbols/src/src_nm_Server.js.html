 29 /**
 30  * Luo uuden palvelimen annettuun porttiin ja osoitteeseen. Kun palvelimeen tulee dataa, emittoi
 31  * se siihen kuuluvan eventin paketin ensimmäisen tavun perusteella. esim. NET.LOGIN
 32  *
 33  * @class Yleiset Serverin funktiot ja ominaisuudet
 34  *
 35  * @param {Number} port            Portti, jota palvelin kuuntelee.
 36  * @param {String} [address]       IP-osoite, jota palvelin kuuntelee. Jos tätä ei anneta, järjestelmä
 37  *                                 yrittää kuunnella kaikkia osoitteita (0.0.0.0).
 38  * @param {Boolean} [debug=false]  Spämmitäänkö konsoliin paljon "turhaa" tietoa?
 39  */
 40 function Server(port, address, debug) {
 41   if (this.debug = debug) { log.notice('Server running on debug mode, expect spam!'.red); }
 42   /**
 43    * cbNetwork-node UDP-palvelin
 44    * @type cbNetwork.Server
 45    * @see <a href="http://vesq.github.com/cbNetwork-node/doc/symbols/Server.html">cbNetwork.Server</a>
 46    */
 47   this.server = new cbNetwork.Server(port, address);
 48 
 49   /** Sisältää pelin nykyisestä tilanteesta kertovat muuttujat. */
 50   this.gameState = {};
 51 
 52   /** Sisältää kaikki kartat */
 53   this.maps = {};
 54 
 55   /** Sisältää palvelimen pelaajat, eli luokan {@link Player} jäsenet. */
 56   this.players = {};
 57 
 58   /** Sisältää palvelimen ammukset, eli luokan {@link Weapon} jäsenet. */
 59   this.bullets = {};
 60   /** @private */
 61   this.lastBulletId = 0;
 62 
 63   /** Sisältää palvelimella maassa olevat tavarat, kts. {@link Item}. */
 64   this.items = {};
 65 
 66   // Alustetaan moduulit
 67 
 68   /**
 69    * Pelaajille lähetettävät viestit. Tämä on instanssi {@link NetMessages}-luokasta.
 70    * @type NetMessages
 71    */
 72   this.messages = new NetMsgs(this);
 73 
 74   /**
 75    * Asetukset tälle palvelimelle
 76    * @type Config
 77    */
 78   this.config = new Config(this);
 79 
 80   /**
 81    * Sisältää palvelimella pyörivän {@link Game}-luokan instanssin
 82    * @type Game
 83    */
 84   this.game = new Game(this);
 85 
 86   /**
 87    * Sisältää palvelimen komennot, joita voidaan kutsua joko palvelimen konsolista tai klientiltä
 88    * @type Commands
 89    */
 90   this.commands = new Command(this);
 91 
 92   /**
 93    * Sisältää palvelimen konsoli-io:n käsittelyyn käytettävän {@link Input}-luokan instanssin
 94    * @type Input
 95    */
 96   this.input = new Input(this);
 97 
 98   // Alustetaan palvelin (esim. kartta, pelaajat, tavarat)
 99   this.initialize();
100 
101   log.info('Server initialized successfully.');
102 }

106 /** Alustaa palvelimen */
107 Server.prototype.initialize = function () {
108   // Kuunnellaan klinuja
109   var self = this;
110   this.server.on('message', function recvMsg(client) {
111     self.handlePacket(client);
112   });
113 
114   this.server.on('close', function onClose() {
115     if (!self.gameState.closing) {
116       // Hups! cbNetwork serveri kaatui alta.
117       log.fatal('cbNetwork unexpectedly closed, server will terminate.');
118       self.close(true);
119     }
120     // Muussa tapauksessa sulkeutuminen oli odotettavissa
121   });
122 
123   // Alustetaan pelitilanne
124   this.gameState.playerCount = 0;
125   this.gameState.gameMode = this.config.gameMode;
126   this.gameState.maxPlayers = this.config.maxPlayers;
127 
128   // Ladataan kartta
129   this.gameState.map = new Map(this, this.config.map);
130   if (!this.gameState.map.loaded) {
131     log.fatal('Could not load map "%0"', this.config.map);
132     this.close();
133     return;
134   }
135 
136   // UNIMPLEMENTED Mappien rotaatio
137   this.maps[this.gameState.map.name] = this.gameState.map;
138 
139   // Alustetaan pelaajat
140   for (var i = 1; i <= this.config.maxPlayers; ++i) {
141     new Player(this, i);
142   }
143 
144   // Alustetaan itemit
145   var mapConfig = this.gameState.map.config;
146   var itemId = 0;
147   for (var i = mapConfig.healthItems - 1; i--;) {
148     new Item(this, ++itemId, ITM.HEALTH);
149   }
150   for (var i = mapConfig.mgunItems - 1; i--;) {
151     new Item(this, ++itemId, ITM.AMMO);
152   }
153   for (var i = mapConfig.bazookaItems - 1; i--;) {
154     new Item(this, ++itemId, ITM.ROCKET);
155   }
156   for (var i = mapConfig.shotgunItems - 1; i--;) {
157     new Item(this, ++itemId, ITM.SHOTGUN);
158   }
159   for (var i = mapConfig.launcherItems - 1; i--;) {
160     new Item(this, ++itemId, ITM.LAUNCHER);
161   }
162   for (var i = mapConfig.chainsawItems - 1; i--;) {
163     new Item(this, ++itemId, ITM.FUEL);
164   }
165 };

167 /**
168  * Hoitaa saapuneiden viestien käsittelyn.
169  *
170  * @param {cbNetwork.Client}  cbNetwork-noden Client-luokan instanssi, jolta dataa tulee.
171  * @see <a href="http://vesq.github.com/cbNetwork-node/doc/symbols/Client.html">cbNetwork.Client</a>
172  */
173 Server.prototype.handlePacket = function (client) {
174   var data = client.data
175     , msgType = data.getByte()
176     , currentPlayerId;
177 
178   // Onko servu sammumassa?
179   if (this.gameState.closing) {
180     reply = new Packet(2);
181     reply.putByte(NET.SERVERCLOSING);
182     reply.putByte(NET.END);
183     client.reply(reply);
184     return;
185   }
186 
187   if (msgType === NET.LOGIN) {
188     // Login paketissa ei ole pelaajan ID:tä vielä, joten se on käsiteltävä erikseen
189     this.emit(NET.LOGIN, client);
190     return;
191   }
192 
193   // Luetaan lähetetty pelaajan ID, joka on pelaajan järjestysnumero ja aina väliltä 1...MAX_PLAYERS
194   currentPlayerId = data.getByte();
195   // Tai jos ei ole niin sitten ei päästetä sisään >:(
196   if (currentPlayerId < 1 || currentPlayerId > this.gameState.maxPlayers) {
197     log.notice('Possible hack attempt from ' + client.address + ' Invalid player ID (' + currentPlayerId + ')');
198     return;
199   }
200 
201   // Haetaan pelaajan instanssi Player-luokasta
202   player = this.players[currentPlayerId];
203 
204   // Tarkistetaan onko pelaaja potkittu
205   if (player.kicked && player.clientId === client.id) {
206     reply = new Packet(7);
207     reply.putByte(NET.KICKED);
208     reply.putByte(player.kickerId);
209     reply.putByte(currentPlayerId);
210     reply.putString(player.kickReason);
211     client.reply(reply);
212     return;
213   }
214   // Vielä yksi tarkistus
215   if (player.clientId !== client.id || !player.active) {
216     reply = new Packet(1);
217     reply.putByte(NET.NOLOGIN);
218     client.reply(reply);
219     return;
220   }
221 
222   // Logout on erikseen, koska sen jälkeen ei varmasti tule mitään muuta
223   if (msgType === NET.LOGOUT) {
224     this.emit(NET.LOGOUT, client, currentPlayerId);
225     return;
226   }
227 
228   // Lasketaan pelaajan ja serverin välinen lagi
229   player.lag = timer() - player.lastActivity;
230   // Päivitetään pelaajan olemassaolo
231   player.lastActivity = timer();
232 
233   // Luupataan kaikkien pakettien läpi
234   while (msgType) {
235     // Lähetetään tietoa paketista käsiteltäväksi
236     this.emit(msgType, client, player);
237     msgType = data.getByte();
238   }
239 
240   // Lähetetään dataa pelaajalle
241   this.sendReply(client, player);
242 
243   // Valmis! :)
244 };

246 /**
247  * Hoitaa datan lähetyksen.
248  *
249  * @param {cbNetwork.Client} client  cbNetworkin Client-luokan instanssi
250  * @param {Player} player            Pelaaja, keneltä on saatu dataa ja kenelle lähetetään vastaus tässä.
251  * @see <a href="http://vesq.github.com/cbNetwork-node/doc/symbols/Client.html">cbNetwork.Client</a>
252  */
253 Server.prototype.sendReply = function (client, player) {
254   var reply = new Packet()
255     , playerIds = Object.keys(this.players)
256     , plr;
257 
258   // Lähetetään kaikkien pelaajien tiedot
259   for (var i = playerIds.length; i--;) {
260     plr = this.players[playerIds[i]];
261     // Onko pyydetty nimet
262     if (player.sendNames) {
263       if (plr.active) {
264         reply.putByte(NET.PLAYERNAME);  // Nimet
265         reply.putByte(plr.playerId);    // Pelaajan tunnus
266         reply.putString(plr.name);      // Nimi
267         reply.putByte(plr.zombie);      // Onko botti
268         reply.putByte(plr.team);        // Joukkue
269       }
270     }
271 
272     // Lähetetään niiden pelaajien tiedot jotka ovat hengissä ja näkyvissä
273     if (plr.active) {
274       var x1 = player.x
275         , y1 = player.y
276         , x2 = plr.x
277         , y2 = plr.y
278         , visible = !((Math.abs(x1 - x2) > 450) || (Math.abs(y1 - y2) > 350));
279 
280       // Onko näkyvissä vai voidaanko muuten lähettää
281       if (player.sendNames || visible || plr.health <= 0) {
282         // Näkyy
283         reply.putByte(NET.PLAYER);    // Pelaajan tietoja
284         reply.putByte(plr.playerId);  // Pelaajan tunnus
285         reply.putShort(plr.x);        // Sijainti
286         reply.putShort(plr.y);        // Sijainti
287         reply.putShort(plr.angle);    // Kulma
288 
289         // Spawn-protect
290         var isProtected = 0;
291         if (plr.spawnTime + this.gameState.spawnProtection > timer()) {
292           isProtected = 1;
293         }
294 
295         // Muutetaan team arvo välille 0-1
296         var teamBit = (plr.team === 2 ? 1 : 0);
297 
298         // Tungetaan yhteen tavuun useampi muuttuja
299         var b = ((plr.weapon % 16) << 0)  // Ase (bitit 0-3)
300               + ((plr.hasAmmos << 4))     // Onko ammuksia (bitti 4)
301               + ((teamBit << 6))          // Joukkue/tiimi (bitti 6)
302               + ((isProtected << 7));     // Haavoittumaton (bitti 7)
303         reply.putByte(b);
304 
305         if (plr.health <= 0) {
306           // Client-puolella health-arvo lasketaan seuraavasta tavusta niin, että jos se ylittää
307           // 128, on oikea health-määrä vastaanotettu tavu - 256.
308           reply.putByte(Math.min(255, Math.max(0, plr.health + 256)));
309         } else {
310           reply.putByte(plr.health);      // Terveys
311         }
312         reply.putShort(plr.kills);      // Tapot
313         reply.putShort(plr.deaths);     // Kuolemat
314       } else if (this.gameState.radarArrows || this.gameState.playMode === 2) {
315         // Ei näy. Lähetetään tutkatieto. playMode === 2 tarkoittaa TDM-pelimuotoa
316         if (player.team === plr.team || this.gameState.radarArrows) {
317           // Lähetetään tutkatiedot jos joukkueet ovat samat tai asetuksista on laitettu että
318           // kaikkien joukkueiden pelaajien tutkatiedot lähetetään
319           reply.putByte(NET.RADAR); // Tutkatietoa tulossa
320           // UNIMPLEMENTED
321           // Missä kulmassa tutkan pitäisi olla
322           reply.putByte(0);         // Kulma muutettuna välille 0-255
323           reply.putByte(plr.team);  // Pelaajan joukkue
324         }
325       }
326     }
327   }
328 
329   // UNIMPLEMENTED
330   // Kartan vaihtaminen
331 
332   // Lähetetään kaikki pelaajalle osoitetut viestit
333   this.messages.fetch(player.playerId, reply);
334 
335 
336   // Jos on pyydetty nimilista niin palautetaan myös kaikkien tavaroiden tiedot
337   if (player.sendNames) {
338     player.sendNames = false;
339     var itemIds = Object.keys(this.items);
340     for (var i = itemIds.length; i--;) {
341       var item = this.items[itemIds[i]];
342       this.messages.add(player.playerId, {
343         msgType: NET.ITEM,
344         itemId: item.id,
345         itemType: item.type,
346         x: item.x,
347         y: item.y
348       });
349     }
350   }
351 
352   // UNIMPLEMENTED
353   // Pelisession aikatiedot
354 
355   reply.putByte(NET.END);
356   client.reply(reply);
357 
358   // Dodiin, valmiita ollaan :)
359   return;
360 };

362 Server.prototype.serverMessage = function (msg, to) {
363 
364   if (!to) {
365     log.write('<Server>'.blue + ' %0', msg);
366     this.messages.addToAll({
367       msgType: NET.SERVERMSG,
368       msgText: msg
369     });
370   } else {
371     log.write('<Server @%0>'.blue + ' %1', this.players[to].name, msg);
372     this.messages.add(to, {
373       msgType: NET.SERVERMSG,
374       msgText: msg
375     });
376   }
377 }

379 /**
380  * Kirjaa pelaajan sisään peliin.
381  * @param {cbNetwork.Client} client  cbNetworkin Client-luokan jäsen.
382  * @returns {Boolean}                Onnistuiko pelaajan liittäminen peliin vai ei.
383  * @see <a href="http://vesq.github.com/cbNetwork-node/doc/symbols/Client.html">cbNetwork.Client</a>
384  */
385 Server.prototype.login = function (client) {
386   var data = client.data
387     , version = data.getString()
388     , replyData
389     , nickname
390     , playerIds
391     , randomPlace;
392 
393   // Täsmääkö clientin ja serverin versiot
394   if (version !== Server.VERSION) {
395     log.notice('Player trying to connect with incorrect client version.');
396     replyData = new Packet(3);
397     replyData.putByte(NET.LOGIN);
398     replyData.putByte(NET.LOGINFAILED);
399     replyData.putByte(NET.WRONGVERSION);
400     replyData.putString(Server.VERSION);
401     client.reply(replyData);
402     return;
403   }
404 
405   // Versio on OK, luetaan pelaajan nimi
406   nickname = data.getString().trim();
407   log.info('Player %0 is trying to login...', nickname.green);
408 
409   // Käydään kaikki nimet läpi ettei samaa nimeä vain ole jo suinkin olemassa
410   playerIds = Object.keys(this.players);
411   for (var i = playerIds.length; i--;) {
412     var player = this.players[playerIds[i]];
413     if (player.name.toLowerCase() === nickname.toLowerCase()) {
414       if (player.kicked || !player.active) {
415         player.name = "";
416       } else {
417         // Nimimerkki oli jo käytössä.
418         log.info(' -> Nickname %0 already in use.', nickname.green);
419         replyData = new Packet(3);
420         replyData.putByte(NET.LOGIN);
421         replyData.putByte(NET.LOGINFAILED);
422         replyData.putByte(NET.NICKNAMEINUSE);
423         client.reply(replyData);
424         return;
425       }
426     }
427   }
428 
429   // Etsitään inaktiivinen pelaaja
430   for (var i = playerIds.length; i--;) {
431     var player = this.players[playerIds[i]];
432     if (this.gameState.playerCount < this.config.maxPlayers && !player.active) {
433       // Tyhjä paikka löytyi
434       player.clientId = client.id;
435       player.active = true;
436       player.loggedIn = false;
437       player.name = nickname;
438       randomPlace = this.gameState.map.findSpot();
439       player.x = randomPlace.x;
440       player.y = randomPlace.y;
441       player.hackTestX = player.x;
442       player.hackTestY = player.y;
443       player.angle = Math.floor(Math.random() * 360 + 1);
444       player.zombie = false;
445       player.health = 100;
446       player.kills = 0;
447       player.deaths = 0;
448       player.weapon = WPN.PISTOL;
449       player.lastActivity = new Date().getTime();
450       player.spawnTime = player.lastActivity;
451       player.admin = false;
452       player.kicked = false;
453       player.kickReason = "";
454       if (this.gameState.gameMode === 2) {
455         // UNIMPLEMENTED
456         // Tasainen jako joukkueihin TDM-pelimoodissa
457         player.team = Math.floor(Math.random() * 2 + 1) + 1; // Rand(1,2)
458       }
459 
460       // Lähetetään vastaus clientille
461       replyData = new Packet(16);
462       replyData.putByte(NET.LOGIN);
463       replyData.putByte(NET.LOGINOK);
464       replyData.putByte(player.playerId);
465       replyData.putByte(this.gameState.gameMode);
466       replyData.putString(this.gameState.map.name);
467       replyData.putInt(this.gameState.map.crc32);
468       // UNIMPLEMENTED
469       replyData.putString(" "); // Kartan URL josta sen voi ladata, mikäli se puuttuu
470       client.reply(replyData);
471       log.info(' -> login successful, assigned ID (%0)', String(player.playerId).magenta);
472 
473       // Lisätään viestijonoon ilmoitus uudesta pelaajasta, kaikille muille paitsi boteille ja itselle.
474       this.messages.addToAll({
475         msgType: NET.LOGIN,
476         msgText: nickname,
477         playerId: player.playerId,
478         playerId2: player.zombie
479       }, player.playerId);
480       return;
481     }
482   }
483 
484   // Vapaita paikkoja ei ollut
485   log.info(' -> Server is full!');
486   replyData = new Packet(3);
487   replyData.putByte(NET.LOGIN);
488   replyData.putByte(NET.LOGINFAILED);
489   replyData.putByte(NET.TOOMANYPLAYERS);
490   client.reply(replyData);
491   return;
492 };

551 /**
552  * Sammuttaa palvelimen. Emittoi eventit {@link Server#closing} ja {@link Server#closed}
553  */
554 Server.prototype.close = function (now) {
555   if (this.gameState.closing) {
556     // Ollaan jo sulkemassa, ei aloiteta samaa prosessia uudelleen.
557     return;
558   }
559   this.gameState.closing = true;
560   log.info('Server going down...');
561   this.emit('closing');
562 
563   // Pysäytetään Game-moduulin päivitys
564   this.game.stop();
565 
566   var self = this;
567   setTimeout(function closeServer() {
568     self.server.close();
569     self.emit('closed');
570     process.exit();
571   }, now ? 0 : 1000);
572 };
