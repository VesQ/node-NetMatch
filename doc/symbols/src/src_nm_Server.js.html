 23 /**
 24  * Luo uuden palvelimen annettuun porttiin ja osoitteeseen. Kun palvelimeen tulee dataa, emittoi
 25  * se <i>message</i> eventin, kts. {@link Server#event:message}
 26  *
 27  * @class Yleiset Serverin funktiot ja ominaisuudet
 28  *
 29  * @param {Object} c            Asetukset, kts. {@link Server.config}
 30  * @param {Number} c.port       Portti, jota palvelin kuuntelee.
 31  * @param {String} [c.address]  IP-osoite, jota palvelin kuuntelee. Jos tätä ei anneta, järjestelmä
 32  *                              yrittää kuunnella kaikkia osoitteita.
 33  */
 34 function Server(c) {
 35   if('object' !== typeof c || !c.hasOwnProperty('port')) {
 36     log.fatal("Initialization of NetMatch server failed because of incorrect starting parameters.");
 37     return false;
 38   }
 39   /**
 40    * cbNetwork-node UDP-palvelin
 41    * @type cbNetwork.Server
 42    * @see <a href="http://vesq.github.com/cbNetwork-node/doc/symbols/Server.html">cbNetwork.Server</a>
 43    */
 44   this.server = new cbNetwork.Server(c.port, c.address);
 45 
 46   var self = this;
 47   this.server.on('message', function emitMessage(client) {
 48     self.emit('message', client);
 49   });
 50 
 51   /**
 52    * Pelaajille lähetettävät viestit. Tämä on instanssi {@link NetMessages}-luokasta.
 53    * @see NetMessages
 54    */
 55   this.messages = new NetMessages(this);
 56 
 57   /**
 58    * Asetukset tälle palvelimelle
 59    * @type Object
 60    * @see Server.config
 61    */
 62   this.config = Server.config;
 63 
 64   /** Sisältää pelin nykyisestä tilanteesta kertovat muuttujat. */
 65   this.gameState = {};
 66   this.gameState.playerCount = 0;
 67   this.gameState.gameMode = this.config.gameMode;
 68   this.gameState.maxPlayers = this.config.maxPlayers;
 69 
 70   // Ladataan kartta
 71   this.gameState.map = new Map(this.config.map);
 72   if (!this.gameState.map.loaded) {
 73     log.fatal('Could not load map "%0"', this.config.map);
 74     this.close();
 75     return;
 76   }
 77 
 78   /** Sisältää kaikki kartat */
 79   this.maps = {};
 80   this.maps[this.gameState.map.name] = this.gameState.map;
 81 
 82   /**
 83    * Sisältää palvelimen pelaajat, eli luokan {@link Player} jäsenet.
 84    * Pääset yksittäisen pelaajan dataan näin:
 85    * @example
 86    * // Oletetaan että muuttujaan server on luotu NetMatch-palvelin.
 87    *
 88    * // Tulostaa konsoliin ID:n 4 pelaajan nimen.
 89    * console.log( server.players[4].name );
 90    */
 91   this.players = {};
 92 
 93   // Alustetaan pelaajat
 94   for (var i = 1; i <= this.config.maxPlayers; ++i) {
 95     var pl = new Player();
 96     pl.playerId = i;
 97     pl.active = false;
 98     pl.team = 1;
 99     pl.botName = "Bot_" + i;
100     pl.clientId = "";
101     pl.active = false;
102     pl.loggedIn = false;
103     pl.name = "";
104     var skill = 21 - pl.playerId; // Botteja hieman eritasoisiksi vissiinkin?
105     pl.fightRotate = 1.5 + (skill / 1.5);
106     pl.shootingAngle = 4.0 + (pl.playerId * 1.5);
107     pl.fov = 100 + (skill * 3.5);
108     pl.hasAmmos = false;
109     pl.admin = false;
110     pl.kicked = false;
111     pl.kickReason = "";
112 
113     this.players[i] = pl;
114   }
115 
116   /**
117    * Sisältää palvelimen ammukset, eli luokan {@link Weapon} jäsenet.
118    */
119   this.bullets = [];
120 
121   /**
122    * @private
123    */
124   this.lastBulletId = 0;
125 
126   /**
127    * Sisältää palvelimella maassa olevat tavarat, kts. {@link Item}.
128    */
129   this.items = {};
130 
131   // Alustetaan itemit
132   var mapConfig = this.gameState.map.config;
133   var itemId = 0;
134   for (var i = mapConfig.healthItems - 1; i--;) {
135     new Item(this, ++itemId, ITM.HEALTH);
136   }
137   for (var i = mapConfig.mgunItems - 1; i--;) {
138     new Item(this, ++itemId, ITM.AMMO);
139   }
140   for (var i = mapConfig.bazookaItems - 1; i--;) {
141     new Item(this, ++itemId, ITM.ROCKET);
142   }
143   for (var i = mapConfig.shotgunItems - 1; i--;) {
144     new Item(this, ++itemId, ITM.FUEL);
145   }
146   for (var i = mapConfig.launcherItems - 1; i--;) {
147     new Item(this, ++itemId, ITM.SHOTGUN);
148   }
149   for (var i = mapConfig.chainsawItems - 1; i--;) {
150     new Item(this, ++itemId, ITM.LAUNCHER);
151   }
152 
153   /**
154    * Sisältää palvelimella pyörivän {@link Game}-luokan instanssin
155    */
156   this.game = new Game(this);
157 
158   // Avataan Input
159   new Input(this);
160 
161   log.info('Server initialized successfully.');
162 }

271 /**
272  * Kirjaa pelaajan sisään peliin.
273  * @param {Client} client  cbNetworkin Client-luokan jäsen.
274  * @returns {Boolean}      Onnistuiko pelaajan liittäminen peliin vai ei.
275  */
276 Server.prototype.login = function (client) {
277   var data = client.data
278     , replyData
279     , version = data.getString()
280     , nickname
281     , playerIds;
282 
283   // Täsmääkö clientin ja serverin versiot
284   if (version !== Server.VERSION) {
285     // Eivät täsmää, lähetetään virheilmoitus
286     replyData = new Packet(3);
287     replyData.putByte(NET.LOGIN);
288     replyData.putByte(NET.LOGINFAILED);
289     replyData.putByte(NET.WRONGVERSION);
290     replyData.putString(Server.VERSION);
291     client.reply(replyData);
292     return false;
293   }
294 
295   // Versio on OK, luetaan pelaajan nimi
296   nickname = data.getString().trim();
297   log.info('Player "' + nickname + '" (' + client.address + ') is trying to connect...');
298 
299   // Käydään kaikki nimet läpi ettei samaa nimeä vain ole jo suinkin olemassa
300   playerIds = Object.keys(this.players);
301   for (var i = playerIds.length; i--;) {
302     var player = this.players[playerIds[i]];
303     if (player.name.toLowerCase() === nickname.toLowerCase()) {
304       if (player.kicked || !player.active) {
305         player.name = "";
306       } else {
307         // Nimimerkki oli jo käytössä.
308         log.notice('Nickname "' + nickname + '" already in use.');
309         replyData = new Packet(3);
310         replyData.putByte(NET.LOGIN);
311         replyData.putByte(NET.LOGINFAILED);
312         replyData.putByte(NET.NICKNAMEINUSE);
313         client.reply(replyData);
314         return false;
315       }
316     }
317   }
318 
319   // Etsitään vapaa "pelipaikka"
320   for (var i = playerIds.length; i--;) {
321     var player = this.players[playerIds[i]];
322     if (this.gameState.playerCount < this.config.maxPlayers && !player.active) {
323       // Tyhjä paikka löytyi
324       player.clientId = client.id;
325       player.active = true;
326       player.loggedIn = false;
327       player.name = nickname;
328       // UNIMPLEMENTED
329       player.x = Math.round(-100 + Math.random() * 200);
330       player.y = Math.round(-100 + Math.random() * 200);
331       player.hackTestX = player.x;
332       player.hackTestY = player.y;
333       player.angle = Math.floor(Math.random() * 360 + 1);
334       player.zombie = false;
335       player.health = 100;
336       player.kills = 0;
337       player.deaths = 0;
338       player.weapon = WPN.PISTOL;
339       player.lastActivity = new Date().getTime();
340       player.spawnTime = player.lastActivity;
341       player.admin = false;
342       player.kicked = false;
343       player.kickReason = "";
344       if (this.gameState.gameMode === "TDM") {
345         // UNIMPLEMENTED
346         // Tasainen jako joukkueihin
347         player.team = Math.floor(Math.random()*2 + 1) + 1; // Rand(1,2)
348       }
349 
350       // Lähetetään vastaus clientille
351       replyData = new Packet(16);
352       replyData.putByte(NET.LOGIN);
353       replyData.putByte(NET.LOGINOK);
354       replyData.putByte(player.playerId);
355       replyData.putByte(this.gameState.gameMode);
356       replyData.putString(this.gameState.map.name);
357       replyData.putInt(this.gameState.map.crc32);
358       // UNIMPLEMENTED
359       replyData.putString(" "); // Kartan URL josta sen voi ladata, mikäli se puuttuu
360       client.reply(replyData);
361       log.info(player.name + " logged in, assigned ID " + String(player.playerId).magenta);
362 
363       // Lisätään viestijonoon ilmoitus uudesta pelaajasta
364       var msgData = {
365         msgType: NET.LOGIN,
366         msgText: nickname,
367         playerId: player.playerId,
368         playerId2: player.zombie
369       };
370       for (var j = playerIds.length; j--;) {
371         var plr = this.players[playerIds[j]];
372         // Lähetetään viesti kaikille muille paitsi boteille ja itselle
373         if (plr.active && !plr.zombie && (plr.playerId !== player.playerId)) {
374           this.messages.add(plr.playerId, msgData);
375         }
376       }
377       return true;
378     }
379   }
380 
381   // Vapaita paikkoja ei ollut
382   log.notice('No free slots.');
383   replyData = new Packet(3);
384   replyData.putByte(NET.LOGIN);
385   replyData.putByte(NET.LOGINFAILED);
386   replyData.putByte(NET.TOOMANYPLAYERS);
387   client.reply(replyData);
388   return false;
389 }

391 /**
392  * Kirjaa pelaajan ulos pelistä.
393  * @param {Integer} playerId  Pelaajan ID.
394  */
395 Server.prototype.logout = function (playerId) {
396   var player = this.players[playerId]
397     , playerIds = Object.keys(this.players);
398 
399   player.active = false;
400   player.loggedIn = false;
401   player.admin = false;
402   log.info(player.name + ' logged out.');
403 
404   // Lähetetään viesti kaikille muille paitsi boteille
405   for (var i = playerIds.length; i--;) {
406     var plr = this.players[playerIds[i]];
407     // Lähetetään viesti kaikille muille paitsi boteille ja itselle
408     if (plr.active && !plr.zombie && (plr.playerId !== playerId)) {
409       this.messages.add(plr.playerId, { msgType: NET.LOGOUT, playerId: playerId });
410     }
411   }
412 }

414 /**
415  * Sammuttaa palvelimen. Emittoi eventit {@link Server#closing} ja {@link Server#closed}
416  */
417 Server.prototype.close = function () {
418   if (this.gameState.closing) {
419     // Ollaan jo sulkemassa, ei aloiteta samaa prosessia uudelleen.
420     return;
421   }
422   this.emit('closing');
423   log.info('Closing server...');
424   this.gameState.closing = true;
425 
426   // Pysäytetään Game-moduulin päivitys
427   clearInterval(this.game.interval);
428 
429   var self = this;
430   setTimeout(function closeServer() {
431     self.server.close();
432     log.info('Server closed!');
433     self.emit('closed');
434   }, 1000);
435 }
