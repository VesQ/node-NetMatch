 35 /**
 36  * Luo uuden palvelimen annettuun porttiin ja osoitteeseen. Kun palvelimeen tulee dataa, emittoi
 37  * se siihen kuuluvan eventin paketin ensimmäisen tavun perusteella. esim. NET.LOGIN
 38  *
 39  * @class Yleiset Serverin funktiot ja ominaisuudet
 40  *
 41  * @param {Number} port            Portti, jota palvelin kuuntelee.
 42  * @param {String} [address]       IP-osoite, jota palvelin kuuntelee. Jos tätä ei anneta, järjestelmä
 43  *                                 yrittää kuunnella kaikkia osoitteita (0.0.0.0).
 44  * @param {Boolean} [debug=false]  Spämmitäänkö konsoliin paljon "turhaa" tietoa?
 45  */
 46 function Server(args, version) {
 47   /** Palvelimen debug-tila */
 48   this.debug = args.d;
 49 
 50   if (this.debug) { log.notice('Server running on debug mode, expect spam!'.red); }
 51 
 52   /**
 53    * Palvelimen versio
 54    * @const
 55   */
 56   this.VERSION = version;
 57 
 58   /** Sisältää pelin nykyisestä tilanteesta kertovat muuttujat. */
 59   this.gameState = {};
 60 
 61   /** Sisältää kaikki kartat */
 62   this.maps = {};
 63 
 64   /** Sisältää palvelimen pelaajat, eli luokan {@link Player} jäsenet. */
 65   this.players = {};
 66 
 67   /** Sisältää palvelimen ammukset, eli luokan {@link Bullet} jäsenet. */
 68   this.bullets = {};
 69   /** @private */
 70   this.lastBulletId = 0;
 71 
 72   /** Sisältää palvelimella maassa olevat tavarat, kts. {@link Item}. */
 73   this.items = {};
 74 
 75   // Alustetaan moduulit
 76 
 77   /**
 78    * Pelaajille lähetettävät viestit. Tämä on instanssi {@link NetMessages}-luokasta.
 79    * @type NetMessages
 80    */
 81   this.messages = new NetMsgs(this);
 82 
 83   /**
 84    * Asetukset tälle palvelimelle
 85    * @type Config
 86    */
 87   this.config = new Config(this);
 88 
 89   /**
 90    * Sisältää palvelimella pyörivän {@link Game}-luokan instanssin
 91    * @type Game
 92    */
 93   this.game = new Game(this);
 94 
 95   /**
 96    * Sisältää palvelimen komennot, joita voidaan kutsua joko palvelimen konsolista tai klientiltä
 97    * @type Commands
 98    */
 99   this.commands = new Command(this);
100 
101   /**
102    * Sisältää palvelimen konsoli-io:n käsittelyyn käytettävän {@link Input}-luokan instanssin
103    * @type Input
104    */
105   this.input = new Input(this);
106 
107   /**
108    * Tämän palvelimen palvelinlistaukseen liittyvä toiminallisuus
109    * @type Registration
110    */
111   this.registration = new Registration(this);
112 
113   // Alustetaan palvelin (esim. kartta, pelaajat, tavarat)
114   if (this.initialize(args.p, args.a, args.c)) {
115     log.info('Server initialized successfully.');
116   } else {
117     log.fatal('Server initialization failed!');
118     if ('undefined' === typeof this.server) {
119       // cbNetworkkia ei keretty alustaa
120       this.close(true);
121     } else {
122       this.close();
123     }
124   }
125 }

129 /** Alustaa palvelimen */
130 Server.prototype.initialize = function (port, address, config) {
131   var self = this;
132 
133   // Ladataan konffit
134   if (!this.config.load(config)) { return false; }
135   // Komentoriviparametrit voittaa.
136   if (port) { this.config.port = port; }
137   if (address) { this.config.address = address; }
138 
139   // Portti on pakollinen.
140   if (!this.config.port || 'number' !== typeof this.config.port) {
141     log.fatal('You need to specify a port for NetMatch! Try `%0`', 'netmatch --help'.yellow);
142     return false;
143   }
144 
145   /**
146    * cbNetwork-node UDP-palvelin
147    * @type cbNetwork.Server
148    * @see <a href="http://vesq.github.com/cbNetwork-node/doc/symbols/Server.html">cbNetwork.Server</a>
149    */
150   this.server = new cbNetwork.Server(this.config.port, this.config.address);
151 
152   this.server.on('message', function recvMsg(client) {
153     self.handlePacket(client);
154   });
155 
156   this.server.on('close', function onClose() {
157     if (!self.gameState.closing) {
158       // Hups! cbNetwork serveri kaatui alta.
159       log.fatal('cbNetwork unexpectedly closed, server will terminate.');
160       self.close(true);
161     }
162     // Muussa tapauksessa sulkeutuminen oli odotettavissa
163   });
164 
165   // Alustetaan pelitilanne
166   this.gameState.playerCount = 0;
167   this.gameState.botCount = this.config.botCount;
168   this.gameState.botDepartLimit = this.config.botDepartLimit;
169   this.gameState.gameMode = this.config.gameMode;
170   this.gameState.maxPlayers = this.config.maxPlayers;
171   this.gameState.radarArrows = this.config.radarArrows;
172   this.gameState.sessionComplete = false;
173   this.gameState.mapNumber = 0; // Missä config.map listan kartassa mennään
174 
175   // Ladataan kartta
176   this.gameState.map = new Map(this, this.config.map[0]);
177   if (!this.gameState.map.loaded) {
178     // Kartan lataus epäonnistui
179     log.fatal('Could not load map "%0"', this.config.map[0]);
180     return false;
181   }
182 
183   // Alustetaan tavarat paikoilleen
184   this.gameState.map.initItems();
185 
186   // Jos kartalla on botCount-asetus, asetetaan se botCountiksi, mikäli nykyinen on < 0
187   if ('number' === typeof this.gameState.map.config.botCount && this.gameState.botCount < 0) {
188     this.gameState.botCount = this.gameState.map.config.botCount;
189   }
190 
191   // Jos kartalla on botDepartLimit-asetus, asetetaan se nykyiseksi, mikäli nykyinen on < 0
192   if ('number' === typeof this.gameState.map.config.botDepartLimit && this.gameState.botDepartLimit < 0) {
193     this.gameState.botDepartLimit = this.gameState.map.config.botDepartLimit;
194   }
195 
196   this.maps[this.gameState.map.name] = this.gameState.map;
197 
198   // Alustetaan pelaajat
199   for (var i = 1; i <= 64; ++i) {
200     this.players[i] = new Player(this, i);
201   }
202 
203   // Alustetaan botit
204   this.initBots();
205 
206   // Lisätäänkö palvelin palvelinlistaukseen
207   if (this.config.register) {
208     this.registration.apply(function initRegister(e) {
209       if (e) { log.error(e); }
210       else   { log.info('Server registered successfully.'); }
211     });
212   }
213 
214   // Käynnistetään pelimekaniikka, joka päivittyy configeissa määriteltyyn tahtiin.
215   this.game.start(this.config.updatesPerSec);
216 
217   return true;
218 };

220 /**
221  * Hoitaa saapuneiden viestien käsittelyn.
222  *
223  * @param {cbNetwork.Client}  cbNetwork-noden Client-luokan instanssi, jolta dataa tulee.
224  * @see <a href="http://vesq.github.com/cbNetwork-node/doc/symbols/Client.html">cbNetwork.Client</a>
225  */
226 Server.prototype.handlePacket = function (client) {
227   var data = client.data
228     , msgType = data.getByte()
229     , currentPlayerId
230     , reply
231     , player;
232 
233   // Registeröinniltä paketti
234   if (client.data.clientId === 544437095) {
235     if (String(client.data.memBlock.slice(4)) === 'GSS+') {
236       this.emit('register', client.data);
237     } else if (String(client.data.memBlock.slice(4)) === 'PING') {
238       reply = new Packet(4);
239       reply.putString('PONG');
240       client.reply(reply);
241     }
242     return;
243   }
244 
245   // Onko servu sammumassa?
246   if (this.gameState.closing) {
247     reply = new Packet(2);
248     reply.putByte(NET.SERVERCLOSING);
249     reply.putByte(NET.END);
250     client.reply(reply);
251     return;
252   }
253 
254   if (msgType === NET.LOGIN) {
255     // Login paketissa ei ole pelaajan ID:tä vielä, joten se on käsiteltävä erikseen
256     this.emit(NET.LOGIN, client);
257     return;
258   }
259 
260   // Luetaan lähetetty pelaajan ID, joka on pelaajan järjestysnumero ja aina väliltä 1...MAX_PLAYERS
261   currentPlayerId = data.getByte();
262   // Tai jos ei ole niin sitten ei päästetä sisään >:(
263   if (currentPlayerId < 1 || currentPlayerId > this.gameState.maxPlayers) {
264     log.notice('Possible hack attempt from ' + client.address + ' Invalid player ID (' + currentPlayerId + ')');
265     return;
266   }
267 
268   // Haetaan pelaajan instanssi Player-luokasta
269   player = this.players[currentPlayerId];
270 
271   // Tarkistetaan onko pelaaja potkittu
272   if (player.kicked && player.clientId === client.id) {
273     reply = new Packet(7);
274     reply.putByte(NET.KICKED);
275     if (player.kicker) {
276       reply.putByte(player.kicker.id);
277     } else {
278       // Palvelin potkaisi
279       reply.putByte(0);
280     }
281     reply.putByte(currentPlayerId);
282     reply.putString(player.kickReason);
283     client.reply(reply);
284     return;
285   }
286   // Vielä yksi tarkistus
287   if (player.clientId !== client.id || !player.active) {
288     reply = new Packet(1);
289     reply.putByte(NET.NOLOGIN);
290     client.reply(reply);
291     return;
292   }
293 
294   // Logout on erikseen, koska sen jälkeen ei varmasti tule mitään muuta
295   if (msgType === NET.LOGOUT) {
296     this.emit(NET.LOGOUT, client, player);
297     return;
298   }
299 
300   // Lasketaan pelaajan ja serverin välinen lagi
301   player.lag = Date.now() - player.lastActivity;
302   // Päivitetään pelaajan olemassaolo
303   player.lastActivity = Date.now();
304 
305   // Luupataan kaikkien pakettien läpi
306   while (msgType) {
307     // Lähetetään tietoa paketista käsiteltäväksi
308     this.emit(msgType, client, player);
309     msgType = data.getByte();
310   }
311 
312   // Jos erä on päättynyt niin lähetetään kaikkien pelaajien kaikki tiedot
313   if (this.gameState.sessionComplete) {
314     player.sendNames = true;
315   }
316 
317   // Lähetetään dataa pelaajalle
318   this.sendReply(client, player);
319 
320   // Valmis! :)
321 };

323 /**
324  * Hoitaa datan lähetyksen.
325  *
326  * @param {cbNetwork.Client} client  cbNetworkin Client-luokan instanssi
327  * @param {Player} player            Pelaaja, keneltä on saatu dataa ja kenelle lähetetään vastaus tässä.
328  * @see <a href="http://vesq.github.com/cbNetwork-node/doc/symbols/Client.html">cbNetwork.Client</a>
329  */
330 Server.prototype.sendReply = function (client, player) {
331   var reply = new Packet()
332     , playerIds = Object.keys(this.players)
333     , plr
334     , server = this
335     , timeLeft;
336 
337   // Lähetetään kaikkien pelaajien tiedot
338   this.loopPlayers(function (plr) {
339     // Onko pyydetty nimet
340     if (player.sendNames) {
341       if (plr.active) {
342         reply.putByte(NET.PLAYERNAME);  // Nimet
343         reply.putByte(plr.id);          // Pelaajan tunnus
344         reply.putString(plr.name);      // Nimi
345         reply.putByte(plr.zombie);      // Onko botti
346         reply.putByte(plr.team);        // Joukkue
347       }
348     }
349 
350     // Lähetetään niiden pelaajien tiedot jotka ovat hengissä ja näkyvissä
351     if (plr.active) {
352       var x1 = player.x
353         , y1 = player.y
354         , x2 = plr.x
355         , y2 = plr.y
356         , visible = !((Math.abs(x1 - x2) > 450*2) || (Math.abs(y1 - y2) > 350*2));
357 
358       // Onko näkyvissä vai voidaanko muuten lähettää
359       if (player.sendNames || visible || plr.health <= 0) {
360         // Näkyy
361         reply.putByte(NET.PLAYER); // Pelaajan tietoja
362         reply.putByte(plr.id);     // Pelaajan tunnus
363         reply.putShort(plr.x);     // Sijainti
364         reply.putShort(plr.y);     // Sijainti
365         reply.putShort(plr.angle); // Kulma
366 
367         // Spawn-protect
368         var isProtected = 0;
369         if (plr.spawnTime + server.config.spawnProtection > Date.now()) {
370           isProtected = 1;
371         }
372 
373         // Muutetaan team arvo välille 0-1
374         var teamBit = (plr.team === 2 ? 1 : 0);
375 
376         // Tungetaan yhteen tavuun useampi muuttuja
377         var b = ((plr.weapon % 16) << 0)  // Ase (bitit 0-3)
378               + ((plr.hasAmmos << 4))     // Onko ammuksia (bitti 4)
379               + ((teamBit << 6))          // Joukkue/tiimi (bitti 6)
380               + ((isProtected << 7));     // Haavoittumaton (bitti 7)
381         reply.putByte(b);
382 
383         if (plr.health <= 0) {
384           // Client-puolella health-arvo lasketaan seuraavasta tavusta niin, että jos se ylittää
385           // 128, on oikea health-määrä vastaanotettu tavu - 256.
386           reply.putByte(Math.min(255, Math.max(0, plr.health + 256)));
387         } else {
388           reply.putByte(plr.health);      // Terveys
389         }
390         reply.putShort(plr.kills);      // Tapot
391         reply.putShort(plr.deaths);     // Kuolemat
392       } else if (server.gameState.radarArrows || (server.gameState.gameMode > 1 && player.team === plr.team)) {
393         // Ei näy. Lähetetään tutkatieto. gameMode > 1 tarkoittaa kaikkia muita kuin DM-moodeja
394         // Lähetetään tutkatiedot jos joukkueet ovat samat tai asetuksista on laitettu että
395         // kaikkien joukkueiden pelaajien tutkatiedot lähetetään
396         reply.putByte(NET.RADAR); // Tutkatietoa tulossa
397         var angle = Math.atan2(y1 - y2, x1 - x2) + Math.PI; // Kulma radiaaneina välillä 0...2pi
398         reply.putByte((angle / (2 * Math.PI)) * 255); // Kulma muutettuna välille 0-255
399         reply.putByte(plr.team);  // Pelaajan joukkue
400       }
401     }
402   });
403 
404   // Kartan vaihtaminen, mikäli tarpeellista
405   if (this.gameState.sessionComplete && this.config.map.length > 1 && player.mapName !== this.gameState.map.name) {
406     timeLeft = this.gameState.sessionStarted + this.config.periodLength * 1000 - Date.now();
407     if (timeLeft < -5) {
408       reply.putByte(NET.MAPCHANGE);
409       reply.putString(this.gameState.map.name);
410       reply.putInt(this.gameState.map.crc32);
411     }
412   }
413 
414   // Lähetetään kaikki pelaajalle osoitetut viestit
415   this.messages.fetch(player, reply);
416 
417 
418   // Jos on pyydetty nimilista niin palautetaan myös kaikkien tavaroiden tiedot
419   if (player.sendNames) {
420     player.sendNames = false;
421     var itemIds = Object.keys(this.items);
422     for (var i = itemIds.length; i--;) {
423       var item = this.items[itemIds[i]];
424       this.messages.add(player.id, {
425         msgType: NET.ITEM,
426         itemId: item.id,
427         itemType: item.type,
428         x: item.x,
429         y: item.y
430       });
431     }
432   }
433 
434   // Pelisession aikatiedot
435   reply.putByte(NET.SESSIONTIME);
436   reply.putInt(this.config.periodLength);                       // Erän pituus
437   reply.putInt((Date.now() - this.gameState.sessionStarted) / 1000); // Kuinka kauan on pelattu
438   reply.putByte(this.gameState.sessionComplete);                // Onko erä loppu
439 
440   // Tieto siitä että debug-viestejä voi laittaa uudelleen
441   player.debugState = 0;
442 
443   reply.putByte(NET.END);
444   client.reply(reply);
445 
446   // Dodiin, valmiita ollaan :)
447   return;
448 };

474 /**
475  * Kirjaa pelaajan sisään peliin.
476  * @param {cbNetwork.Client} client  cbNetworkin Client-luokan jäsen.
477  * @returns {Boolean}                Onnistuiko pelaajan liittäminen peliin vai ei.
478  * @see <a href="http://vesq.github.com/cbNetwork-node/doc/symbols/Client.html">cbNetwork.Client</a>
479  */
480 Server.prototype.login = function (client) {
481   var data = client.data
482     , version = data.getString()
483     , replyData
484     , nickname
485     , playerIds
486     , randomPlace
487     , player
488     , teamCheckLoop, reds = 0, greens = 0;
489 
490   // Täsmääkö clientin ja serverin versiot
491   if (version !== this.VERSION) {
492     log.notice('Player trying to connect with incorrect client version.');
493     replyData = new Packet(3);
494     replyData.putByte(NET.LOGIN);
495     replyData.putByte(NET.LOGINFAILED);
496     replyData.putByte(NET.WRONGVERSION);
497     replyData.putString(this.VERSION);
498     client.reply(replyData);
499     return;
500   }
501 
502   // Versio on OK, luetaan pelaajan nimi
503   nickname = data.getString().trim();
504   log.info('Player %0 is trying to login...', nickname.green);
505 
506   // Tarkistetaan onko palvelin täynnä
507   if (this.gameState.playerCount + this.gameState.botCount >= this.gameState.maxPlayers) {
508     // Vapaita paikkoja ei ollut
509     log.info(' -> Server is full!');
510     replyData = new Packet(3);
511     replyData.putByte(NET.LOGIN);
512     replyData.putByte(NET.LOGINFAILED);
513     replyData.putByte(NET.TOOMANYPLAYERS);
514     client.reply(replyData);
515     return;
516   }
517 
518   // Käydään kaikki nimet läpi ettei samaa nimeä vain ole jo suinkin olemassa
519   playerIds = Object.keys(this.players);
520   for (var i = 0; i < playerIds.length; i++) {
521     player = this.players[playerIds[i]];
522     if (player.name.toLowerCase() === nickname.toLowerCase()) {
523       if (player.kicked || !player.active) {
524         player.name = "";
525       } else {
526         // Nimimerkki oli jo käytössä.
527         log.info(' -> Nickname %0 already in use.', nickname.green);
528         replyData = new Packet(3);
529         replyData.putByte(NET.LOGIN);
530         replyData.putByte(NET.LOGINFAILED);
531         replyData.putByte(NET.NICKNAMEINUSE);
532         client.reply(replyData);
533         return;
534       }
535     }
536   }
537 
538   // Etsitään inaktiivinen pelaaja
539   for (i = 0; i < playerIds.length; i++) {
540     player = this.players[playerIds[i]];
541     if (!player.active) {
542       // Tyhjä paikka löytyi
543       player.clientId = client.id;
544       player.active = true;
545       player.loggedIn = false;
546       player.name = nickname;
547       randomPlace = this.gameState.map.findSpot();
548       player.x = randomPlace.x;
549       player.y = randomPlace.y;
550       player.hackTestX = player.x;
551       player.hackTestY = player.y;
552       player.angle = Math.floor(Math.random() * 360 + 1);
553       player.zombie = false;
554       player.health = 100;
555       player.kills = 0;
556       player.deaths = 0;
557       player.weapon = WPN.PISTOL;
558       player.lastActivity = new Date().getTime();
559       player.spawnTime = player.lastActivity;
560       player.admin = false;
561       player.kicked = false;
562       player.kickReason = "";
563       player.setTeamEvenly();
564 
565       this.gameState.playerCount++;
566 
567       // Lähetetään vastaus clientille
568       replyData = new Packet(16);
569       replyData.putByte(NET.LOGIN);
570       replyData.putByte(NET.LOGINOK);
571       replyData.putByte(player.id);
572       // Zombiemoodi on täysin palvelinpuolen moodi joka näytetään klienteille TDM:nä
573       replyData.putByte(this.gameState.gameMode === 3 ? 2 : this.gameState.gameMode);
574       replyData.putString(this.gameState.map.name);
575       replyData.putInt(this.gameState.map.crc32);
576       replyData.putString(this.config.mapDownloadUrl); // Kartan URL josta sen voi ladata, mikäli se puuttuu
577       client.reply(replyData);
578       log.info(' -> login successful, assigned ID (%0)', String(player.id).magenta);
579 
580       // Päivitetään tiedot servulistaukseen
581       this.registration.update();
582 
583       // Lisätään viestijonoon ilmoitus uudesta pelaajasta, kaikille muille paitsi boteille ja itselle.
584       this.messages.addToAll({
585         msgType: NET.LOGIN,
586         msgText: nickname,
587         player: player
588       }, player.id);
589       return;
590     }
591   }
592 };

594 /**
595  * Kirjaa pelaajan ulos pelistä.
596  * @param {Player} player  Pelaaja joka kirjataan ulos
597  */
598 Server.prototype.logout = function (player) {
599   player.active = false;
600   player.loggedIn = false;
601   player.admin = false;
602   log.info('%0 logged out.', player.name.green);
603 
604   // Vähennetään pelaajamäärää vain jos kyseessä ei ollut botti
605   if (!player.zombie) {
606     this.gameState.playerCount--;
607   }
608 
609   // Päivitetään tiedot servulistaukseen
610   this.registration.update();
611

669 /** Sammuttaa palvelimen. Emittoi eventit {@link Server#closing} ja {@link Server#closed} */
670 Server.prototype.close = function (now) {
671   if (this.gameState.closing) {
672     // Ollaan jo sulkemassa, ei aloiteta samaa prosessia uudelleen.
673     return;
674   }
675   this.gameState.closing = true;
676   log.info('Server going down...');
677   this.emit('closing');
678 
679   // Pysäytetään Game-moduulin päivitys
680   this.game.stop();
681 
682   // Pyydetään, että palvelin poistetaan listauksesta
683   if (this.registration.registered) {
684     log.info('Unregistering server...');
685     this.registration.remove(function (e) {
686       if (e) { log.error(e); }
687       else   { log.info('Server unregistered.'); }
688     });
689   }
690 
691   var self = this;
692   setTimeout(function closeServer() {
693     if (self.server) {
694       self.server.close();
695     }
696     self.emit('closed');
697     process.exit();
698   }, now ? 0 : 1000);
699 };

701 /** Alustaa botit. */
702 Server.prototype.initBots = function () {
703   var server = this
704     , botCount = this.gameState.botCount
705     , loopedBotsCount = 0
706     , team = rand(1, 2)
707     , map = this.gameState.map;
708 
709   this.loopPlayers(function serverInitBots(plr) {
710     var randomPlace;
711 
712     // Tarkistetaan, että pysytään bottimäärän sisällä eikä muokata ihmispelaajaa
713     if (loopedBotsCount >= botCount || (plr.active && !plr.zombie)) {
714       return;
715     }
716 
717     plr.clientId  = 'bot:' + plr.id;
718     plr.name      = plr.botName;
719     plr.zombie    = true;
720     plr.active    = true;
721     plr.loggedIn  = true;
722     plr.isDead    = false;
723     plr.health    = 100;
724     if (!server.gameState.sessionComplete) {
725       // Nollataan bottien statsit vain jos ei olla erän lopetustilassa
726       plr.kills     = 0;
727       plr.deaths    = 0;
728     }
729     plr.weapon    = server.getBotWeapon();
730     randomPlace = map.findSpot();
731     plr.x = randomPlace.x;
732     plr.y = randomPlace.y;
733     plr.lastValidX = plr.x;
734     plr.lastValidY = plr.y;
735     plr.angle = rand(0, 360);
736     if (server.gameState.gameMode === 3) {
737       // Zombie-modi, botit vastaan pelaajat ja boteilla on 10hp
738       plr.team = 2;
739       plr.health = 10;
740     } else if (server.gameState.gameMode > 1) {
741       plr.team = team;
742       team++;
743       if (team > 2) { team = 1; }
744     } else {
745       plr.team = 1;
746     }
747 
748     // Luodaan botille tekoäly ja asetetaan sen taitotaso samaksi kuin pelaaja-ID
749     plr.botAI = new BotAI(server, plr);
750     plr.botAI.setSkill(plr.id);
751 
752     loopedBotsCount++;
753   });
754 };

756 /**
757  * Lisää uuden botin, vaihtoehtoisesti jonkin epäaktiivisen pelaajan paikalle. Huom! Ei välitä
758  * Server.gameState.botCount arvosta, vaan botti lentää heti pihalle jos määrä ylittyy. Katso
759  * lisätietoja: {@link Game#updateBotsAmount}.
760  *
761  * @param {Player} [player]  Pelaaja, jonka paikalle bottia yritetään laittaa.
762  *
763  * @returns {Boolean}  Onnistuiko botin lisääminen
764  */
765 Server.prototype.addBot = function (player) {
766   var bot, playerIds, plr, randomPlace;
767 
768   if (player instanceof Player) {
769     // Tarkistetaan ettei paikalla ole jo aktiivista pelaajaa
770     if (player.active) {
771       return false;
772     }
773     bot = player;
774   } else {
775     // Haetaan seuraava vapaa paikka botille
776     playerIds = Object.keys(this.players);
777     for (var i = 0; i < playerIds.length; i++) {
778       plr = this.players[playerIds[i]];
779       if (!plr.active) {
780         bot = plr;
781         break;
782       }
783     }
784   }
785 
786   // Tarkistetaan onko meillä nyt sopiva objekti valmiina
787   if (!(bot instanceof Player)) {
788     // Ei ollut.
789     return false;
790   }
791 
792   bot.clientId  = 'bot:' + bot.id;
793   bot.name      = bot.botName;
794   bot.zombie    = true;
795   bot.active    = true;
796   bot.loggedIn  = true;
797   bot.isDead    = false;
798   bot.health    = 100;
799   bot.kills     = 0;
800   bot.deaths    = 0;
801   bot.weapon = this.getBotWeapon();
802   randomPlace = this.gameState.map.findSpot();
803   bot.x = randomPlace.x;
804   bot.y = randomPlace.y;
805   bot.lastValidX = bot.x;
806   bot.lastValidY = bot.y;
807   bot.angle = rand(0, 360);
808   if (this.gameState.gameMode === 3) {
809     // Zombie-modi, botit vastaan pelaajat ja boteilla on 10hp
810     bot.health = 10;
811   }
812   // Arvotaan botille joukkue
813   bot.setTeamEvenly();
814 
815   // Luodaan botille tekoäly ja asetetaan sen taitotaso samaksi kuin pelaaja-ID
816   bot.botAI = new BotAI(this, bot);
817   bot.botAI.setSkill(bot.id);
818 
819   // Lisätään viestijonoon ilmoitus uudesta pelaajasta
820   this.messages.addToAll({
821     msgType: NET.LOGIN,
822     msgText: bot.name,
823     player: bot
824   });
825 
826   log.info('%0 joined the forces of AI.', bot.name.green);
827 };

829 /** Arpoo aseen boteille sallittujen listalta. */
830 Server.prototype.getBotWeapon = function () {
831   var weapons;
832 
833   if (this.config.botWeapons && this.config.botWeapons.length > 0) {
834     weapons = this.config.botWeapons;
835   } else if ('undefined' === this.gameState.map.config.botWeapons) {
836     weapons = [1, 2, 3, 4, 5, 6];
837   } else {
838     weapons = this.gameState.map.config.botWeapons;
839   }
840   return weapons[rand(0, weapons.length - 1)];
841 };

843 /**
844  * Luo uuden ammuksen.
845  *
846  * @param {Player} player  Pelaaja joka ampui ammuksen
847  */
848 Server.prototype.createBullet = function (player) {
849   var bullet, bulletAmount;
850 
851   // Ei tehdä ammusta jos erä on päättynyt
852   if (this.gameState.sessionComplete) {
853     return;
854   }
855 
856   if ('undefined' === typeof player) {
857     log.error("Tried to create a bullet for an undefined player!");
858     return;
859   }
860 
861   switch (player.weapon) {
862     case WPN.LAUNCHER:
863       // Kranaatinheittimestä lähtee kaksi ammusta
864       bulletAmount = 2;
865       break;
866     case WPN.SHOTGUN:
867       // Haulikosta lähtee kuusi ammusta
868       bulletAmount = 6;
869       break;
870     default:
871       // Oletuksena ammutaan vain yksi kuti
872       bulletAmount = 1;
873   }
874 
875   for (var i = 1; i <= bulletAmount; i++) {
876     if (player.weapon === WPN.LAUNCHER && i === 2) {
877       // Toinen ammus kranaatinlaukaisimesta lähtee peilattuna
878       bullet = new Bullet(this, player, ++this.lastBulletId, true);
879     } else {
880       bullet = new Bullet(this, player, ++this.lastBulletId);
881     }
882     if (!bullet.failed && bullet.initialize()) {
883       // Jos ammuksen alustus onnistui (esim. ei ammuttu seinän sisällä), lisätään se listaan.
884       this.bullets[bullet.id] = bullet;
885       // Lisätään ammusviesti lähetettäväksi jokaiselle pelaajalle
886       this.messages.addToAll({
887         msgType: NET.NEWBULLET,          // Viestin tyyppi
888         bullet: bullet,                  // Ammus
889         sndPlay: (i === 1),              // Ääni toistetaan vain ensimmäisen luodin kohdalla
890         weapon: player.weapon,           // Millä aseella ammuttiin
891         player: player,                  // Kuka ampui
892         x: bullet.x,                     // Mistä ammus lähti
893         y: bullet.y,                     // Mistä ammus lähti
894         handShooted: player.handShooted  // Kummalla kädellä ammuttiin (pistooli)
895       });
896     } else {
897       // Ammuksen luonti epäonnistui
898       this.lastBulletId--;
899       log.debug('Failed to initialize a new bullet shot by %0', player.name.green);
900     }
901   }
902 };

904 /**
905  * Kartan vaihto. Jos parametrina annetaan merkkijono, niin vaihdetaan kyseiseen karttaan.
906  * Muulloin edetään karttalistassa määriteltyyn seuraavaan karttaan.
907  *
908  * @param {String} [mapName]  Kartta johon vaihdetaan
909  *
910  * @returns {Boolean}  Onnistuiko kartan vaihto
911  */
912 Server.prototype.changeMap = function (mapName) {
913   var nextMapName, mapPath, nextMap;
914 
915   if ('string' === typeof mapName) {
916     nextMapName = mapName;
917   } else {
918     // Ei annettu parametrina kartan nimeä, joten mennään listassa eteenpäin
919     this.gameState.mapNumber++;
920     if (this.gameState.mapNumber >= this.config.map.length) {
921       this.gameState.mapNumber = 0;
922     }
923 
924     nextMapName = this.config.map[this.gameState.mapNumber];
925   }
926 
927   log.info('Changing map to %0', nextMapName.green);
928 
929   // Tarkistetaan onko kartta jo ladattu muistiin
930   if (this.maps[nextMapName]) {
931     // Oli ladattu, joten ei tarvitse alustaa uutta
932     nextMap = this.maps[nextMapName];
933   } else {
934     // Karttaa ei ollut vielä olemassa joten luodaan uusi
935     nextMap = new Map(this, nextMapName);
936     if (!nextMap.loaded) {
937       // Kartan lataus epäonnistui
938       log.error('Could not load map "%0"', nextMapName);
939       return false;
940     }
941 
942     // Laitetaan kartta muistiin ettei sitä tarvitse ladata enää uudelleen
943     this.maps[nextMapName] = nextMap;
944   }
945 
946   // Jos kartalla on botCount-asetus, asetetaan se botCountiksi, mikäli configissa se on < 0
947   if ('number' === typeof nextMap.config.botCount && this.config.botCount < 0) {
948     this.gameState.botCount = nextMap.config.botCount;
949   }
950 
951   // Jos kartalla on botDepartLimit-asetus, asetetaan se nykyiseksi, mikäli configissa se on < 0
952   if ('number' === typeof nextMap.config.botDepartLimit && this.config.botDepartLimit < 0) {
953     this.gameState.botDepartLimit = nextMap.config.botDepartLimit;
954   }
955 
956   // Alustetaan tavarat paikoilleen
957   this.items = {};
958   nextMap.initItems();
959 
960   // Asetetaan uusi kartta nykyisen kartan paikalle
961   this.gameState.map = nextMap;
962 
963   // Alustetaan botit
964   this.initBots();
965 
966   // Tapetaan kaikki pelaajat
967   this.loopPlayers(function mapChangeKill(player) {
968     player.health = -10;
969     player.timeToDeath = Date.now();
970   });
971 };
