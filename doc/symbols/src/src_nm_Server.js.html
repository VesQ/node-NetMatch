 33 /**
 34  * Luo uuden palvelimen annettuun porttiin ja osoitteeseen. Kun palvelimeen tulee dataa, emittoi
 35  * se siihen kuuluvan eventin paketin ensimmäisen tavun perusteella. esim. NET.LOGIN
 36  *
 37  * @class Yleiset Serverin funktiot ja ominaisuudet
 38  *
 39  * @param {Number} port            Portti, jota palvelin kuuntelee.
 40  * @param {String} [address]       IP-osoite, jota palvelin kuuntelee. Jos tätä ei anneta, järjestelmä
 41  *                                 yrittää kuunnella kaikkia osoitteita (0.0.0.0).
 42  * @param {Boolean} [debug=false]  Spämmitäänkö konsoliin paljon "turhaa" tietoa?
 43  */
 44 function Server(args, version) {
 45   /** Palvelimen debug-tila */
 46   this.debug = args.d;
 47 
 48   if (this.debug) { log.notice('Server running on debug mode, expect spam!'.red); }
 49 
 50   /**
 51    * Palvelimen versio
 52    * @const
 53   */
 54   this.VERSION = version;
 55 
 56   /** Sisältää pelin nykyisestä tilanteesta kertovat muuttujat. */
 57   this.gameState = {};
 58 
 59   /** Sisältää kaikki kartat */
 60   this.maps = {};
 61 
 62   /** Sisältää palvelimen pelaajat, eli luokan {@link Player} jäsenet. */
 63   this.players = {};
 64 
 65   /** Sisältää palvelimen ammukset, eli luokan {@link Bullet} jäsenet. */
 66   this.bullets = {};
 67   /** @private */
 68   this.lastBulletId = 0;
 69 
 70   /** Sisältää palvelimella maassa olevat tavarat, kts. {@link Item}. */
 71   this.items = {};
 72 
 73   // Alustetaan moduulit
 74 
 75   /**
 76    * Pelaajille lähetettävät viestit. Tämä on instanssi {@link NetMessages}-luokasta.
 77    * @type NetMessages
 78    */
 79   this.messages = new NetMsgs(this);
 80 
 81   /**
 82    * Asetukset tälle palvelimelle
 83    * @type Config
 84    */
 85   this.config = new Config(this);
 86 
 87   /**
 88    * Sisältää palvelimella pyörivän {@link Game}-luokan instanssin
 89    * @type Game
 90    */
 91   this.game = new Game(this);
 92 
 93   /**
 94    * Sisältää palvelimen komennot, joita voidaan kutsua joko palvelimen konsolista tai klientiltä
 95    * @type Commands
 96    */
 97   this.commands = new Command(this);
 98 
 99   /**
100    * Sisältää palvelimen konsoli-io:n käsittelyyn käytettävän {@link Input}-luokan instanssin
101    * @type Input
102    */
103   this.input = new Input(this);
104 
105   /**
106    * Tämän palvelimen palvelinlistaukseen liittyvä toiminallisuus
107    * @type Registration
108    */
109   this.registration = new Registration(this);
110 
111   // Alustetaan palvelin (esim. kartta, pelaajat, tavarat)
112   this.initialize(args.p, args.a, args.c);
113 
114   log.info('Server initialized successfully.');
115 }

119 /** Alustaa palvelimen */
120 Server.prototype.initialize = function (port, address, config) {
121   var self = this;
122 
123   // Ladataan konffit
124   this.config.load(config);
125   // Komentoriviparametrit voittaa.
126   if (port) { this.config.port = port; }
127   if (address) { this.config.address = address; }
128 
129   /**
130    * cbNetwork-node UDP-palvelin
131    * @type cbNetwork.Server
132    * @see <a href="http://vesq.github.com/cbNetwork-node/doc/symbols/Server.html">cbNetwork.Server</a>
133    */
134   this.server = new cbNetwork.Server(this.config.port, this.config.address);
135 
136   this.server.on('message', function recvMsg(client) {
137     self.handlePacket(client);
138   });
139 
140   this.server.on('close', function onClose() {
141     if (!self.gameState.closing) {
142       // Hups! cbNetwork serveri kaatui alta.
143       log.fatal('cbNetwork unexpectedly closed, server will terminate.');
144       self.close(true);
145     }
146     // Muussa tapauksessa sulkeutuminen oli odotettavissa
147   });
148 
149   // Alustetaan pelitilanne
150   this.gameState.playerCount = 0;
151   this.gameState.botCount = 5;
152   this.gameState.gameMode = this.config.gameMode;
153   this.gameState.maxPlayers = this.config.maxPlayers;
154   this.gameState.radarArrows = this.config.radarArrows;
155 
156   // Ladataan kartta
157   this.gameState.map = new Map(this, this.config.map);
158   if (!this.gameState.map.loaded) {
159     log.fatal('Could not load map "%0"', this.config.map);
160     this.close();
161     return;
162   }
163 
164   // Jos kartalla on botCount-asetus, overridaa se configin
165   if ('number' === typeof this.gameState.map.config.botCount) {
166     this.gameState.botCount = this.gameState.map.config.botCount;
167   }
168 
169   // UNIMPLEMENTED Mappien rotaatio
170   this.maps[this.gameState.map.name] = this.gameState.map;
171 
172   // Alustetaan pelaajat
173   for (var i = 1; i <= this.gameState.maxPlayers; ++i) {
174     this.players[i] = new Player(this, i);
175   }
176 
177   // Alustetaan botit
178   this.initBots();
179 
180   // Lisätäänkö palvelin palvelinlistaukseen
181   if (this.config.register) {
182     this.registration.apply(function initRegister(e) {
183       if (e) { log.error(e); }
184       else   { log.info('Server registered successfully.'); }
185     });
186   }
187 
188   // Käynnistetään pelimekaniikka, joka päivittyy 60 kertaa sekunnissa
189   this.game.start(30);
190 };

192 /**
193  * Hoitaa saapuneiden viestien käsittelyn.
194  *
195  * @param {cbNetwork.Client}  cbNetwork-noden Client-luokan instanssi, jolta dataa tulee.
196  * @see <a href="http://vesq.github.com/cbNetwork-node/doc/symbols/Client.html">cbNetwork.Client</a>
197  */
198 Server.prototype.handlePacket = function (client) {
199   var data = client.data
200     , msgType = data.getByte()
201     , currentPlayerId
202     , reply
203     , player;
204 
205   // Registeröinniltä paketti
206   if (client.data.clientId === 544437095) {
207     if (String(client.data.memBlock.slice(4)) === 'GSS+') {
208       this.emit('register', client.data);
209     } else if (String(client.data.memBlock.slice(4)) === 'PING') {
210       reply = new Packet(4);
211       reply.putString('PONG');
212       client.reply(reply);
213     }
214     return;
215   }
216 
217   // Onko servu sammumassa?
218   if (this.gameState.closing) {
219     reply = new Packet(2);
220     reply.putByte(NET.SERVERCLOSING);
221     reply.putByte(NET.END);
222     client.reply(reply);
223     return;
224   }
225 
226   if (msgType === NET.LOGIN) {
227     // Login paketissa ei ole pelaajan ID:tä vielä, joten se on käsiteltävä erikseen
228     this.emit(NET.LOGIN, client);
229     return;
230   }
231 
232   // Luetaan lähetetty pelaajan ID, joka on pelaajan järjestysnumero ja aina väliltä 1...MAX_PLAYERS
233   currentPlayerId = data.getByte();
234   // Tai jos ei ole niin sitten ei päästetä sisään >:(
235   if (currentPlayerId < 1 || currentPlayerId > this.gameState.maxPlayers) {
236     log.notice('Possible hack attempt from ' + client.address + ' Invalid player ID (' + currentPlayerId + ')');
237     return;
238   }
239 
240   // Haetaan pelaajan instanssi Player-luokasta
241   player = this.players[currentPlayerId];
242 
243   // Tarkistetaan onko pelaaja potkittu
244   if (player.kicked && player.clientId === client.id) {
245     reply = new Packet(7);
246     reply.putByte(NET.KICKED);
247     if (player.kicker) {
248       reply.putByte(player.kicker.id);
249     } else {
250       // Palvelin potkaisi
251       reply.putByte(0);
252     }
253     reply.putByte(currentPlayerId);
254     reply.putString(player.kickReason);
255     client.reply(reply);
256     return;
257   }
258   // Vielä yksi tarkistus
259   if (player.clientId !== client.id || !player.active) {
260     reply = new Packet(1);
261     reply.putByte(NET.NOLOGIN);
262     client.reply(reply);
263     return;
264   }
265 
266   // Logout on erikseen, koska sen jälkeen ei varmasti tule mitään muuta
267   if (msgType === NET.LOGOUT) {
268     this.emit(NET.LOGOUT, client, player);
269     return;
270   }
271 
272   // Lasketaan pelaajan ja serverin välinen lagi
273   player.lag = Date.now() - player.lastActivity;
274   // Päivitetään pelaajan olemassaolo
275   player.lastActivity = Date.now();
276 
277   // Luupataan kaikkien pakettien läpi
278   while (msgType) {
279     // Lähetetään tietoa paketista käsiteltäväksi
280     this.emit(msgType, client, player);
281     msgType = data.getByte();
282   }
283 
284   // Lähetetään dataa pelaajalle
285   this.sendReply(client, player);
286 
287   // Valmis! :)
288 };

290 /**
291  * Hoitaa datan lähetyksen.
292  *
293  * @param {cbNetwork.Client} client  cbNetworkin Client-luokan instanssi
294  * @param {Player} player            Pelaaja, keneltä on saatu dataa ja kenelle lähetetään vastaus tässä.
295  * @see <a href="http://vesq.github.com/cbNetwork-node/doc/symbols/Client.html">cbNetwork.Client</a>
296  */
297 Server.prototype.sendReply = function (client, player) {
298   var reply = new Packet()
299     , playerIds = Object.keys(this.players)
300     , plr
301     , server = this;
302 
303   // Lähetetään kaikkien pelaajien tiedot
304   this.loopPlayers(function (plr) {
305     // Onko pyydetty nimet
306     if (player.sendNames) {
307       if (plr.active) {
308         reply.putByte(NET.PLAYERNAME);  // Nimet
309         reply.putByte(plr.id);          // Pelaajan tunnus
310         reply.putString(plr.name);      // Nimi
311         reply.putByte(plr.zombie);      // Onko botti
312         reply.putByte(plr.team);        // Joukkue
313       }
314     }
315 
316     // Lähetetään niiden pelaajien tiedot jotka ovat hengissä ja näkyvissä
317     if (plr.active) {
318       var x1 = player.x
319         , y1 = player.y
320         , x2 = plr.x
321         , y2 = plr.y
322         , visible = !((Math.abs(x1 - x2) > 450*2) || (Math.abs(y1 - y2) > 350*2));
323 
324       // Onko näkyvissä vai voidaanko muuten lähettää
325       if (player.sendNames || visible || plr.health <= 0) {
326         // Näkyy
327         reply.putByte(NET.PLAYER); // Pelaajan tietoja
328         reply.putByte(plr.id);     // Pelaajan tunnus
329         reply.putShort(plr.x);     // Sijainti
330         reply.putShort(plr.y);     // Sijainti
331         reply.putShort(plr.angle); // Kulma
332 
333         // Spawn-protect
334         var isProtected = 0;
335         if (plr.spawnTime + server.config.spawnProtection > Date.now()) {
336           isProtected = 1;
337         }
338 
339         // Muutetaan team arvo välille 0-1
340         var teamBit = (plr.team === 2 ? 1 : 0);
341 
342         // Tungetaan yhteen tavuun useampi muuttuja
343         var b = ((plr.weapon % 16) << 0)  // Ase (bitit 0-3)
344               + ((plr.hasAmmos << 4))     // Onko ammuksia (bitti 4)
345               + ((teamBit << 6))          // Joukkue/tiimi (bitti 6)
346               + ((isProtected << 7));     // Haavoittumaton (bitti 7)
347         reply.putByte(b);
348 
349         if (plr.health <= 0) {
350           // Client-puolella health-arvo lasketaan seuraavasta tavusta niin, että jos se ylittää
351           // 128, on oikea health-määrä vastaanotettu tavu - 256.
352           reply.putByte(Math.min(255, Math.max(0, plr.health + 256)));
353         } else {
354           reply.putByte(plr.health);      // Terveys
355         }
356         reply.putShort(plr.kills);      // Tapot
357         reply.putShort(plr.deaths);     // Kuolemat
358       } else if (server.gameState.radarArrows || server.gameState.playMode === 2) {
359         // Ei näy. Lähetetään tutkatieto. playMode === 2 tarkoittaa TDM-pelimuotoa
360         if (player.team === plr.team || server.gameState.radarArrows) {
361           // Lähetetään tutkatiedot jos joukkueet ovat samat tai asetuksista on laitettu että
362           // kaikkien joukkueiden pelaajien tutkatiedot lähetetään
363           reply.putByte(NET.RADAR); // Tutkatietoa tulossa
364           var angle = Math.atan2(y1 - y2, x1 - x2) + Math.PI; // Kulma radiaaneina välillä 0...2pi
365           reply.putByte((angle / (2 * Math.PI)) * 255); // Kulma muutettuna välille 0-255
366           reply.putByte(plr.team);  // Pelaajan joukkue
367         }
368       }
369     }
370   });
371 
372   // UNIMPLEMENTED
373   // Kartan vaihtaminen
374 
375   // Lähetetään kaikki pelaajalle osoitetut viestit
376   this.messages.fetch(player, reply);
377 
378 
379   // Jos on pyydetty nimilista niin palautetaan myös kaikkien tavaroiden tiedot
380   if (player.sendNames) {
381     player.sendNames = false;
382     var itemIds = Object.keys(this.items);
383     for (i = itemIds.length; i--;) {
384       var item = this.items[itemIds[i]];
385       this.messages.add(player.id, {
386         msgType: NET.ITEM,
387         itemId: item.id,
388         itemType: item.type,
389         x: item.x,
390         y: item.y
391       });
392     }
393   }
394 
395   // UNIMPLEMENTED
396   // Pelisession aikatiedot
397   player.debugState = 0;
398 
399   reply.putByte(NET.END);
400   client.reply(reply);
401 
402   // Dodiin, valmiita ollaan :)
403   return;
404 };

430 /**
431  * Kirjaa pelaajan sisään peliin.
432  * @param {cbNetwork.Client} client  cbNetworkin Client-luokan jäsen.
433  * @returns {Boolean}                Onnistuiko pelaajan liittäminen peliin vai ei.
434  * @see <a href="http://vesq.github.com/cbNetwork-node/doc/symbols/Client.html">cbNetwork.Client</a>
435  */
436 Server.prototype.login = function (client) {
437   var data = client.data
438     , version = data.getString()
439     , replyData
440     , nickname
441     , playerIds
442     , randomPlace
443     , player;
444 
445   // Täsmääkö clientin ja serverin versiot
446   if (version !== this.VERSION) {
447     log.notice('Player trying to connect with incorrect client version.');
448     replyData = new Packet(3);
449     replyData.putByte(NET.LOGIN);
450     replyData.putByte(NET.LOGINFAILED);
451     replyData.putByte(NET.WRONGVERSION);
452     replyData.putString(this.VERSION);
453     client.reply(replyData);
454     return;
455   }
456 
457   // Versio on OK, luetaan pelaajan nimi
458   nickname = data.getString().trim();
459   log.info('Player %0 is trying to login...', nickname.green);
460 
461   // Käydään kaikki nimet läpi ettei samaa nimeä vain ole jo suinkin olemassa
462   playerIds = Object.keys(this.players);
463   for (var i = playerIds.length; i--;) {
464     player = this.players[playerIds[i]];
465     if (player.name.toLowerCase() === nickname.toLowerCase()) {
466       if (player.kicked || !player.active) {
467         player.name = "";
468       } else {
469         // Nimimerkki oli jo käytössä.
470         log.info(' -> Nickname %0 already in use.', nickname.green);
471         replyData = new Packet(3);
472         replyData.putByte(NET.LOGIN);
473         replyData.putByte(NET.LOGINFAILED);
474         replyData.putByte(NET.NICKNAMEINUSE);
475         client.reply(replyData);
476         return;
477       }
478     }
479   }
480 
481   // Etsitään inaktiivinen pelaaja
482   for (i = playerIds.length; i--;) {
483     player = this.players[playerIds[i]];
484     if (this.gameState.playerCount < this.config.maxPlayers && !player.active) {
485       // Tyhjä paikka löytyi
486       player.clientId = client.id;
487       player.active = true;
488       player.loggedIn = false;
489       player.name = nickname;
490       randomPlace = this.gameState.map.findSpot();
491       player.x = randomPlace.x;
492       player.y = randomPlace.y;
493       player.hackTestX = player.x;
494       player.hackTestY = player.y;
495       player.angle = Math.floor(Math.random() * 360 + 1);
496       player.zombie = false;
497       player.health = 100;
498       player.kills = 0;
499       player.deaths = 0;
500       player.weapon = WPN.PISTOL;
501       player.lastActivity = new Date().getTime();
502       player.spawnTime = player.lastActivity;
503       player.admin = false;
504       player.kicked = false;
505       player.kickReason = "";
506       if (this.gameState.gameMode === 2) {
507         // UNIMPLEMENTED
508         // Tasainen jako joukkueihin TDM-pelimoodissa
509         player.team = Math.floor(Math.random() * 2 + 1) + 1; // Rand(1,2)
510       }
511       this.gameState.playerCount++;
512 
513       // Lähetetään vastaus clientille
514       replyData = new Packet(16);
515       replyData.putByte(NET.LOGIN);
516       replyData.putByte(NET.LOGINOK);
517       replyData.putByte(player.id);
518       replyData.putByte(this.gameState.gameMode);
519       replyData.putString(this.gameState.map.name);
520       replyData.putInt(this.gameState.map.crc32);
521       // UNIMPLEMENTED
522       replyData.putString(" "); // Kartan URL josta sen voi ladata, mikäli se puuttuu
523       client.reply(replyData);
524       log.info(' -> login successful, assigned ID (%0)', String(player.id).magenta);
525 
526       // Päivitetään tiedot servulistaukseen
527       this.registration.update();
528 
529       // Lisätään viestijonoon ilmoitus uudesta pelaajasta, kaikille muille paitsi boteille ja itselle.
530       this.messages.addToAll({
531         msgType: NET.LOGIN,
532         msgText: nickname,
533         player: player
534       }, player.id);
535       return;
536     }
537   }
538 
539   // Vapaita paikkoja ei ollut
540   log.info(' -> Server is full!');
541   replyData = new Packet(3);
542   replyData.putByte(NET.LOGIN);
543   replyData.putByte(NET.LOGINFAILED);
544   replyData.putByte(NET.TOOMANYPLAYERS);
545   client.reply(replyData);
546   return;
547 };

620 /**
621  * Sammuttaa palvelimen. Emittoi eventit {@link Server#closing} ja {@link Server#closed}
622  */
623 Server.prototype.close = function (now) {
624   if (this.gameState.closing) {
625     // Ollaan jo sulkemassa, ei aloiteta samaa prosessia uudelleen.
626     return;
627   }
628   this.gameState.closing = true;
629   log.info('Server going down...');
630   this.emit('closing');
631 
632   // Pysäytetään Game-moduulin päivitys
633   this.game.stop();
634 
635   // Pyydetään, että palvelin poistetaan listauksesta
636   if (this.registration.registered) {
637     log.info('Unregistering server...');
638     this.registration.remove(function (e) {
639       if (e) { log.error(e); }
640       else   { log.info('Server unregistered.'); }
641     });
642   }
643 
644   var self = this;
645   setTimeout(function closeServer() {
646     self.server.close();
647     self.emit('closed');
648     process.exit();
649   }, now ? 0 : 1000);
650 };

699 /**
700  * Luo uuden ammuksen.
701  *
702  * @param {Player} player  Pelaaja joka ampui ammuksen
703  */
704 Server.prototype.createBullet = function (player) {
705   var bullet, bulletAmount;
706 
707   switch (player.weapon) {
708     case WPN.LAUNCHER:
709       // Kranaatinheittimestä lähtee kaksi ammusta
710       bulletAmount = 2;
711       break;
712     case WPN.SHOTGUN:
713       // Haulikosta lähtee kuusi ammusta
714       bulletAmount = 6;
715       break;
716     default:
717       // Oletuksena ammutaan vain yksi kuti
718       bulletAmount = 1;
719   }
720 
721   for (var i = 1; i <= bulletAmount; i++) {
722     if (player.weapon === WPN.LAUNCHER && i === 2) {
723       // Toinen ammus kranaatinlaukaisimesta lähtee peilattuna
724       bullet = new Bullet(this, player, ++this.lastBulletId, true);
725     } else {
726       bullet = new Bullet(this, player, ++this.lastBulletId);
727     }
728     if (bullet.initialize()) {
729       // Jos ammuksen alustus onnistui (esim. ei ammuttu seinän sisällä), lisätään se listaan.
730       this.bullets[bullet.id] = bullet;
731       // Lisätään ammusviesti lähetettäväksi jokaiselle pelaajalle
732       this.messages.addToAll({
733         msgType: NET.NEWBULLET,          // Viestin tyyppi
734         bullet: bullet,                  // Ammus
735         sndPlay: (i === 1),              // Ääni toistetaan vain ensimmäisen luodin kohdalla
736         weapon: player.weapon,           // Millä aseella ammuttiin
737         player: player,                  // Kuka ampui
738         x: bullet.x,                     // Mistä ammus lähti
739         y: bullet.y,                     // Mistä ammus lähti
740         handShooted: player.handShooted  // Kummalla kädellä ammuttiin (pistooli)
741       });
742     } else {
743       // Ammuksen luonti epäonnistui
744       this.lastBulletId--;
745       log.debug('Failed to initialize a new bullet shot by %0', player.name.green);
746     }
747   }
748 }
