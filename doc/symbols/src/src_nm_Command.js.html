 32 /**
 33  * Help-komento tulostaa tietoja halutusta komennosta.
 34  * @methodOf Commands
 35  * @param {String} [command]  Minkä komennot salat paljastetaan
 36  */
 37 Commands.help = {
 38   params: [
 39     {name: 'command', type: 'command', optional: true, help: 'Which command\'s help to show'}
 40   ],
 41   help: 'Shows help about this server\'s commands. See ´commands´ for list of available commands.',
 42   remote: true,
 43   action: function commandsHelp() {
 44     var server = this,
 45       command = arguments[1] || 'help';
 46     if (arguments[0]) {
 47       this.commands.getHelpString(command).split('\n').forEach(function (m) {
 48         server.serverMessage(m, arguments[0]);
 49       });
 50     } else {
 51       console.log(this.commands.getHelpString(command, true));
 52     }
 53   }
 54 };

167 /**
168  * Kirjaa pelaajan sisään adminiksi.
169  * @methodOf Commands
170  * @param {Player} who    Pelaaja, joka kirjautuu
171  * @param {String} pass   Salasana
172  */
173 Commands.login = {
174   params: [
175     {name: 'pass', type: 'string',  optional: false, help: 'Password'}
176   ],
177   help: 'Logs a player in.',
178   remote: true,
179   action: function commandsLogin() {
180     var player = arguments[0]
181       , pass   = arguments[1];
182     if (!player) {
183       log.warn('Couldn\'t find player!');
184       return;
185     }
186     if (player.admin) {
187       this.serverMessage('You are already an admin!', player);
188       return;
189     }
190     if (pass === this.config.password) {
191       player.admin = true;
192       this.serverMessage('You are now an admin!', player);
193     } else {
194       log.warn('Incorrect password!');
195     }
196   }
197 };

201 (function () { // Ei vuodeta muuttujia
202   var fs = require('fs')
203   , path = require('path')
204   , files = fs.readdirSync(__dirname + '/Commands')
205   , cmds = files
206     // Filtteröi ei-js-filut.

211       var cmd = fn.toLowerCase().split('.')[0];
212       try {
213         Commands[cmd] = require(__dirname + '/Commands/' + fn);
214         return cmd.green;
215       } catch (e) {
216         log.error('Failed to load module "%0". See --debug (-d) for stack trace.', cmd.red);
217         log.debug(e.stack);
218         return cmd.red;
219       }
220     });
221   log.info('Found %0 command-module(s): %1',
222     String(cmds.length).magenta, cmds.join(', '));
223 }());

232  */

237 /**
238  * Kutstuu haluttua komentoa halutuilla parametreillä. Komentojen konteksti (this) on aina serveri.
239  * @param {String} name      Mitä komentoa kutsutaan
240  * @param {Array}  [args]    Millä parametreillä kutsutaan
241  * @param {Player} [player]  Kuka kutsui komentoa (undefined mikäli konsolista)
242  */
243 Command.prototype.call = function (name, args, player) {
244   var cmd = Commands[name], param;
245   if (!cmd) { log.error('Command "%0" not recognized. You need ´help´.', name.yellow); return; }
246 
247   // Tarkistetaan sallitaanko komento klienteillä
248   if (player && !cmd.remote) {
249     log.warn('Player %0 tried to call ´%1´, denied.', player.name.green, name);
250     this.server.serverMessage('The command you tried to call is server-side only.', player);
251     return;
252   }
253 
254   // Validoidaan argumentit - parametrien tyyppejä ei tarkasteta vaan se on tehtävä manuaalisesti.
255   for (var i = 0; i < cmd.params.length; i++) {
256     param = cmd.params[i];
257     // Jos parametri ei ole valinnainen ja argumentteja on liian vähän
258     if (!param.optional && args.length <= i) {
259       if (player) {
260         this.server.serverMessage('You must give parameter {' + param.type + '} ' +
261           param.name + '. See ´help ' + name + '´', player);
262       } else {
263         log.error('You must give parameter %0 %1. For more information see ´help %2´',
264           ('{' + param.type + '}').grey, param.name.red, name);
265       }
266       return;
267     }
268   }
269 
270   // Kutsutaan funktiota
271   args.unshift(player); // Lisätään pelaaja ensimmäiseksi parametriksi
272   try {
273     cmd.action.apply(this.server, args);
274   } catch (e) {
275     log.error('Failed to call command %0 as %1 with [%2]. See --debug (-d) for stack trace.',
276       name.yellow, (player && player.name || 'server').green, args.join(', ').green);
277     log.debug(e.stack);
278     // Also notify player if any
279     if (player) {
280       this.server.serverMessage('Oops, narrowly escaped a crash! Just whoa..', player);
281     }
282   }
283 };

285 /**
286  * Palauttaa komennon tiedot merkkijonona.
287  * @param {String}  name    Komento, jonka tiedot haluat
288  * @param {Boolean} format  Formatoidaanko? (värit & wtf-8)
289  * @return {String}  Hienosti muotoiltu merkkijono.
290  */
291 Command.prototype.getHelpString = function (name, format) {
292   /**#nocode+*/
293   var cmd = Commands[name], help
294     // Merkkijonojen täyttäminen ilmalla
295     , pad = function (s, l, r) {
296       if (r) { return new Array(Math.max(l - s.length + 1, 0)).join(' ') + s; }
297       else   { return s + new Array(Math.max(l - s.length + 1, 0)).join(' '); }
298     };
299   if (!cmd) { return 'Could not find help about "' + name + '". You need ´help´.'; }
300   if (format) {
301     help = ' Description: '.yellow + cmd.help +
302       // Jos on parametrejä
303       (cmd.params.length ? '\n Params\n   '.yellow +
304         // Jos niitä on useampi niin muotoillaan wtf-8:lla
305         (cmd.params.length > 1 ? '├→' : '└→').yellow +
306         // Listataan ne
307         join(cmd.params.map(function paramLoop(param) {
308           return pad((' {' + param.type + '} ').grey, 21, false)  // Tyyppi
309               +  pad((param.optional ? '[' + param.name + ']' : ' ' + param.name).green, 19, false) // Nimi
310               +  ' – ' + param.help; // Selitys
311         // Muotoillaan wtf-8:lla taulukosta merkkijono.
312         }), '\n   ├→'.yellow, '\n   └→'.yellow)
313       : ''); // Ei parametrejä
314   } else {
315     help = ' Description: ' + cmd.help +
316       (cmd.params.length ? '\n  Parameters:\n' : '') + // Sitten listataan parametrit mikäli niitä on.
317       (cmd.params.map(function paramLoop(param) {
318         return pad('  {' + param.type + '} ', 15, true) + // Tyyppi
319                pad((param.optional ? '[' + param.name + ']' : param.name), 18, false) + // Nimi
320                ' -- ' + param.help; // Selitys
321       }).join('\n'));
322   }
323   return help;
324   /**#nocode-*/
325 };

327 /**
328  * Palauttaa listan ehdotuksista annetulle komentorivin alulle. esim 'l' -> ['list', 'login']
329  * @param {String}   partial   Käyttäjän aloittama rivi, kun hän painaa tabia.
330  * @returns {Array}  Löydetyt  ehdotukset, jos niitä on vain yksi, sillä korvataan koko rivi.
331  */
332 Command.prototype.suggest = function (partial) {
333   /**#nocode+*/
334   var startsWith = function (str1, str2) { return str1.slice(0, str2.length) === str2; }
335     // Yhdistetään komento-osa, välissä olevat parametrit sekä ehdotus.
336     , merge = function (cmd, args, suggestion) {
337       return [cmd].concat(args.slice(0, args.length - 1), suggestion).join(' ') + ' ';
338     }
339     , suggestions = [], param, plr
340     , cmdPart = partial.split(' ')[0]             // Rivin ensimmäinen sana on tietenkin komento-osa.
341     , cmd = Commands[cmdPart]                     // Yritetään lukea komento, jos se on täydellinen.
342     , argPart = partial.split(' ').slice(1)       // Parametriosa talteen
343     , lastArg = argPart[argPart.length - 1] || '' // Viimeinen parametri, sitä täydennetään.
344     , server = this.server                        // Otetaan talteen closurea varten
345     , cmds = Object.keys(Commands)                // Otetaan talteen lista komentojen nimistä
346     , plrIds = Object.keys(this.server.players);  // ja lista pelaajien tunnisteista
347 
348   // Jos rivi alkaa komennolla ja sillä on parametrejä.
349   if (cmd && cmd.params) {
350     // Valitaan parametri, jota ollaan kirjoittamassa (tai eka, jos mitään ei ole vielä.)
351     param = cmd.params[argPart.length > 0 ? argPart.length - 1 : 0];
352     if (!param) { return; } // Parametriä ei löydy, eli kaikki parametrit on jo täytetty!
353 
354     switch (param.type) {
355     // Täydennetään pelaajien nimimerkit
356     case 'player':
357       for (var j = plrIds.length; j--;) {
358         plr = server.players[plrIds[j]];
359         if (plr.active && plr.name && startsWith(plr.name, lastArg)) {
360           suggestions.push(merge(cmdPart, argPart, plr.name));
361         }
362       }
363       break;
364     // Täydennetään komentojen nimet esim. help/commands/kick
365     case 'command':
366       cmds.map(function commandLoop(cmdName) {
367         if (startsWith(cmdName, lastArg)) {
368           suggestions.push(merge(cmdPart, argPart, cmdName));
369         }
370       });
371       break;
372     // Täydennetään alakomennot esim. config get/set/save
373     case 'sub':
374       cmd.sub.map(function commandLoop(cmdName) {
375         if (startsWith(cmdName, lastArg)) {
376           suggestions.push(merge(cmdPart, argPart, cmdName));
377         }
378       });
379     }
380   } else {
381     // Jos rivi ei ala komennolla täydennetään rivin alku komennoksi.
382     cmds.map(function commandLoop(cmdName) {
383       if (startsWith(cmdName, partial)) {
384         suggestions.push(cmdName + ' ');
385       }
386     });
387   }
388 
389   return suggestions;
390   /**#nocode-*/
391 };
