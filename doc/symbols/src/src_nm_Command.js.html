 29 /**

 31  * @param {String} [command]  Minkä komennot salat paljastetaan
 32  */
 33 Commands.help = {
 34   params: [
 35     {name: 'command', type: 'command', optional: true, help: 'Which command\'s help to show'}
 36   ],
 37   help: 'Shows help about this server\'s commands. See ´commands´ for list of available commands.',
 38   remote: true,
 39   action: function commandsHelp() {
 40     var server = this;
 41     if (!arguments[1]) {
 42       arguments[1] = 'help';
 43     }
 44     if (arguments[0]) {
 45       this.commands.getHelpString(arguments[1]).split('\n').forEach(function (m) {
 46       server.serverMessage(m, arguments[0].playerId);
 47     });
 48     } else {
 49       console.log(this.commands.getHelpString(arguments[1]));
 50     }
 51   }
 52 };

172     }

174       this.serverMessage('You are already an admin!', player.playerId);
175       return;
176     }
177     if (pass === this.config.password) {
178       player.admin = true;
179       this.serverMessage('You are now an admin!', player.playerId)
180     } else {
181       log.warn('Incorrect password!');

209 /**
210  * Kutstuu haluttua komentoa halutuilla parametreillä. Komentojen konteksti (this) on aina serveri.
211  * @param {String} name      Mitä komentoa kutsutaan
212  * @param {Array}  [args]    Millä parametreillä kutsutaan
213  * @param {Player} [player]  Kuka kutsui komentoa (undefined mikäli konsolista)
214  */
215 Command.prototype.call = function (name, args, player) {
216   var c = Commands[name];
217   if (!c) {
218     log.error('Command "%0" not recognized. You need ´help´.', name.yellow);
219     return;
220   }
221   // Tarkistetaan sallitaanko komento klienteillä
222   if (player && !c.remote) {
223     log.warn('Player %0 tried to call ´%1´, denied.', player.name.green, name);
224     return;
225   }
226 
227   // Validoidaan argumentit - parametrien tyyppejä ei tarkasteta tässä vaan se on tehtävä manuaalisesti
228   for (var i = 0; i < c.params.length; i++) {
229     var p = c.params[i];
230     // Jos parametri ei ole valinnainen ja argumentteja on liian vähän
231     if (!p.optional && args.length <= i) {
232       log.error('You must give parameter %0 %1. For more information see ´help %2´', ('{' + p.type + '}').grey, p.name.red, name);
233       return;
234     }
235   }
236   // Kutsutaan funktiota
237   args.unshift(player); // Lisätään pelaaja ensimmäiseksi parametriksi
238   c.action.apply(this.server, args);
239 };

241 /**
242  * Luo merkkijonon, joka kertoo komennon tiedot hienosti muotoiltuna.
243  * @param {String} name  Komento, jonka tiedot haluat
244  */
245 Command.prototype.getHelpString = function (name) {
246   var c = Commands[name];
247   if (!c) {
248     return 'Could not find help about "' + name + '". You need ´help´.';
249   }
250   var h =                     ' Description: ' + c.help +
251     (c.params.length ?      '\n  Parameters: ' : '');
252   // List parameters
253   for (var i = 0; i < c.params.length; i++) {
254     var p = c.params[i];
255     h += '\n' +
256       // Type
257       padString('  {' + p.type + '} ', 15, true) +
258       // Name
259       padString((p.optional ? '[' + p.name + ']' : p.name), 10, false) +
260       // Description
261       ' – ' + p.help;
262   }
263   return h;
264 };

266 /**
267  * Hoitaa komentojen ja pelaajien nimimerkkien täydentämisen annetun parametrin perusteella
268  * @param {String}  partial
269  * @returns {Array}  Löydetyt ehdotukset
270  */
271 Command.prototype.suggest = function (partial) {
272   var suggestions = [], plr
273     , server = this.server
274     , cmds = Object.keys(Commands)
275     , cmdPart = partial.split(' ')[0]
276     , args = partial.split(' ').slice(1)
277     , playerIds = Object.keys(this.server.players)
278     , c = Commands[cmdPart]; // Se saattaa alkaa jo täydellisellä komennolla
279 
280   // Jos rivi alkaa komennolla.
281   if (c) {
282     // Luupataan sen argumentit ja lisätään ehdotuksiin
283     c.params.forEach(function paramLoop(p, i) {
284       // Valitaan, mitä pitää tarkistaa parametrin tyypin perusteella
285       switch (p.type) {
286       case "player": // Pelaajien nikit on hyvä täydentää
287         for (var j = playerIds.length; j--;) {
288           plr = server.players[playerIds[j]];
289           if (plr.name && startsWith(plr.name, args[i]) && plr.active) {
290             // Pelaajan nimi alkaa argumentin alulla.
291             suggestions.push(cmdPart + ' ' + plr.name);
292           }
293         }
294         break;
295       case "command": // Samoin komentojen nimet
296         cmds.forEach(function commandLoop(s) {
297           if (startsWith(s, args[i])) {
298             // Komento alkaa argumentint alulla.
299             suggestions.push(cmdPart + ' ' + s);
300           }
301         });
302         break;
303         // TODO: Muut + säätö
304       }
305     });
306   } else { // Muussa tapauksessa täydennetään komennoksi.
307     Object.keys(Commands).map(function (i) {
308       if (startsWith(i, partial)) {
309         suggestions.push(i + ' ');
310       }
311     });
312   }
313 
314   return suggestions;
315 };

317 /** @ignore */
318 function startsWith(str1, str) {
319   return str1.slice(0, str.length) === str;
320 }

322 // Thanks for benvie at #node for help!
323 /** @ignore */
324 function padString(s, l, r) {
325   if (r) {
326     return Array(Math.max(l - s.length + 1, 0)).join(' ') + s;
327   } else {
328     return s + Array(Math.max(l - s.length + 1, 0)).join(' ')
329   }
