 51 server.on(NET.PLAYER, function NetPlayer(client, player) {
 52   // Pelaajan dataa
 53   var data = client.data
 54     , x     = data.getShort()   // x-position
 55     , y     = data.getShort()   // y-position
 56     , angle = data.getShort()   // kulma
 57     , b     = data.getByte();   // Tämä tavu sisältää useamman muuttujan (alempana)
 58 
 59   // Puretaan b-tavu muuttujiin
 60   // Jos halutaan esim. lukea 4 bittiä kohdasta 0, menee lauseke seuraavasti:
 61   // var value = (b << (32-0-4)) >> (32-4)
 62   var weapon   =   (b << 28) >> 28    // Valittuna oleva ase (bitit 0-3)
 63     , hasAmmo  = -((b << 27) >> 31)   // Onko valitussa aseessa ammuksia (bitti 4)
 64     , shooting = -((b << 26) >> 31);  // Ampuuko (bitti 5)
 65 
 66   var picked  = data.getByte(); // Poimitun itemin id (0, jos ei poimittu)
 67 
 68   // Arvot päivitetään vain jos pelaaja on hengissä
 69   if (!player.isDead) {
 70     player.x        = x;
 71     player.y        = y;
 72     player.angle    = angle;
 73     player.weapon   = weapon;
 74     player.hasAmmos = hasAmmo;
 75 
 76     // UNIMPLEMENTED
 77     // speedhack
 78 
 79     if (shooting === 1) {
 80       server.createBullet(player);
 81     }
 82 
 83     // Poimittiinko jotain
 84     if (picked > 0 && server.items.hasOwnProperty(picked)) {
 85       var itemType = server.items[picked].pick();
 86       // Poimittiinko healthpack
 87       if (itemType === ITM.HEALTH) {
 88         player.health = Math.min(100, player.health + 50);
 89       }
 90     }
 91   }
 92   if (player.health > 0) {
 93     player.isDead = false;
 94   }
 95   if (!player.loggedIn) {
 96     player.loggedIn = true;
 97   }
 98 });

105 server.on(NET.TEXTMESSAGE, function NetTextMessage(client, player) {
106   // Pelaaja lähetti tsättiviestin
107   var txtMessage = client.data.getString().trim(), cmd;
108   if (txtMessage.charAt(0) === '/') {
109     cmd = txtMessage.split(' ')[0].slice(1);
110     // Komennot vaativat admin-oikeudet
111     if (player.admin) {
112       log.notice('Player %0 called ´%1´', player.name.green, txtMessage);
113       server.commands.call(cmd, split(txtMessage).splice(1), player);
114     // Sallitaan kirjautumisen yrittäminen kuitenkin. :P
115     } else if (cmd === 'login') {
116       cmd = txtMessage.split(' ').splice(1);
117       log.notice('Player %0 is trying to login with password ´%1´.', player.name.green, String(cmd[0]).red);
118       server.commands.call('login', cmd, player);
119     } else {
120       log.warn('Player %0 tried to call ´%1´ without admin rights, access denied.', player.name.green, cmd);
121       server.serverMessage('You need to login as admin to use commands.', player);
122     }
123   } else {
124     // Ei ollut komento, logataanpas tämä.
125     log.write('<' + player.name + '> ' + txtMessage);
126 
127     var msgData = {
128       msgType: NET.TEXTMESSAGE,
129       player: player,
130       msgText: txtMessage
131     };
132     // Lähetetään kaikille muille paitsi boteille
133     if (txtMessage.charAt(0) === '*') {
134       // Lähetetään vain omalle joukkueelle tämä viesti
135       server.messages.addToTeam(player.team, msgData);
136     } else {
137       // Ei ollut tähteä ekana kirjaimena, joten tämä viesit on julkinen ja lähtee kaikille.
138       server.messages.addToAll(msgData);
139     }
140   }
141 });
