  5 // Vaatii cbNetwork ja node-optimist-paketin: https://github.com/substack/node-optimist
  6 /**#nocode+*/
  7 var argv = require('optimist')
  8     .default({p : 1337, a : undefined})
  9     .alias({'p' : 'port', 'a' : 'address', 'd' : 'debug'})
 10     .argv
 11   , Packet = require('cbNetwork').Packet
 12   , NET = require('./Constants').NET
 13   , ITM = require('./Constants').ITM
 14   , Server = require('./Server')
 15   , log = require('./Utils').log
 16   , colors = require('colors')
 17   , timer = require('./Utils').timer
 18   , Bullet = require('./Weapon').Bullet;
 19 /**#nocode-*/
 20 
 21 process.title = "NetMatch server";
 22 
 23 // Tehdään uusi palvelin.
 24 /** @ignore */
 25 var server = new Server({ port: argv.p, address: argv.a});
 26 
 27 // Tehdään jotain sitä mukaa, kun dataa tulloo sissään.
 28 server.on('message', function onServerMessage(client) {
 29   var data = client.data
 30     , msgType = data.getByte()
 31     , currentPlayerId
 32     , player
 33     , reply
 34     , sendNames = false
 35     , txtMessage = ""
 36     , playerIds;
 37 
 38   // Jos palvelin on sulkeutumassa, lähetetään pelaajalle siitä tieto heti
 39   if (this.gameState.closing) {
 40     reply = new Packet(2);
 41     reply.putByte(NET.SERVERCLOSING);
 42     reply.putByte(NET.END);
 43     client.reply(reply);
 44     return;
 45   }
 46 
 47   // Jos oli uusi pelaaja
 48   if (msgType === NET.LOGIN) {
 49     server.login(client);
 50     return;
 51   }
 52 
 53   // Luetaan lähetetty pelaajan ID, joka on pelaajan järjestysnumero ja aina väliltä 1...MAX_PLAYERS
 54   currentPlayerId = data.getByte();
 55 
 56   if (currentPlayerId < 1 || currentPlayerId > server.gameState.maxPlayers) {
 57     // Pelaajatunnus on väärä
 58     log.notice('Possible hack attempt from ' + client.address + ' Invalid player ID (' + currentPlayerId + ')');
 59     return;
 60   }
 61 
 62   // Haetaan pelaajan instanssi Player-luokasta
 63   player = this.players[currentPlayerId];
 64 
 65 
 66   // Tarkistetaan onko pelaaja potkittu
 67   if (player.kicked && player.clientId === client.id) {
 68     reply = new Packet(7);
 69     reply.putByte(NET.KICKED);
 70     reply.putByte(player.kickerId);
 71     reply.putByte(currentPlayerId);
 72     reply.putString(player.kickReason);
 73     client.reply(reply);
 74     return;
 75   }
 76 
 77   if (player.clientId !== client.id || !player.active) {
 78     //log.notice("Possible hack attempt from " + client.address + " Not logged in (" + currentPlayerId + ")");
 79     reply = new Packet(1);
 80     reply.putByte(NET.NOLOGIN);
 81     client.reply(reply);
 82     return;
 83   }
 84 
 85   // Pelaaja poistuu pelistä
 86   if (msgType === NET.LOGOUT) {
 87     server.logout(currentPlayerId);
 88     return;
 89   }
 90 
 91   // Lasketaan pelaajan ja serverin välinen lagi
 92   player.lag = timer() - player.lastActivity;
 93 
 94   // Päivitetään pelaajan olemassaolo
 95   player.lastActivity = timer();
 96 
 97   // Luetaan saapuneita viestejä
 98   while (msgType) {
 99     if (msgType === NET.PLAYER) {
100       // Pelaajan dataa
101       var x     = data.getShort()   // x-position
102         , y     = data.getShort()   // y-position
103         , angle = data.getShort()   // kulma
104         , b     = data.getByte();   // Tämä tavu sisältää useamman muuttujan (alempana)
105 
106         // Puretaan b-tavu muuttujiin
107       // Jos halutaan esim. lukea 4 bittiä kohdasta 0, menee lauseke seuraavasti:
108       // var value = (b << (32-0-4)) >> (32-4)
109       var weapon   = (b << 28) >> 28      // Valittuna oleva ase (bitit 0-3)
110         , hasAmmo  = -((b << 27) >> 31)   // Onko valitussa aseessa ammuksia (bitti 4)
111         , shooting = -((b << 26) >> 31);  // Ampuuko (bitti 5)
112 
113       var picked  = data.getByte(); // Poimitun itemin id (0, jos ei poimittu)
114 
115       // Arvot päivitetään vain jos pelaaja on hengissä
116       if (!player.isDead) {
117         player.x        = x;
118         player.y        = y;
119         player.angle    = angle;
120         player.weapon   = weapon;
121         player.hasAmmos = hasAmmo;
122 
123         // UNIMPLEMENTED
124         // speedhack
125 
126         if (shooting === 1) {
127           if (argv.d) {
128             log.info('Player ' + player.name.yellow + ' is shooting a new bullet from weapon ' + player.weapon);
129           }
130           new Bullet(server, currentPlayerId);
131           //console.dir(this.bullets);
132         }
133 
134         // Poimittiinko jotain
135         if (picked > 0 && server.items.hasOwnProperty(picked)) {
136           var itemType = server.items[picked].pick();
137           // Poimittiinko healthpack
138           if (itemType === ITM.HEALTH) {
139             player.health = Math.min(100, player.health + 50);
140           }
141         }
142       }
143       if (player.health > 0) {
144         player.isDead = false;
145       }
146       if (!player.loggedIn) {
147         player.loggedIn = true;
148       }
149 
150     } else if (msgType === NET.PLAYERNAME) {
151       // Nimilista pyydetty
152       sendNames = true;
153 
154     } else if (msgType === NET.TEXTMESSAGE) {
155       // Pelaaja lähetti tsättiviestin
156       txtMessage = data.getString().trim();
157       if (txtMessage.charAt(0) === '/') {
158         // UNIMPLEMENTED
159         // Komento
160         log.info('Player ' + player.name + ' sent a command: ' + txtMessage);
161         txtMessage = "";
162       } else {
163         // Ei ollut komento, logataanpas tämä.
164         log.write('<' + player.name + '> ' + txtMessage);
165       }
166 
167     } else if (msgType === NET.MAPCHANGE) {
168       // Pelaaja lähetti kartan nimen
169       player.mapName = data.getString().trim();
170 
171     } else {
172       // Viestit loppui tai tuli tuntematon viesti
173       break;
174     }
175     // Seuraava viesti
176     msgType = data.getByte();
177   }
178 
179   // UNIMPLEMENTED
180   // Jos erä on päättynyt niin lähetetään kaikkien pelaajien tiedot
181   // sendNames = true;
182 
183   // Lähetetään clientille dataa
184   reply = new Packet();
185 
186   // Lähetetään kaikkien pelaajien tiedot
187   playerIds = Object.keys(server.players);
188   for (var i = playerIds.length; i--;) {
189     var plr = server.players[playerIds[i]];
190     // Onko pyydetty nimet
191     if (sendNames) {
192       if (plr.active) {
193         reply.putByte(NET.PLAYERNAME);  // Nimet
194         reply.putByte(plr.playerId);    // Pelaajan tunnus
195         reply.putString(plr.name);      // Nimi
196         reply.putByte(plr.zombie);      // Onko botti
197         reply.putByte(plr.team);        // Joukkue
198       }
199     }
200 
201     // Onko lähetetty tsättiviesti
202     if (txtMessage.length > 0) {
203       var msgData = {
204         msgType: NET.TEXTMESSAGE,
205         playerId: currentPlayerId,
206         msgText: txtMessage
207       };
208       // Lähetetään kaikille muille paitsi boteille
209       if (plr.active && !plr.zombie) {
210         if (txtMessage.charAt(0) === '*') {
211           // Lähetetään vain omalle joukkueelle tämä viesti
212           if (plr.team === player.team) {
213             server.messages.add(plr.playerId, msgData);
214           }
215         } else {
216           // Ei ollut tähteä ekana kirjaimena, joten tämä viesit on julkinen ja lähtee kaikille.
217           server.messages.add(plr.playerId, msgData);
218         }
219       }
220     }
221 
222     // Lähetetään niiden pelaajien tiedot jotka ovat hengissä ja näkyvissä
223     if (plr.active) {
224       var visible = true
225         , x1 = player.x
226         , y1 = player.y
227         , x2 = plr.x
228         , y2 = plr.y;
229       if ((Math.abs(x1 - x2) > 450) || (Math.abs(y1 - y2) > 350)) {
230         visible = false;
231       }
232 
233       // Onko näkyvissä vai voidaanko muuten lähettää
234       if (sendNames || visible || plr.health <= 0) {
235         // Näkyy
236         reply.putByte(NET.PLAYER);    // Pelaajan tietoja
237         reply.putByte(plr.playerId);  // Pelaajan tunnus
238         reply.putShort(plr.x);        // Sijainti
239         reply.putShort(plr.y);        // Sijainti
240         reply.putShort(plr.angle);    // Kulma
241 
242         // Spawn-protect
243         var isProtected = 0;
244         if (plr.spawnTime + server.gameState.spawnProtection > timer()) {
245           isProtected = 1;
246         }
247 
248         // Muutetaan team arvo välille 0-1
249         var teamBit = (plr.team === 2 ? 1 : 0);
250 
251         // Tungetaan yhteen tavuun useampi muuttuja
252         var b = ((plr.weapon % 16) << 0)  // Ase (bitit 0-3)
253               + ((plr.hasAmmos << 4))     // Onko ammuksia (bitti 4)
254               + ((teamBit << 6))          // Joukkue/tiimi (bitti 6)
255               + ((isProtected << 7));     // Haavoittumaton (bitti 7)
256         reply.putByte(b);
257 
258         reply.putByte(plr.health);      // Terveys
259         reply.putShort(plr.kills);      // Tapot
260         reply.putShort(plr.deaths);     // Kuolemat
261       } else if (server.gameState.radarArrows || server.gameState.playMode === 2) {
262         // Ei näy. Lähetetään tutkatieto. playMode === 2 tarkoittaa TDM-pelimuotoa
263         if (player.team === plr.team || server.gameState.radarArrows) {
264           // Lähetetään tutkatiedot jos joukkueet ovat samat tai asetuksista on laitettu että
265           // kaikkien joukkueiden pelaajien tutkatiedot lähetetään
266           reply.putByte(NET.RADAR); // Tutkatietoa tulossa
267           // UNIMPLEMENTED
268           // Missä kulmassa tutkan pitäisi olla
269           reply.putByte(0);         // Kulma muutettuna välille 0-255
270           reply.putByte(plr.team);  // Pelaajan joukkue
271         }
272       }
273     }
274   }
275 
276   // UNIMPLEMENTED
277   // Kartan vaihtaminen
278 
279   // Lähetetään kaikki pelaajalle osoitetut viestit
280   server.messages.fetch(currentPlayerId, reply);
281 
282 
283   // Jos on pyydetty nimilista niin palautetaan myös kaikkien tavaroiden tiedot
284   if (sendNames) {
285     var itemIds = Object.keys(this.items);
286     for (var i = itemIds.length; i--;) {
287       var item = this.items[itemIds[i]];
288       this.messages.add(currentPlayerId, {
289         msgType: NET.ITEM,
290         itemId: item.id,
291         itemType: item.type,
292         x: item.x,
293         y: item.y
294       });
295     }
296   }
297 
298   // UNIMPLEMENTED
299   // Pelisession aikatiedot
300 
301   reply.putByte(NET.END);
302   client.reply(reply);
303 
304   // Dodiin, valmiita ollaan :)
305   return;
306 });
