 14 /**
 15  * Alustaa uuden pelimekaniikasta huolehtivan Game-luokan instanssin.
 16  * @class Pelimekaniikan päivittämiseen liittyvät toiminnot
 17  *
 18  * @param {Server} server  Nykyisen {@link Server}-luokan instanssi
 19  */
 20 function Game(server) {
 21   this.server = server;
 22   this.lastUpdate = 0;
 23   this.frameTime = 0;
 24   this.isNextMapLoaded = false;
 25 }
 26 
 27 /**
 28  * Käynnistää palvelimen
 29  *
 30  * @param {Number} updatesPerSecond  Kuinka monta kertaa sekunnissa palvelinta päivitetään
 31  */
 32 Game.prototype.start = function (updatesPerSecond) {
 33   this.server.gameState.sessionStarted = Date.now();
 34   this.interval = setInterval(this.update, 1000 / updatesPerSecond, this);
 35 };
 36 
 37 /**
 38  * Pysäyttää pelimekaniikan päivityksen.
 39  */
 40 Game.prototype.stop = function () {
 41   clearInterval(this.interval);
 42 };
 43 
 44 /**
 45  * Päivittää pelimekaniikan, mikäli viime päivityksestä on kulunut tarpeeksi aikaa ja
 46  * pelaajia on palvelimella. Tämän funktion kutsu hoidetaan automaattisesti.
 47  * @private
 48  */
 49 Game.prototype.update = function (self) {
 50   self.updateFrameTimer();
 51   self.updateBotsAI();
 52   self.updateRoundTime();
 53   self.updateStats();
 54   self.updatePlayers();
 55   self.updateTimeouts();
 56   self.updateBotsAmount();
 57   self.updateBullets();
 58 
 59   self.lastUpdate = Date.now();
 60 };
 61 
 62 /**
 63  * Päivittää vakionopeusajastimen
 64  * @private
 65  */
 66 Game.prototype.updateFrameTimer = function () {
 67   var curTime = Date.now();
 68   if (!this.lastUpdate) {
 69     this.lastUpdate = curTime;
 70   }
 71   this.frameTime = (curTime - this.lastUpdate) / 1000;
 72   this.lastUpdate = curTime;
 73 };

 75 /**
 76  * Käy botit läpi ja kutsuu jokaisen päivitysrutiinia.
 77  * @private
 78  */
 79 Game.prototype.updateBotsAI = function () {
 80   var self = this;
 81   // Tarkistetaan onko pelaajia pelissä. Jos ei, niin ei päivitetä botteja.
 82   if (this.server.gameState.playerCount <= 0) {
 83     return;
 84   }
 85 
 86   // Onko erä loppu
 87   if (this.server.gameState.sessionComplete) {
 88     return;
 89   }
 90 
 91   // Pyyhitään jokaiselta pelaajalta debuggaukset, jos debugataan.
 92   if (this.server.debug) {
 93     this.server.loopPlayers (function (player) {
 94       if (!player.zombie && player.active && player.loggedIn && !player.debugState) {
 95         self.server.messages.add(player.id, {
 96           msgType: NET.DEBUGDRAWING,
 97           drawType: DRAW.CLEAR
 98         });
 99         player.debugState = 1;
100       }
101     });
102   }
103 
104   // Päivitetään bottien tekoälyt
105   this.server.loopPlayers (function (player) {
106     if (player.zombie && player.active && !player.isDead) {
107       player.botAI.update();
108     }
109   });
110 };

112 /**
113  * Tarkistaa onko erä päättynyt ja hoitaa vastaavat päivitykset, jos on
114  * @private
115  */
116 Game.prototype.updateRoundTime = function () {
117   var timeLeft, server = this.server;
118 
119   // Jos asetuksissa on erän pituus pienempi tai yhtä suuri kuin 0, niin kello ei ole käytössä.
120   if (server.config.periodLength <= 0) {
121     return;
122   }
123 
124   // Jos pelaajia ei ole palvelimella niin kello ei käy.
125   if (server.gameState.playerCount <= 0) {
126     server.gameState.sessionStarted = Date.now();
127     server.gameState.sessionComplete = false;
128     this.isNextMapLoaded = false;
129     return;
130   }
131 
132   // Lasketaan jäljellä oleva aika
133   timeLeft = server.gameState.sessionStarted + server.config.periodLength * 1000 - Date.now();
134 
135   if (timeLeft <= 0) {
136     server.gameState.sessionComplete = true;
137 
138     // Poistetaan kaikki ammukset
139     server.bullets = {};
140 
141     // Kierrätetään seuraava kartta peliin, jos karttoja on listassa yli yksi, eikä karttaa
142     // ole vielä vaihdettu.
143     if (!this.isNextMapLoaded && timeLeft < -5000 && server.config.map.length > 1) {
144       server.changeMap();
145       this.isNextMapLoaded = true;
146     }
147 
148     // Tapetaan jokainen pelaaja
149     server.loopPlayers(function roundEndKill(player) {
150       player.health = -10;
151       player.timeToDeath = Date.now();
152     });
153   }
154 
155   // Onko erä päättynyt 10 sekuntia sitten
156   if (timeLeft < -10000) {
157     server.gameState.sessionStarted = Date.now();
158     server.gameState.sessionComplete = false;
159     this.isNextMapLoaded = false;
160     server.loopPlayers(function mapStartLoop(player) {
161       player.timeToDeath = Date.now() - server.config.deathDelay * 2;
162       player.kills = 0;
163       player.deaths = 0;
164 
165       // Tarkistetaan onko pelaajalla jo ladattuna sama kartta kuin palvelimella
166       if (player.active && player.loggedIn && !player.zombie && server.config.map.length > 1 && player.mapName !== server.gameState.map.name) {
167         // Kartta oli eri.
168         if (player.mapName) {
169           log.notice('Player %0 had a map %1 while the server was running map %2',
170             player.name.green, player.mapName.green, server.gameState.map.name.green);
171         } else {
172           log.notice('Player %0 did not have a map set!');
173         }
174         server.logout(player);
175       }
176     });
177   }
178 };

188 /**
189  * Päivittää pelaajat. Hoitaa kuolleista herätykset ja sen etteivät pelaajat ole kartan sisällä.
190  * @private
191  */
192 Game.prototype.updatePlayers = function () {
193   var playerIds = Object.keys(this.server.players);
194   for (var i = playerIds.length; i--;) {
195     var player = this.server.players[playerIds[i]];
196 
197     // Jos pelaaja on kuollut ja kuolemasta on kulunut tarpeeksi aikaa, herätetään henkiin.
198     if (player.health <= 0 && player.timeToDeath + this.server.config.deathDelay < Date.now()) {
199       if (this.server.debug) {
200         log.write('Reviving %0 from the deads.', player.name.green);
201       }
202       var randomPlace = this.server.gameState.map.findSpot();
203       player.x = randomPlace.x;
204       player.y = randomPlace.y;
205       player.health = 100;
206       player.lastValidX = player.x;
207       player.lastValidY = player.y;
208       player.hackTestX = player.x;
209       player.hackTestY = player.y;
210       player.spawnTime = Date.now();
211       if (player.zombie) {
212         player.isDead = false;
213         player.weapon = this.server.getBotWeapon();
214         if (this.server.gameState.gameMode === 3) {
215           // Zombie-moodi, boteilla on vain 10hp eivätkä ne ole koskaan nakkeja
216           player.health = 10;
217           player.spawnTime = 0;
218         }
219       }
220     }
221 
222     // UNIMPLEMENTED
223     // Onko pelajaa kartalla
224   }
225 };

227 /**
228  * Poistaa pelaajat, joista ei ole hetkeen kuulunut mitään.
229  * @private
230  */
231 Game.prototype.updateTimeouts = function () {
232   var server = this.server;
233 
234   server.loopPlayers(function (player) {
235     if ((!player.active && !player.loggedIn) || player.zombie) {
236       // Pelaaja ei ole aktiivinen eikä sisäänkirjautunut taikka pelaaja on botti, joten
237       // ei tarkisteta tältä timeouttia.
238       return;
239     }
240     if (player.lastActivity + server.config.maxInactiveTime < Date.now()) {
241       // Timeout tuli, poistetaan pelaaja.
242       player.active = false;
243       player.loggedIn = false;
244       player.admin = false;
245       log.info('%0 timed out.', player.name.green);
246 
247       server.gameState.playerCount--;
248 
249       // Päivitetään tiedot servulistaukseen
250       server.registration.update();
251 
252       // Kerrotaan siitä muillekin
253       server.messages.addToAll({ msgType: NET.LOGOUT, player: player });
254     }
255   });
256 };

258 /**
259  * Pitää bottien lukumäärän oikeana
260  * @private
261  */
262 Game.prototype.updateBotsAmount = function () {
263   var server = this.server
264     , gs = server.gameState
265     , loopedBotsCount = 0;
266 
267   // Pidetään huolta ettei botDepartCount eikä botCount ylitä maxPlayersin arvoa
268   if (gs.botCount > gs.maxPlayers) {
269     gs.botCount = gs.maxPlayers;
270   }
271   if (gs.botDepartLimit > gs.maxPlayers) {
272     gs.botDepartLimit = gs.maxPlayers;
273   }
274 
275   // Pidetään huolta että botCount menee myös botDepartLimitin mukaan
276   if (gs.botDepartLimit < gs.botCount + gs.playerCount) {
277     gs.botCount = gs.botDepartLimit - gs.playerCount;
278   } else if (gs.botDepartLimit > gs.botCount + gs.playerCount) {
279     // Tarkistetaan että onko botDepartLimitin takia joskus poistettu botteja mutta ei lisätty
280     // takaisin sallittua määrää
281     if (server.config.botCount < 0 && gs.map.config && gs.map.config.botCount > gs.botCount) {
282       // Mennään kartan configin arvojen mukaan
283       gs.botCount = gs.map.config.botCount;
284     } else {
285       // Muulloin mennään normaalin configin arvojen mukaan
286       if (server.config.botCount > gs.botCount) {
287         gs.botCount = server.config.botCount;
288       }
289     }
290   }
291 
292   server.loopPlayers(function (player) {
293     // Tarkistetaan vain aktiiviset botit
294     if (!player.zombie || !player.active) {
295       return;
296     }
297 
298     // Tarkistetaan ollaanko ylitetty tämän botin kohdalla bottiraja
299     if (loopedBotsCount >= gs.botCount) {
300       // Yli ollaan menty. Kirjataan botti ulos.
301       server.logout(player);
302     }
303 
304     loopedBotsCount++;
305   });
306 
307   // Pidetään huolta ettei botteja ole liian vähän
308   if (loopedBotsCount < gs.botCount) {
309     for (var i = loopedBotsCount; i < gs.botCount; i++) {
310       server.addBot();
311     }
312   }
313 };

315 /**
316  * Hoitaa ammusten siirtelyn, osumisen ja poistamisen.
317  * @private
318  */
319 Game.prototype.updateBullets = function () {
320   var bulletIds = Object.keys(this.server.bullets)
321     , bullet;
322 
323   // Käydään kaikki ammukset läpi
324   for (var i = bulletIds.length; i--;) {
325     bullet = this.server.bullets[bulletIds[i]];
326     bullet.update();
327   }
328 };

332  * @param {Number} amount  Pikselimäärä tai astemäärä joka siirrytään/käännytään yhden sekunnin aikana
333  * @returns {Number}
