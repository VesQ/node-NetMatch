 11 /**
 12  * Alustaa uuden pelimekaniikasta huolehtivan Game-luokan instanssin.
 13  * @class Pelimekaniikan päivittämiseen liittyvät toiminnot
 14  *
 15  * @param {Server} server  Nykyisen {@link Server}-luokan instanssi
 16  */
 17 function Game(server) {
 18   this.server = server;
 19   this.lastUpdate = 0;
 20   this.frameTime = 0;
 21   this.interval = setInterval(this.update, 1000 / server.gameState.updatesPerSecond, this);
 22 }
 23 
 24 /**
 25  * Pysäyttää pelimekaniikan päivityksen.
 26  */
 27 Game.prototype.stop = function () {
 28   clearInterval(this.interval);
 29 }

 31 /**
 32  * Päivittää pelimekaniikan, mikäli viime päivityksestä on kulunut tarpeeksi aikaa ja
 33  * pelaajia on palvelimella. Tämän funktion kutsu hoidetaan automaattisesti.
 34  * @private
 35  */
 36 Game.prototype.update = function (self) {
 37   self.updateFrameTimer();

 42   self.updateTimeouts();
 43   self.updateBotsAmount();
 44   self.updateRegistration();
 45   self.updateBullets();
 46 
 47   self.lastUpdate = timer();
 48 }

 50 /**
 51  * Päivittää vakionopeusajastimen
 52  * @private
 53  */
 54 Game.prototype.updateFrameTimer = function () {
 55   var curTime = timer();
 56   if (!this.lastUpdate) {
 57     this.lastUpdate = curTime;
 58   }
 59   this.frameTime = (curTime - this.lastUpdate) / 1000;
 60   this.lastUpdate = curTime;
 61 }

 87 /**
 88  * Päivittää pelaajat. Hoitaa kuolleista herätykset ja sen etteivät pelaajat ole kartan sisällä.
 89  * @private
 90  */
 91 Game.prototype.updatePlayers = function () {
 92   var playerIds = Object.keys(this.server.players);
 93   for (var i = playerIds.length; i--;) {
 94     var player = this.server.players[playerIds[i]];
 95 
 96     // Jos pelaaja on kuollut ja kuolemasta on kulunut tarpeeksi aikaa, herätetään henkiin.
 97     if (player.health <= 0 && player.timeToDeath + this.server.config.deathDelay < Date.now()) {
 98       if (this.server.debug) {
 99         log.write('Reviving %0 from the deads.', player.name.green);
100       }
101       var randomPlace = this.server.gameState.map.findSpot();
102       player.x = randomPlace.x;
103       player.y = randomPlace.y;
104       player.health = 100;
105       player.lastValidX = player.x;
106       player.lastValidY = player.y;
107       player.hackTestX = player.x;
108       player.hackTestY = player.y;
109       player.spawnTime = Date.now();
110       // UNIMPLEMENTED
111       // Jos tämä on botti niin arvotaan ase
112     }
113 
114     // UNIMPLEMENTED
115     // Onko pelajaa kartalla
116   }
117 }

158 /**
159  * Palauttaa siirtymän tai kääntymän (pikseliä tai astetta sekunnissa)
