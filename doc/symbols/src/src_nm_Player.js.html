 13 /**
 14  * Luo uuden pelaajan.
 15  *
 16  * @class Kaikki pelaajat, niin ihmispelaajat kuin botitkin, ovat tämän luokan jäseniä.
 17  *
 18  * @param {Server} server  NetMatch-palvelin, joka pyörittää pelaajaa
 19  * @param {Byte} playerId  Luotavan pelaajan ID
 20  *
 21  * @property {Integer} playerId      Pelaajan tunnus välillä 1-MAX_PLAYERS
 22  * @property {String}  clientId      ClientId eli ip-osoite ja tunnus
 23  * @property {Integer} active        Onko tämä pelaaja pelissä
 24  * @property {Integer} loggedIn      Pelaaja on kirjautunut
 25  * @property {String}  name          Pelaajan nimimerkki
 26  * @property {String}  statName      Pelaajan nimimerkki
 27  * @property {String}  botName       Botin nimi
 28  * @property {Integer} lastActivity  Koska viimeksi pelaajalta on saatu dataa
 29  * @property {Integer} x             Sijainti
 30  * @property {Integer} y             Sijainti
 31  * @property {Integer} angle         Kulma
 32  * @property {Integer} passCount     Kunka monta kierrosta pelaaja on ollut päivittämättä
 33  * @property {Integer} obj           Pelihahmo
 34  * @property {Integer} leftLegObj    Vasen jalka
 35  * @property {Integer} rightLegObj   Oikea jalka
 36  * @property {Integer} fireObj       Suuliekkianimaatio
 37  * @property {Integer} fireObj2      Suuliekkianimaatio (toinen pistooli)
 38  * @property {Integer} showFire      Näytetäänkö suuliekki
 39  * @property {Float}   prevPosX      Pelaajan edellinen sijainti
 40  * @property {Float}   prevPosY      Pelaajan edellinen sijainti
 41  * @property {Float}   legPos        Jalkojen asento kävelyanimaatiossa
 42  * @property {Integer} lastValidX    Viimeisin paikka jossa pelaaja on varmasti ollut kentällä
 43  * @property {Integer} lastValidY    Viimeisin paikka jossa pelaaja on varmasti ollut kentällä
 44  * @property {Integer} weapon        Käytössä oleva ase
 45  * @property {Integer} lastShoot     Koska on viimeksi ammuttu
 46  * @property {Integer} lastSound     Koska on viimeksi soitettu ampumisääni (käytetään moottorisahassa)
 47  * @property {Integer} health        Terveys
 48  * @property {Integer} prevHealth    Terveys edellisellä kierroksella
 49  * @property {Integer} kills         Tappojen lukumäärä
 50  * @property {Integer} deaths        Kuolemien lukumäärä
 51  * @property {Float}   killRatio     Tapposuhde
 52  * @property {Boolean} isDead        Onko pelaaja kuollut
 53  * @property {Integer} timeToDeath   Kuolinaika
 54  * @property {Integer} zombie        Onko tämä pelaaja botti
 55  * @property {Float}   nextAngle     Kulma johon botin pitäisi liikkua
 56  * @property {Float}   lastAngle     Kuinka paljon botin ja kohdekulman ero oli edellisellä kierroksella
 57  * @property {Float}   rotation      Tällä käännetään objektia (TurnObject)
 58  * @property {Float}   sideStep      Botin sivuaskeleet
 59  * @property {integer} nextAction    Ajankohta jolloin botille arvotaan uusi suunta
 60  * @property {Integer} tooClose      Botti on liian lähellä seinää tai toista pelaajaa
 61  * @property {Float}   fightRotate   Kääntymisnopeus kun ollaan havaittu vastustaja
 62  * @property {Float}   shootingAngle Ammutaan kun uhri on tämän kulman sisällä botin edessä
 63  * @property {Integer} shootedBy     Kuka on viimeksi ampunut pelaajaa
 64  * @property {Float}   fov           Field of View
 65  * @property {Integer} idleSound     Esim. moottorisahan tyhjäkäynti
 66  * @property {Integer} idleTimer     Koska soitettu viimeksi
 67  * @property {Integer} hasAmmos      Onko pelaajalla ammuksia nykyisessä aseessa
 68  * @property {Byte}    team          Joukkue
 69  * @property {String}  mapName       Pelaajalla ladattuna oleva kartta
 70  * @property {Integer} outOfMap      Kuinka monta päivityskierrosta pelaaja on ollut poissa kartalta
 71  * @property {Integer} spawnTime     Syntymäaika
 72  * @property {Integer} visible       Ukon vilkuttaminen syntymän jälkeen
 73  * @property {Byte}    admin         Onko pelaaja admin vai ei
 74  * @property {Byte}    kicked        Onko pelaaja potkittu pois
 75  * @property {String}  kickReason    Mikä syy on annettu potkuille
 76  * @property {Byte}    kickerId      Kuka potkaisi
 77  * @property {Integer} hackTestX     Pelaajan viimeisin sijainti nopeuden tarkistuksessa
 78  * @property {Integer} hackTestY     Pelaajan viimeisin sijainti nopeuden tarkistuksessa
 79  * @property {Integer} lag           Pelaajan lagi millisekunneissa
 80  * @property {Integer} spHackTimer   Viimeisimmässä nopeuden tarkistuksessa otettu Timer()
 81  * @property {Byte}    handShooted   Kummalla kädellä on viimeksi ammuttu (pistooli) 0=vasen 1=oikea
 82  * @property {Boolean} sendNames     Pelaaja pyysi palvelimelta pelaajien nimilistauksen, lähetään kivasti myös itemien tiedot ":D"
 83  */
 84 function Player(server, playerId) {
 85   Obj.call(this, 0, 0, 0);
 86   this.server = server;
 87 
 88   // Alustetaan pelaaja
 89   this.playerId = playerId;
 90   this.team = 1;
 91   this.botName = server.gameState.map.config.botNames[playerId];
 92   this.clientId = "";
 93   this.name = "";
 94   var skill = 21 - playerId; // Botteja hieman eritasoisiksi vissiinkin?
 95   this.fightRotate = 1.5 + (skill / 1.5);
 96   this.shootingAngle = 4.0 + (playerId * 1.5);
 97   this.fov = 100 + (skill * 3.5);
 98   this.kickReason = "";
 99 
100   // Lisätään tämä palvelimen players-kokoelmaan ja poistetaan vanha, jos sellainen oli olemassa.
101   if (server.players.hasOwnProperty(playerId)) {
102     delete server.players[playerId];
103   }
104   server.players[playerId] = this;
105 }
106 Player.prototype = new Obj();
107 Player.prototype.constructor = Player;

109 /**
110  * Tappaa pelaajan.
111  * @param {Bullet} [bullet]  Tappajapelaajan ammus, joka hoiti viimeistelyn. Jos tätä ei anneta,
112  *                           uskotellaan että pelaaja teki itsemurhan pistoolilla.
113  */
114 Player.prototype.kill = function (bullet) {
115   var killer, weapon;
116 
117   if (this.server.debug) {
118     if (!bullet) {
119       log.write('Player %0 committed suicide.', this.name);
120     } else {
121       log.write('Player %0 was killed by player %1 with bullet %2 (%3)', this.name.green,
122         this.server.players[bullet.playerId].name.green, String(bullet.bulletId).magenta,
123         Weapons[bullet.weapon].name.yellow);
124     }
125   }
126 
127   this.isDead = true;
128   this.timeToDeath = Date.now();
129   this.deaths++;
130   this.health = 0;
131 
132   if (!bullet) {
133     killer = this; // Itsemurha
134     weapon = WPN.PISTOL;
135   } else {
136     killer = this.server.players[bullet.playerId];
137     weapon = bullet.weapon;
138   }
139 
140   // Onko tappaja vielä pelissä
141   if (killer.active) {
142     if (killer.playerId == this.playerId || (killer.team === this.team && this.server.gameState.playMode > 1)) {
143       // Teamkilleri tai itsemurha, vähennetään tappo.
144       killer.kills--;
145     } else {
146       // Tapettiin vastustaja, lisätään tappo.
147       killer.kills++;
148     }
149 
150     // Lähetetään tappoviesti
151     this.server.messages.addToAll({
152       msgType: NET.KILLMESSAGE,   // Mikä viesti
153       weapon: weapon,             // Millä aseella tapettiin
154       playerId: killer.playerId,  // Tappaja
155       playerId2: this.playerId    // Uhri
156     });
157   }
158 }

160 /**
161  * Vahingoittaa pelaajaa räjähdyksen etäisyyden arvoisesti
162  *
163  * @param {Bullet} bullet  Ammus, joka räjähtää pelaajan lähellä
164  * @param {Number} dist    Räjähdyksen etäisyys pelaajasta
165  */
166 Player.prototype.applyExplosion = function (bullet, dist) {
167   if (this.health <= 0 || this.isDead) {
168     return;
169   }
170   var damageRange = Weapons[bullet.weapon].damageRange;
171 
172   if (this.server.debug) {
173     log.info('Applying explosion from bullet %0 (%1) to %2',
174       String(bullet.bulletId).magenta, Weapons[bullet.weapon].name.yellow, this.name.green);
175   }
176 
177   // Uhrille tieto ampujasta
178   this.shootedBy = bullet.playerId;
179 
180   // Lasketaan vahingon määrä, joka riippuu etäisyydestä räjähdykseen, räjähdyksen laajuudesta ja
181   // räjähtäneen ammuksen damage-kentän arvosta.
182   this.health -= ((damageRange - dist) / damageRange) * Weapons[bullet.weapon].damage;
183 
184   // Tarkistetaan kuolema
185   if (this.health <= 0) {
186     this.kill(bullet);
187   }
188 }
