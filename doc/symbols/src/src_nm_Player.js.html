 13 /**
 14  * Luo uuden pelaajan.
 15  *
 16  * @class Kaikki pelaajat, niin ihmispelaajat kuin botitkin, ovat tämän luokan jäseniä.
 17  *
 18  * @param {Server} server  NetMatch-palvelin, joka pyörittää pelaajaa
 19  * @param {Byte} playerId  Luotavan pelaajan ID
 20  *
 21  * @extends Obj
 22  *
 23  * @property {Integer} id            Pelaajan tunnus välillä 1-MAX_PLAYERS
 24  * @property {String}  clientId      ClientId eli ip-osoite ja tunnus
 25  * @property {Integer} active        Onko tämä pelaaja pelissä
 26  * @property {Integer} loggedIn      Pelaaja on kirjautunut
 27  * @property {String}  name          Pelaajan nimimerkki
 28  * @property {String}  statName      Pelaajan nimimerkki
 29  * @property {String}  botName       Botin nimi
 30  * @property {Integer} lastActivity  Koska viimeksi pelaajalta on saatu dataa
 31  * @property {Integer} x             Sijainti
 32  * @property {Integer} y             Sijainti
 33  * @property {Integer} angle         Kulma
 34  * @property {Integer} passCount     Kunka monta kierrosta pelaaja on ollut päivittämättä
 35  * @property {Float}   prevPosX      Pelaajan edellinen sijainti
 36  * @property {Integer} lastValidX    Viimeisin paikka jossa pelaaja on varmasti ollut kentällä
 37  * @property {Integer} lastValidY    Viimeisin paikka jossa pelaaja on varmasti ollut kentällä
 38  * @property {Integer} weapon        Käytössä oleva ase
 39  * @property {Integer} lastShoot     Koska on viimeksi ammuttu
 40  * @property {Integer} health        Terveys
 41  * @property {Integer} prevHealth    Terveys edellisellä kierroksella
 42  * @property {Integer} kills         Tappojen lukumäärä
 43  * @property {Integer} deaths        Kuolemien lukumäärä
 44  * @property {Float}   killRatio     Tapposuhde
 45  * @property {Boolean} isDead        Onko pelaaja kuollut
 46  * @property {Integer} timeToDeath   Kuolinaika
 47  * @property {Integer} zombie        Onko tämä pelaaja botti
 48  * @property {Player}  shootedBy     Kuka on viimeksi ampunut pelaajaa
 49  * @property {Integer} hasAmmos      Onko pelaajalla ammuksia nykyisessä aseessa
 50  * @property {Byte}    team          Joukkue
 51  * @property {String}  mapName       Pelaajalla ladattuna oleva kartta
 52  * @property {Integer} outOfMap      Kuinka monta päivityskierrosta pelaaja on ollut poissa kartalta
 53  * @property {Integer} spawnTime     Syntymäaika
 54  * @property {Integer} visible       Ukon vilkuttaminen syntymän jälkeen
 55  * @property {Byte}    admin         Onko pelaaja admin vai ei
 56  * @property {Byte}    kicked        Onko pelaaja potkittu pois
 57  * @property {String}  kickReason    Mikä syy on annettu potkuille
 58  * @property {Player}  kicker        Kuka potkaisi
 59  * @property {Integer} hackTestX     Pelaajan viimeisin sijainti nopeuden tarkistuksessa
 60  * @property {Integer} hackTestY     Pelaajan viimeisin sijainti nopeuden tarkistuksessa
 61  * @property {Integer} lag           Pelaajan lagi millisekunneissa
 62  * @property {Integer} spHackTimer   Viimeisimmässä nopeuden tarkistuksessa otettu Timer()
 63  * @property {Byte}    handShooted   Kummalla kädellä on viimeksi ammuttu (pistooli) 0=vasen 1=oikea
 64  * @property {Boolean} sendNames     Pelaaja pyysi palvelimelta pelaajien nimilistauksen, lähetään kivasti myös itemien tiedot ":D"
 65  */
 66 function Player(server, playerId) {
 67   Obj.call(this, 0, 0, 0);
 68   this.server = server;
 69 
 70   // Alustetaan pelaaja
 71   this.id = playerId;
 72   this.team = 1;
 73   this.botName = server.gameState.map.config.botNames[playerId-1];
 74   if ('undefined' === typeof this.botName) {
 75     // Bottien nimiä ei ollut kaikkia määritelty kartan tiedoissa
 76     this.botName = "Bot_" + playerId;
 77   }
 78   this.clientId = "";
 79   this.name = "";
 80   var skill = 21 - playerId; // Botteja hieman eritasoisiksi vissiinkin?
 81   this.fightRotate = 1.5 + (skill / 1.5);
 82   this.shootingAngle = 4.0 + (playerId * 1.5);
 83   this.fov = 100 + (skill * 3.5);
 84   this.kickReason = "";
 85 
 86   // Lisätään tämä palvelimen players-kokoelmaan ja poistetaan vanha, jos sellainen oli olemassa.
 87   if (server.players.hasOwnProperty(playerId)) {
 88     delete server.players[playerId];
 89   }
 90 }
 91 Player.prototype = new Obj();
 92 Player.prototype.constructor = Player;

 94 /**
 95  * Tappaa pelaajan.
 96  * @param {Bullet} [bullet]  Tappajapelaajan ammus, joka hoiti viimeistelyn. Jos tätä ei anneta,
 97  *                           uskotellaan että pelaaja teki itsemurhan pistoolilla.
 98  */
 99 Player.prototype.kill = function (bullet) {
100   var killer, weapon;
101 
102   // Logataan kuka tappoi kenet.
103   if (bullet && this.server.players[bullet.player.id].name !== this.name) {
104     log.info('%0 was killed by player %1', this.name.green,
105       this.server.players[bullet.player.id].name.green, String(bullet.id).magenta);
106   } else {
107     log.info('%0 committed suicide.', this.name.green);
108   }
109 
110   this.isDead = true;
111   this.timeToDeath = Date.now();
112   this.deaths++;
113   this.health = 0;
114 
115   if (!bullet) {
116     killer = this; // Itsemurha
117     weapon = WPN.PISTOL;
118   } else {
119     killer = this.server.players[bullet.player.id];
120     weapon = bullet.weapon;
121   }
122 
123   // Onko tappaja vielä pelissä
124   if (killer.active) {
125     if (killer.id === this.id || (killer.team === this.team && this.server.gameState.playMode > 1)) {
126       // Teamkilleri tai itsemurha, vähennetään tappo.
127       killer.kills--;
128     } else {
129       // Tapettiin vastustaja, lisätään tappo.
130       killer.kills++;
131     }
132 
133     // Lähetetään tappoviesti
134     this.server.messages.addToAll({
135       msgType: NET.KILLMESSAGE, // Mikä viesti
136       weapon: weapon,           // Millä aseella tapettiin
137       player: killer,           // Tappaja
138       player2: this             // Uhri
139     });
140   }
141 };

143 /**
144  * Vahingoittaa pelaajaa räjähdyksen etäisyyden arvoisesti
145  *
146  * @param {Bullet} bullet  Ammus, joka räjähtää pelaajan lähellä
147  * @param {Number} dist    Räjähdyksen etäisyys pelaajasta
148  */
149 Player.prototype.applyExplosion = function (bullet, dist) {
150   if (this.health <= 0 || this.isDead) {
151     return;
152   }
153   var damageRange = Weapons[bullet.weapon].damageRange;
154 
155   log.debug('Applying explosion from %0 (%1) to %2',
156     String(bullet.id).magenta, Weapons[bullet.weapon].name.yellow, this.name.green);
157 
158   // Uhrille tieto ampujasta
159   this.shootedBy = bullet.player;
160 
161   // Lasketaan vahingon määrä, joka riippuu etäisyydestä räjähdykseen, räjähdyksen laajuudesta ja
162   // räjähtäneen ammuksen damage-kentän arvosta.
163   this.health -= ((damageRange - dist) / damageRange) * Weapons[bullet.weapon].damage;
164 
165   // Tarkistetaan kuolema
166   if (this.health <= 0) {
167     this.kill(bullet);
168   }
169 };

171 /**
172  * Tätä funktiota kutsutaan kun ammus osuu suoraan pelaajaan. Tämä hoitaa pelaajan vahingoittamisen.
173  *
174  * @param {Bullet} bullet  Ammus joka osui pelaajaan
175  * @param {Number} x       Osumakohta
176  * @param {Number} y       Osumakohta
177  */
178 Player.prototype.bulletHit = function (bullet, x, y) {
179   if (this.health <= 0 || this.isDead) {
180     return;
181   }
182   // Talletetaan tieto ampujasta
183   this.shootedBy = bullet.player.id;
184 
185   // Lisätään viestijonoon ilmoitus osumasta
186   this.server.messages.addToAll({
187     msgType: NET.BULLETHIT, // Mikä viesti
188     bullet: bullet,         // Ammuksen tunnus
189     player: this,           // Keneen osui
190     x: x,                   // Missä osui
191     y: y,                   // Missä osui
192     weapon: bullet.weapon   // Millä aseella ammus ammuttiin
193   });
194 
195   // Tutkitaan oliko räjähdys ja jos oli, niin meillä ei ole täällä enää muuta tehtävää.
196   if (bullet.checkExplosion(x, y)) {
197     return;
198   }
199 
200   // Ei ollut räjähdys, tiputetaan pelaajan health-kentän arvoa
201   this.health -= Weapons[bullet.weapon].damage;
202 
203   // Kuolema?
204   if (this.health <= 0) {
205     this.kill(bullet);
206   }
207 };
