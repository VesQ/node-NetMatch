  5 var log = require('../Utils').log;
  6 
  7 /**
  8  * Tekee pelaajasta adminin.
  9  * @methodOf Commands
 10  *
 11  * @param {player} who  Kenet opataan
 12  */
 13 var op = {
 14   /**#nocode+*/
 15   params: [
 16     {name: 'who',  type: 'player', optional: false, help: 'Player who to op'}
 17   ],
 18   help: 'Makes a player admin. Like server-side login.',
 19   remote: true,
 20   action: function commandsOp() {
 21     var server = this
 22       , caller = arguments[0]
 23       , player = this.getPlayer(arguments[1])
 24       // Vastaa konsoliin tai pelaajalle, jos ilmenee ongelmia.
 25       , reply = function (s) {
 26         if (caller) { server.serverMessage(s, caller); }
 27         else        { log.warn(s); }
 28       };
 29 
 30     if (!player) {
 31       reply('Couldn\'t find player!');
 32       return;
 33     }
 34 
 35     if (player.admin) {
 36       reply(player.name + ' is already an admin!');
 37       return;
 38     }
 39 
 40     player.admin = true;
 41     this.serverMessage('You are now an admin!', player);
 42     // Kerrotaan kutsujalle my√∂s
 43     if (caller) { this.serverMessage('Done! :)', caller); }
 44   }
 45   /**#nocode-*/
 46 };
 47 
 48 module.exports = op;
