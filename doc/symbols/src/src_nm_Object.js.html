1 /**
  2  * @fileOverview Sisältää {@link Obj}-luokan toteutuksen.
  3  */
  4 
  5 /**#nocode+*/
  6 var wrapAngle = require('./Utils').wrapAngle;
  7 /**#nocode-*/
  8 
  9 /**
 10  * Luo uuden objektin.
 11  * @class Objekti
 12  *
 13  * @param {Number} x      X-koordinaatti
 14  * @param {Number} y      Y-koordinaatti
 15  * @param {Number} angle  Kulma
 16  *
 17  * @property {Number} x      X-koordinaatti
 18  * @property {Number} y      Y-koordinaatti
 19  * @property {Number} angle  Kulma
 20  */
 21 function Obj(x, y, angle) {
 22   this.x = x;
 23   this.y = y;
 24   this.angle = angle;
 25 }
 26 
 27 /**
 28  * Siirtää objektia, ottaen huomioon nykyisen kulman.
 29  * @param {Number} px        Kuinka monta pikseliä objektia siirretään pituussuunnassa
 30  * @param {Number} [sidePx]  Kuinka monta pikseliä objektia siirretään leveyssuunnassa
 31  */
 32 Obj.prototype.move = function (px, sidePx) {
 33   if (px !== 0) {
 34     this.x += Math.cos((this.angle / 180) * Math.PI) * px;
 35     this.y += Math.sin((this.angle / 180) * Math.PI) * px;
 36   }
 37   if ('number' === typeof sidePx && sidePx !== 0) {
 38     this.x += Math.cos((this.angle - 90) / 180 * Math.PI) * sidePx;
 39     this.y += Math.sin((this.angle - 90) / 180 * Math.PI) * sidePx;
 40   }
 41 };
 42 
 43 /**
 44  * Kääntää objektia, pitäen kulman kuitenkin välillä 0-360
 45  * @param {Number} angle  Kuinka monta astetta objektia käännetään
 46  */
 47 Obj.prototype.turn = function (angle) {
 48   this.angle = wrapAngle(this.angle + angle);
 49 };
 50 
 51 module.exports = Obj;
